\exercice{848, bodin, 1998/09/01}

\enonce
Trouver les solutions r\'eelles des \'equations diff\'erentielles
suivantes :
\begin{enumerate}
    \item $y'(t)+2y(t) = 0$ ;
    \item $\displaystyle{\frac{dx}{dt}-x=0}$ ;
    \item $y'(x)+2y(x) = 0 \text{ avec } (y-y')(0)=0$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
