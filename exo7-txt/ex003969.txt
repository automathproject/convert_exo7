\exercice{3969, quercia, 2010/03/11}
\enonce[Dérivées $n$-èmes de $t^{n-1}\ln(t)$ et $t^{n-1}e^{1/t}$]

Calculer $\frac {d^n}{d t^n}\left(t^{n-1}\ln t\right)$, et
$\frac {d^n}{d t^n}\left(t^{n-1}\exp(1/t)\right)$
(essayer $n=1,2,3$).
\finenonce

\noindication

\correction
$\frac {(n-1)!}t$,
  \qquad
    $\frac {(-1)^n}{t^{n+1}}\exp(1/t)$.
 \fincorrection


\finexercice
