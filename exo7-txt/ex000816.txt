\exercice{816, ridde, 1999/11/01}

\enonce
Soit $I_{n} = \int_{0}^{1}{ (1-t^2)^n dt}$.
\begin{enumerate}
\item \'Etablir une relation de r\'ecurrence entre $I_{n}$ et $I_{n + 1}$.
\item Calculer $I_{n}$.
\item En d\'eduire $\sum\limits_{k = 0}^{n}{\frac{ (-1)^k}{2k + 1}C_{n}^{k}}$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
