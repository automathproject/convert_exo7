\exercice{33, bodin, 1998/09/01}

\enonce
Pour $z \in \Cc \setminus \{2i\}$, on pose
$$f(z)=\frac{2z-i}{z-2i}.$$
\begin{enumerate}
    \item R\'esoudre l'\'equation $z^2=i, \  z \in \Cc.$
    \item R\'esoudre l'\'equation $f(z)=z, \ z \in \Cc \setminus \{2i\}.$
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
