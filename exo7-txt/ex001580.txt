\exercice{1580, barraud, 2003/09/01}

\enonce
  \begin{enumerate}
  \item 
    Réduire la matrice 
    $$
    A=
    \begin{pmatrix}
      2&  0& 0\\
      3& -4& 3\\
      3& -6& 5
    \end{pmatrix}
    $$
  \item
    Donner un polynôme annulateur de $A$ de degré 2.
  \item
    En  déduire qu'il existe des coefficients $a_{n}$ et $b_{n}$ tels que
    $A^{n}=a_{n}A+b_{n}$ et les calculer en fonction de $n$.
  \end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
