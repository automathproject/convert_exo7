\exercice{6621, queffelec, 2011/10/16}
\enonce
Trouver le domaine maximal de convergence des séries entières suivantes :
$$(1)\ \sum_{n=1}^\infty {z^n\over n},\quad (2)\ \sum_{n=0}^\infty
(2z)^n\cos n\theta,\quad  (3)\ \sum_{n=0}^\infty
z^{n^2}.$$
\finenonce

\noindication

\nocorrection

\finexercice
