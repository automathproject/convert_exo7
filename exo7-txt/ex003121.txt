\exercice{3121, quercia, 2010/03/08}

\enonce[pgcd$(x,y) = x-y$, ppcm$(x,y) = 300$]
R{\'e}soudre dans $\N^*$ :
$\begin{cases}x \wedge y = x-y \cr x \vee y = 300.\end{cases}$
\finenonce

\noindication

\correction
$(300,150)$, $(150,100)$, $(100,75)$, $(75,60)$, $(60,50)$.
\fincorrection
\finexercice
