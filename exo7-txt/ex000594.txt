\exercice{594, ridde, 1999/11/01}

\enonce
\begin{enumerate}
\item Montrer que si $\forall n \in \Nn$ $u_n \neq 0$ et si $ (u_n) \rightarrow 0$
alors $ \ln (1 + u_n) \sim u_n$.
\item Soit $a$ un r\'eel. D\'eterminer la limite de $ (1 + \dfrac an)^n$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
