\exercice{1934, gineste, 2001/11/01}

\enonce[Comparaison à des séries de Riemann et équivalent]
Etudier les séries de termes généraux
\begin{enumerate}
\item $$ u_n=\cos(\frac{ \pi n^2 }{2n^2+an+1}) \mbox{ avec } a>0 $$
\item $$ v_n=e^{-\sqrt{n}} $$
\item $$ w_n=(1 - \frac{1}{n^2})^n $$
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
