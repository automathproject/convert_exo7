\exercice{2966, quercia, 2010/03/08}

\enonce[Essai de tables]
Les op{\'e}rations suivantes sont-elles des lois de groupe ?

\begin{enumerate}
  \item
    $$\begin{array}{c|ccc}
      &a &b &c \\
      \hline
      a &a &a &a \\
      b &a &b &b \\
      c &a &b &c 
    \end{array}
    $$
  \item
    $$\begin{array}{c|ccc}
      &a &b &c \\
      \hline
      a &b &c &a \\
      b &c &a &b \\
      c &a &b &c 
    \end{array}
    $$
  \item
    $$\begin{array}{c|cccc}
      &a &b &c &d \\
      \hline
      a &a &b &c &d \\
      b &b &a &d &c \\
      c &d &c &b &a \\
      d &c &d &a &b 
    \end{array}
    $$
\end{enumerate}
\finenonce

\noindication

\correction
\begin{enumerate}
 \item Non, $a$ n'est pas r{\'e}gulier.
 \item Oui, $G \approx \Z/3\Z$.
 \item Non, il n'y a pas d'{\'e}l{\'e}ment neutre.
\end{enumerate}

\fincorrection
\finexercice
