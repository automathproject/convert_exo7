\exercice{2900, quercia, 2010/03/08}

\enonce[Calcul de sommes]
Calculer $\sum_{k=0}^n\, kC_n^k$ et $\sum_{k=0}^n \frac{C_n^k}{k+1}$.
\finenonce

\noindication

\correction
 $\sum_{k=0}^n kC_n^k = n2^{n-1}$, \qquad $\sum_{k=0}^n
\frac{C_n^k}{k+1} = \frac {2^{n+1}-1}{n+1}$.  
\fincorrection

\finexercice
