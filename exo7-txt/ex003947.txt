\exercice{3947, quercia, 2010/03/11}
\enonce[Rolle itéré]

Soit $f : {[a,b]} \to \R$ $n$ fois dérivable.

\begin{enumerate}
  \item   Si $f$ s'annule en $n+1$ points distincts dans $[a,b]$,
      montrer qu'il existe $c \in {]a,b[}$ tel que $f^{(n)}(c) = 0$.
  \item   Si $f(a) = f'(a) = \dots = f^{(n-1)}(a) = f(b) = 0$,
      montrer qu'il existe $c \in {]a,b[}$ tel que $f^{(n)}(c) = 0$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
