\exercice{2335, matexo1, 2002/02/01}
\enonce
Trouver les primitives des fonctions suivantes\,:
$${
{1\over x \sqrt{x^2-1}}, \qquad 
{(4x^2+4x+5)^{-1/2}}, \qquad
{\sin^4 x\over \cos^2 x}, \qquad
{\arctan x \over 1+x^2}, \qquad
{1\over x\,(1+\ln^2 x)}. 
}$$
\finenonce
\noindication

\nocorrection

\finexercice
