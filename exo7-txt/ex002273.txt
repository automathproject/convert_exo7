\exercice{2273, barraud, 2008/04/24}
\enonce
 Trouver le $\pgcd(f,g)$ dans $\Zz_3[x]$ et 
$\Zz_5[x]$
de $f=x^4+1$,  $g=x^3+x+1$.
\finenonce 

\noindication

\correction
  Dans $\Zz/3\Zz$~: $\pgcd(x^{4}+1,x^{3}+x+1)=x^{2}+x-1$.

  \noindent 
  Dans $\Zz/5\Zz$~: $\pgcd(x^{4}+1,x^{3}+x+1)=1$.
\fincorrection
\finexercice
