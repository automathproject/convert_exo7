\exercice{4050, quercia, 2010/03/11}
\enonce[Développement asymptotique d'une réciproque]
Soit $f : {[-1,+\infty[} \to {[-e^{-1},+\infty[}, x \mapsto {xe^x.} $
Montrer que $f$ est bijective et $f^{-1}(x) \mathop{\sim}\limits_{+\infty} \ln(x)$.

\finenonce

\noindication

\nocorrection

\finexercice
