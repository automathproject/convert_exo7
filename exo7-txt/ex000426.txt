\exercice{426, cousquer, 2003/10/01}
\enonce
 Trouver un polyn\^ome~$P$ de degr\'e~$\leq2$ tel que 
$$P(1)=-2 \quad \mbox{et} \quad P(-2)=3 \quad \mbox{et} \quad P(0)=-1$$

\finenonce 
\noindication
\correction
 Utiliser la formule d'interpolation de Lagrange~!
$P={1\over3}(X^2-4X-3)$.
\fincorrection
\finexercice
