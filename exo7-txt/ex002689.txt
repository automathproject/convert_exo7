\exercice{2689, matexo1, 2002/02/01}
\enonce
Montrer que
$$\forall x\in\R,\qquad |\sin x| \leq  |x|,$$
$$\forall x\in[0,\frac \pi 2], \qquad 1-\cos x \leq  x \sin x,$$
$$\forall x\in[-1,1],\qquad 
|\arcsin x| \leq  \left| \frac x{\sqrt{1-x^2}} \right|.$$
\finenonce
\noindication
\nocorrection
\finexercice
