\exercice{2981, quercia, 2010/03/08}

\enonce[Morphismes de $\Q$ additif]
D{\'e}terminer tous les morphismes de
\begin{enumerate}
  \item $(\Q,+)$ dans $(\Q,+)$. 
  \item $(\Q,+)$ dans $(\Z,+)$. 
  \item $(\Q,+)$ dans $(\Q^*,\times)$. 
\end{enumerate}
\finenonce

\noindication

\correction
\begin{enumerate}
 \item $x \mapsto ax$, $a\in\Q$.
 \item $x \mapsto 0$.
 \item  $x \mapsto 1$.
\end{enumerate}

\fincorrection

\finexercice
