\exercice{2777, tumpach, 2009/10/25}
\enonce
\begin{enumerate}
\item
Calculer les valeurs propres et les vecteurs propres de la matrice 
$$
A = \left(\begin{array}{ccc}0 & 2 & -2\\1 & -1 & 2\\1 & -3 & 4\end{array}\right).
$$
\item Calculer $A^n$ pour tout $n\in\mathbb{N}$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
