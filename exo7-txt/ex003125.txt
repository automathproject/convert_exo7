\exercice{3125, quercia, 2010/03/08}

\enonce[{\'E}quations {\`a} coefficients entiers]

R{\'e}soudre dans $\Z$ :
\begin{enumerate}
  \item   $95x + 71y = 46$.
  \item   $20x - 53y = 3$.
  \item   $12x + 15y + 20z = 7$.
\end{enumerate}
\finenonce

\noindication

\correction
\begin{enumerate}
  \item $x = 67-71k,\quad y = -89 + 95k$.
  \item $x = 24 + 53k,\quad y = 9 + 20k$.
  \item $x = -49 + 20k - 5m,\quad y = 49 - 20k + 4m,\quad z = -7 + 3k$.

\end{enumerate}
\fincorrection
\finexercice
