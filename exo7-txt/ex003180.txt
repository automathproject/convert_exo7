\exercice{3180, quercia, 2010/03/08}

\enonce[{\'E}quation $X(X-1)P' + P^2 -(2X+1)P + 2X = 0$]
Trouver tous les polyn{\^o}mes $P \in { K[X]}$ tels que :
$X(X-1)P' + P^2 -(2X+1)P + 2X = 0$.
\finenonce

\noindication

\correction
$\deg P < 2  \Rightarrow  P \in \{1,X,X+1\}$.
\fincorrection
\finexercice
