\exercice{274, cousquer, 2003/10/01}

\enonce
 Montrer que  pour  tout $n> 0$ :
\begin{enumerate}
  \item  $7 \hbox{ divise } 3^{2n+1}+2^{n+2}$
  \item   $11\hbox{ divise }2^{6n+3} +3^{2n+1}$
  \item   $6\hbox{ divise }5n^3+n$
  \item   $8 \hbox{ divise } 5^n +2.3^{n-1} +1$ .
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
