\exercice{6311, mayer, 2011/10/16}
\enonce 
\begin{enumerate}
\item Déterminer les extremums de la fonction $f(x,y) =xy$ sur le
cercle unité $S= \{(,x,y) \in \Rr^2 \; ; x^2+y^2=1\}$.
\item Même question pour la fonction $f(x,y) = xy^2$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
