\exercice{2244, barraud, 2008/04/24}
\enonce
Trouver toutes les solutions dans $\Zz$ :
\begin{enumerate}
  \item $\quad 2x+3 \equiv 10 \mod 13$;
  \item $ \quad \begin{cases}
2x+3y\equiv 5 \mod 7\\
5x+2y \equiv 2 \mod 7;
\end{cases}
$
  \item $\quad x^2 +2x+14 \equiv 0 \mod 17$.
\end{enumerate}
\finenonce 

\noindication

\nocorrection
\finexercice
