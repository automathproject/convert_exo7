\exercice{170, cousquer, 2003/10/01}

\enonce
\begin{enumerate}
\item  Démontrer par récurrence:
$$ \sum_{k=0}^n k = \frac{n(n+1)}{2} $$
\item  Calculer de deux manières différentes:
$$ \sum_{k=1}^{n+1} k^3 - \sum_{k=0}^n (k+1)^3. $$
\item  En déduire:
 $$ \sum_{k=0}^n k^2 = {1 \over 6} (2n^3+3n^2+3n). $$
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
