\exercice{1885, maillot, 2001/09/01}

\enonce
Soit $E$ un espace vectoriel norm√©, et $(a,a')\in E^2$,
$(r,r')\in(\R_+^*)^2$. Montrer :
\begin{enumerate}
\item
$B(a,r)=\{a\}+B(0,r)$
\item
$B(a,r)=B(a',r') \Leftrightarrow a=a' \text{ et } r=r'$
\item
$B(a+a',r+r')=B(a,r)+B(a',r')$
\item
$B(a,r)\cap B(a',r')\neq\emptyset\Leftrightarrow\|{a'-a}\|{}<r+r'$.
\end{enumerate}
\finenonce

\noindication
