\exercice{2452, matexo1, 2002/02/01}

\enonce
Calculer par r\'ecurrence le d\'eterminant
$$ \Delta_n = \left|
\begin{array}{ccccc}
a_1+b_1 & b_1     & b_1 & \ldots & b_1 \\
   b_2  & a_2+b_2 & b_2 & \ldots & b_2 \\
  \ldots&\ldots   &\ldots&\ldots&\ldots\\
   b_n  &  b_n    &\ldots&\ldots& a_n+b_n
\end{array}\right|.$$
 
\finenonce

\noindication
\nocorrection

\finexercice
