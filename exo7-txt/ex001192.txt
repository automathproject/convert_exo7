\exercice{1192, barraud, 2003/09/01}

\enonce
  Sans chercher à le résoudre, discuter la nature des solutions du
  système suivant, en fonction de $\alpha,a,b$ et $c$~:
  $$
  \left\{
    \begin{array}{*8{c@{\;}}}
      x&-& y&-&\alpha z&=&a\\
      x&+&2y&+& z&=&b\\
      x&+& y&-& z&=&c
    \end{array}
    \right.
  $$
\finenonce

\noindication

\nocorrection

\finexercice
