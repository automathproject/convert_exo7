\exercice{6586, hueb, 2011/10/16}
\enonce Evaluer
$$\int_{1+i}^{2+4i} z^2 dz$$
\begin{enumerate}
\item le long de la parabole $y = x^2,\ 1 \leq x \leq 2$;
\item le long du segment de droite $(1+i), (2+4i)$;
\item le long des segments $(1+i), (2+i)$ et $(2+i),(2+4i)$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
