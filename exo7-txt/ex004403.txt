\exercice{4403, quercia, 2010/03/12}
\enonce[$\cos z$]
Quels sont les complexes $z$ tels que $\cos z \in {[-1,1]}$ ?
\finenonce

\noindication

\correction
$\cos(x+iy) = \cos x \ch y - i\sin x \sh y  \Rightarrow  \cos z \in {[-1,1]}$
          si et seulement si $z \in \R$.
\fincorrection

\finexercice
