\exercice{3226, quercia, 2010/03/08}

\enonce[Factorisation]
Factoriser $$1 -\frac X{1!} + \frac {X(X-1)}{2!} - \dots +
            (-1)^{n+1}\frac {X(X-1)\cdots(X-n)}{(n+1)!}.$$
\finenonce

\noindication

\correction
$ = (-1)^{n+1}\frac {(X-1)\cdots(X-n)\bigl(X-(n+1)\bigr)}{(n+1)!}$.
\fincorrection
\finexercice
