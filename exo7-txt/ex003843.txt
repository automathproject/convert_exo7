\exercice{3843, quercia, 2010/03/11}
\enonce[Ulm MP$^*$ 2000]
$V$ est un espace hermitien et $u,v,w$ trois vecteurs unitaires.
Montrer que~:
$$\sqrt{1-|(u\mid v)|^2} \le \sqrt{1-|(u\mid w)|^2} + \sqrt{1-|(v\mid w)|^2}.$$

\finenonce

\noindication

\correction
$\sqrt{1-|(u\mid v)|^2} = d(v,\C u) \le d(v,\C w) + |(v\mid w)|d(w,\C u)$.
\fincorrection

\finexercice
