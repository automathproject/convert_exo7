\exercice{5528, rouget, 2010/07/15}
\enonce
Trouver une équation cartésienne des supports des arcs suivants~:

\begin{enumerate}
 \item $\left\{\begin{array}{l}
x=t^2\\
y=-t^2
\end{array}\right.$
 \item $\left\{\begin{array}{l}
x=t^2\\
y=t^3
\end{array}\right.$
  \item $\left\{\begin{array}{l}
x=\frac{t}{1+t^4}\\
y=\frac{t^3}{1+t^4}
\end{array}\right.$
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
