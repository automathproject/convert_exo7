\exercice{1076, gourio, 2001/09/01}

\enonce
Calculer les puissances de :
$$\left(
\begin{array}{ll}
a & b \\
0 & a
\end{array}
\right) ,\ \left(
\begin{array}{ll}
a & b \\
b & a
\end{array}
\right) ,\ \left(
\begin{array}{lll}
1 & 1 & 1 \\
0 & 1 & 1 \\
0 & 0 & 1
\end{array}
\right) .
$$
\finenonce

\noindication

\nocorrection

\finexercice
