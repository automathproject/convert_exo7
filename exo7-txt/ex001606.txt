\exercice{1606, barraud, 2003/09/01}

\enonce
Soit $J$ la matrice $$J=
\begin{pmatrix}
  1 &\cdots &1\\
  \vdots&   &\vdots\\
  1 &\cdots &1
\end{pmatrix}
$$.
\begin{enumerate}
\item 
Trouver une relation entre $J$ et  $J^{2}$.
\item
En déduire les valeurs propres de $J$ et calculer leurs multiplicités.
\item
Donner le polynôme caractéristique de $J$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
