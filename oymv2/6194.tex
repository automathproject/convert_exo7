\texte{
Soit $E$ un espace normé, $X$ et $Y$ deux sous-ensembles de $E$. Montrer que
}
\begin{enumerate}
    \item \question{$X+Y$ est ouvert si $X$ est ouvert;}
    \item \question{$X+Y$ est compact si $X$ et $Y$ sont compacts;}
    \item \question{$X+Y$ est fermé si $X$ est compact et $Y$ fermé.}
\end{enumerate}
}
