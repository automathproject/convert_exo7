\texte{
Montrer que les nombres suivants ne sont pas premiers :
}
\begin{enumerate}
    \item \question{$n^4-20n^2 + 4$ pour $n \in \Nn$.}
    \item \question{$\frac14 (n^3 + (n + 2)^3)$ pour $n \geq 2$.}
    \item \question{$a^4 + 4b^4$ pour $a, b \geq 2$.}
\end{enumerate}
}
