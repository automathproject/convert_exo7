\begin{enumerate}
    \item \question{Calculer la transformée de Fourier de la fonction porte
  symétrique
  $$f(x)= \begin{cases}
    A,\quad & -\frac \tau 2<x<\frac \tau 2\\
    0,\quad & |x| > \frac \tau 2
\end{cases}.$$}
    \item \question{Vérifier que $f$ satisfait aux conditions de Dirichlet.
  Comment doit $f$ être définie en $x=\pm \frac \tau 2$ pour que
  l'intégrale de Fourier converge vers $f(x)$ pour tout $x$?}
\end{enumerate}
