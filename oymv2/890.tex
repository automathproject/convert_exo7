\contenu{
\texte{
D\'eterminer si ${ \Rr}^{2}$, muni des lois internes et externes suivantes, est
ou n'est pas un ${ \Rr}$-espace vectoriel :
}
\begin{enumerate}
    \item \question{$(a,b)+(c,d)=(a+c,b+d);  \lambda(a,b) =(a,\lambda b),   \lambda \in { \Rr}.  $}
    \item \question{$ (a,b)+(c,d)=(a+c,b+d);  \lambda(a,b) = (\lambda^{2}a,\lambda^{2} b),    \lambda \in { \Rr}.  $}
    \item \question{$(a,b)+(c,d)=(c,d);  \lambda(a,b) = (\lambda a,\lambda b),    \lambda \in { \Rr}. $}
\end{enumerate}
}
