\contenu{
\texte{
\'Etudier l'existence de solutions du système :

 $$\left\{\begin{array}{ccccccc} x &-& my   &+& m^2z &=& m   \cr
           mx &-& m^2y &+& mz   &=& 1   \cr
           mx &+& y    &-& m^3z &=& -1. \cr\end{array}\right.$$
}
\reponse{
Système de Cramer ssi $m\ne 0,\pm1,\pm i$; compatible ssi $m\ne 0,\pm i$.
}
}
