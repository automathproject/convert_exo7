\begin{enumerate}
    \item \question{Montrer que sur $ M_{n}({\Rr})$ l'application :
$$\left( A,B\right) \rightarrow tr(^{t}AB) $$
est un produit scalaire.}
    \item \question{Soit $ N$ la norme associ\'{e}e, montrer que :
$$\forall (A,B)\in M_{n}({\Rr}),  N(AB)\leq N(A)N(B). $$}
    \item \question{Montrer que :
$$\forall A\in M_{n}({\Rr}),\left| tr(A)\right| \leq \sqrt{n}N(A). $$}
\end{enumerate}
