\begin{enumerate}
    \item \question{Montrer que $\int_0^{2\pi} e^{re^{it}}\ dt$ est indépendant de
$r>0$.}
    \item \question{Montrer que $\int_0^{2\pi} \mathrm{Log}{(1-re^{it})}\ dt$ est indépendant de
$r\in \mathopen]0,1\mathclose[$ ($\mathrm{Log}$ désigne la branche principale
du logarithme sur $\C\setminus \mathopen] -\infty,0]$).}
\end{enumerate}
