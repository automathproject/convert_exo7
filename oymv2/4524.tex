\contenu{
\texte{
Soit $(f_n)$ une suite de fonctions continues ${[a,b]}  \to {\R}$
convergeant simplement vers une fonction continue $f$.
}
\begin{enumerate}
    \item \question{On suppose que chaque fonction $f_n$ est croissante. Montrer qu'il y a convergence uniforme.}
    \item \question{On suppose qu'à $x$ fixé la suite $(f_n(x))$ est croissante. Montrer qu'il y a convergence uniforme.}
\end{enumerate}
}
