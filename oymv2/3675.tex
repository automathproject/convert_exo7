\contenu{
\texte{
Soient $E$ espace vectoriel euclidien, $\vec a \in E$ et $\alpha,\beta,\gamma \in \R$.
Résoudre l'équation $\alpha (\vec x\mid \vec x) + \beta (\vec x\mid \vec a) + \gamma = 0$.
}
\reponse{
sphère de centre $-\frac {\gamma \vec a}{\beta \|\vec a\,\|^2}$.
}
}
