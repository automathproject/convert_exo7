\texte{
Soient $P,Q$ deux polyn\^omes premiers entre eux.
}
\begin{enumerate}
    \item \question{Montrer qu'alors $P^n$ et $Q^m$ sont premiers entre eux
     o\`u $n,m$ sont deux entiers positifs.}
    \item \question{Montrer de m\^eme que $P+Q$ et $PQ$ sont premiers entre eux.}
\end{enumerate}
}
