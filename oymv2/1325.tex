\texte{
Les questions sont ind\'ependantes. Soit $j$ le
nombre complexe $e^{\frac {2i\pi}{3}}.$
}
\begin{enumerate}
    \item \question{D\'eterminer le sous-goupe du groupe additif $\Cc$ engendr\'e par $i$ et $j$.}
    \item \question{D\'eterminer le sous-goupe du groupe multiplicatif
$\Cc^{\ast}$ engendr\'e par $i$ et $j$.}
\end{enumerate}
