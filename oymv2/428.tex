\chapitre{Polynôme, fraction rationnelle}
\sousChapitre{Autre}

\contenu{
\texte{
Trouver les polynômes $P$ de $\Rr[X]$ tels que $\forall k \in \Zz \, \,
\int_{k}^{k + 1}P (t)dt  = k + 1$ (on pourra utiliser le polynôme
$Q (x) = \int_0^x P (t)dt$).
}
}
