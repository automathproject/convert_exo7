\texte{
Soit $\sigma$ un automorphisme de $\Rr$.
}
\begin{enumerate}
    \item \question{Montrer que si $x \geq 0$ alors $\sigma (x) \geq 0$.}
    \item \question{Montrer que $\sigma$ est croissante.}
    \item \question{D\'eterminer $\sigma$.}
\end{enumerate}
}
