\texte{
Pour chacune des fonctions \(f\), déterminer l'ensemble de définition de \(f\) et calculer \(\frac{\partial f}{\partial z}\) et \(\frac{\partial f}{\partial \bar z}\).
}
\begin{enumerate}
    \item \question{\(f(z)=z^2\bar z\)}
    \item \question{\(f(z)=e^{\bar z}\)}
    \item \question{\(f(z)=\frac{\bar z-i}{z^2+iz-2}\)}
    \item \question{\(f(z)=\frac{\Re(z)}{\Im(z)}\)}
\end{enumerate}
}
