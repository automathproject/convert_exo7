\texte{
Soient les fonctions $f:x\mapsto\arcsin(\sin x)$ et
$g:x\mapsto\arctan\sqrt{\frac{1-\cos x}{1+\cos x}}.$
}
\begin{enumerate}
    \item \question{Simplifier les expressions de $f(x)$ et $g(x)$.}
    \item \question{Construire les graphes de $f$ et $g$.}
\end{enumerate}
}
