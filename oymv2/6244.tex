\texte{
Parmi les métriques suivantes définies sur $\Rr$, lesquelles sont
uniformément équivalentes à la métrique usuelle ?
}
\begin{enumerate}
    \item \question{$\vert x^3-y^3\vert$}
    \item \question{$\vert \arctan x-\arctan y\vert$}
    \item \question{${{\vert x-y\vert}\over{1+\vert x-y\vert}}$}
\end{enumerate}
}
