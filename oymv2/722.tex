\texte{
Appliquer la formule des accroissements finis \`a la
fonction
$$f(x)=a+bx+ce^{\alpha x}$$
(o√π $a,b,c,\alpha$ sont r\'eels,
et $c$ et $\alpha$ sont non nuls)
sur l'intervalle $[0,X]$.
}
\begin{enumerate}
    \item \question{Calculer ``$\theta$'' en fonction de $X$.}
    \item \question{En d\'eduire que
$$x\mapsto \frac{1}{\alpha x}\ln \frac{e^{\alpha x}-1}{\alpha x}$$
est born\'ee sur $\Rr$.}
\end{enumerate}
