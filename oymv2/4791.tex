\contenu{
\texte{
Soit $f : {\R^2} \to \R$ continue.
On pose $g(x) = \sup(f(x,y) \text{ tq } y \in {[0,1]})$.
Montrer que $g$ est continue.
}
\reponse{
Remarquer que la restriction de $f$ {\`a} toute partie compacte est uniform{\'e}ment
continue.
}
}
