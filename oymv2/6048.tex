\texte{
Etablir l'équivalence entre les propriétés suivantes :
}
\begin{enumerate}
    \item \question{$\stackrel{\circ}{A}$ est le plus grand ouvert contenu dans $A$.}
    \item \question{$a\in\stackrel{\circ} A$ si et seulement si il existe un voisinage de $a$
entièrement contenu dans $A$.}
\end{enumerate}
}
