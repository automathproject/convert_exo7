\texte{
Calculer par l'algorithme d'Euclide : $\pgcd(18480,9828)$.
En d\'eduire une \'ecriture de $84$ comme combinaison lin\'eaire de $18480$ et
$9828$.
}
\reponse{
$\pgcd(18480,9828) = 84$;
$25 \times 18480 + (-47) \times 9828 = 84$.
}
}
