\contenu{
\texte{
Montrer que l'ensemble $\{2^n \hskip 2pt | \hskip 2pt n\in \Z \} $ est un sous-groupe
multiplicatif de $\Q^{*}$, ainsi que l'ensemble 
$\{\frac{1+2m}{1+2n}  \hskip 2pt | \hskip 2pt  n,m\in \Z \}$.
}
\indication{Aucune difficult\'e.}
}
