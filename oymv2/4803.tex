\contenu{
\texte{
Soit $A\subset E$ une partie non vide born{\'e}e et $f : A  \to F$ uniform{\'e}ment continue.
Montrer que $f$ est born{\'e}e dans les cas suivants~:
}
\begin{enumerate}
    \item \question{$A$ est convexe.}
    \item \question{$A$ est connexe par arcs.}
    \item \question{$A$ est quelconque et $E$ est de dimension finie.}
\end{enumerate}
}
