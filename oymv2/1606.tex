\contenu{
\texte{
Soit $J$ la matrice $$J=
\begin{pmatrix}
  1 &\cdots &1\\
  \vdots&   &\vdots\\
  1 &\cdots &1
\end{pmatrix}
$$.
}
\begin{enumerate}
    \item \question{Trouver une relation entre $J$ et  $J^{2}$.}
    \item \question{En déduire les valeurs propres de $J$ et calculer leurs multiplicités.}
    \item \question{Donner le polynôme caractéristique de $J$.}
\end{enumerate}
}
