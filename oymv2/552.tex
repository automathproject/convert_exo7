\texte{
Étudier suivant les valeurs données à $u_0$ appartenant à $\mathbb{C}$ les
suites~:
\begin{eqnarray*}
  u_{n+1} & = & \frac{u_n-2}{u_n +4}\\
  u_{n+1} & = & \frac{u_n+2}{u_n+1}\\
  u_{n+1} & = & \frac{-1}{u_n+1}
\end{eqnarray*}
}
