\begin{enumerate}
    \item \question{(Théorème d'Euler) Soit $n\in \N^*$ et $a\in \Z^*$ premier avec $n$. Démontrer que $a^{\phi(n)}\equiv 1$ (mod $n$), où $\phi$ est la fonction d'Euler.}
    \item \question{(Théorème de Fermat) Soit $p$ un nombre premier et $a\in \Z$. Montrer que $a^p\equiv a $(mod $p$).}
\end{enumerate}
