\begin{enumerate}
    \item \question{Soit $D = \text{diag}(\lambda_1,\dots,\lambda_n)$ une matrice diagonale
    à valeurs propres distinctes.
  \begin{enumerate}}
    \item \question{Montrer qu'une matrice $M$ commute avec $D$ si et seulement si $M$ est
    diagonale.}
    \item \question{Montrer que pour toute matrice $M$ diagonale, il existe un polynôme
    $P \in  K_{n-1}[X]$ unique tel que $M = P(D)$.}
\end{enumerate}
