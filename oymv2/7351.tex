\texte{
Soit $E$ un ensemble et $\mathcal{P}(E)$ l'ensemble de ses parties.
}
\begin{enumerate}
    \item \question{Déterminer les propriétés de la loi $(\mathcal{P}(E),\cup )$
(commutativité, associativité, existence et unicité des éléments
neutres, éléments symétrisables).}
    \item \question{Déterminer les propriétés de la loi $(\mathcal{P}(E),\cap )$.}
    \item \question{La loi $\cup$ est-elle distributive par rapport à la loi $\cap$ ?}
\end{enumerate}
}
