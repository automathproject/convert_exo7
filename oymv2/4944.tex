\contenu{
\texte{
Soient ${\cal T}$ un torseur et ${\cal P}$ un plan.
Déterminer le lieu des points $M \in {\cal P}$ tels que
${\cal T}(M) \in \vec{{\cal P}}$.
}
\reponse{
Droite orthogonale à $\vec S\wedge\vec u$,
         $\vec u \perp\vec{{\cal P}}$,
         ou $\varnothing$ ou ${\cal P}$.
}
}
