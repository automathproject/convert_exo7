% https://www.artofproblemsolving.com/wiki/index.php?title=Cauchy_Induction
Soit $A$ une partie de $\N^*$ contenant $1$ et telle que
}
\begin{enumerate}
    \item \question{$\forall n\in \N^*,\: n\in A \Rightarrow 2n \in A$;}
    \item \question{$\forall n\in \N^*,\: n+1\in A \Rightarrow n \in A$.}
\end{enumerate}
}
