\texte{
Soient $A$ et $B$ deux parties non vides born\'ees de $\Rr$.
}
\begin{enumerate}
    \item \question{Montrer que $A \cup B$ est born\'ee et que $\sup (A \cup B) = \max (
\sup (A), \sup (B))$.}
    \item \question{Enoncer un r\'esultat analogue pour $\inf (A \cup B)$.}
    \item \question{Qu'en est-il pour $A \cap B$ ?}
\end{enumerate}
}
