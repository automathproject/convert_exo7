\texte{
Soit $  G  $ un groupe $  Z(G)=\{ h\in G   ;   \forall g \in g   ,
  gh=hg\}  .$
}
\begin{enumerate}
    \item \question{Montrer que $  Z(G)  $ est un sous-groupe distingu\'e de $  G  .$}
    \item \question{Montrer que si $  G/Z(G)  $ est monog\`ene $  G  $ est cyclique.}
\end{enumerate}
}
