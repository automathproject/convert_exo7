}
\reponse{
$\sum_{k=0}^{n-1}\frac1{n^2-k^2}
= \frac1{2n}\sum_{k=0}^{n-1}\Bigl(\frac1{n-k} + \frac1{n+k}\Bigr)
= \frac1{2n}\Bigl(\sum_{k=1}^{2n-1}\frac1k +\frac1n\Bigr)
\sim \frac{\ln n}{2n}$.
}
}
