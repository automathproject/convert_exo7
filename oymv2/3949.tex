\contenu{
\texte{
Soit $f:\R \to \R$ dérivable telle que :
$\forall\ a,b \in \R,\ f(b) - f(a) = (b-a)f'\left(\frac {a+b}2\right)$.
Montrer que $f$ est polynomiale de degré inférieur ou égal à 2.
}
\reponse{
Dériver par rapport à $a$ puis par rapport à $b$.
}
}
