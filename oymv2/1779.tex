\texte{
Soit $E$ un evn, $V$ un  sous-espace vectoriel de $E$.
}
\begin{enumerate}
    \item \question{Montrer que $\overline{V}$ est un sous-espace vectoriel de $E$.}
    \item \question{Montrer que si $\stackrel{\circ}{V} \not = \varnothing$  alors $V=E.$}
\end{enumerate}
}
