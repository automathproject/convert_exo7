\texte{
Soit $\sum a_nz^n$ une série entière de rayon de convergence $R > 0$.
Déterminer les rayons de convergence des séries :
}
\begin{enumerate}
    \item \question{$\sum a_n^2 z^n$.}
\reponse{$R' = R^2$.}
    \item \question{$\sum \frac {a_n}{n!} z^n$.}
\reponse{$R'=\infty$.}
    \item \question{$\sum \frac{n!\,a_n}{n^n}z^n$.}
\reponse{$R' = eR$.}
\end{enumerate}
}
