\texte{
D{\'e}terminer tous les morphismes de
}
\begin{enumerate}
    \item \question{$(\Q,+)$ dans $(\Q,+)$.}
\reponse{$x \mapsto ax$, $a\in\Q$.}
    \item \question{$(\Q,+)$ dans $(\Z,+)$.}
\reponse{$x \mapsto 0$.}
    \item \question{$(\Q,+)$ dans $(\Q^*,\times)$.}
\reponse{$x \mapsto 1$.}
\end{enumerate}
}
