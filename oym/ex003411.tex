\contenu{
  \texte{}
  \question{\'Etudier l'existence de solutions du système :
$$\left\{\begin{array}{ccccccc} 3mx    &+& (3m-7)y &+& (m-5)z  &=& m-1  \cr
                 (2m-1)x &+& (4m-1)y &+& 2mz     &=& m+1  \cr
                 4mx     &+& (5m-7)y &+& (2m-5)z &=& 0.   \cr\end{array}\right.$$}
  \reponse{Si $m \ne 0,-2$ alors système de Cramer ; sinon, système incompatible.}
}
