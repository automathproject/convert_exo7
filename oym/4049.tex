\titre{$(u^v - v^u)/(u-v)$}
\theme{Exercices de Michel Quercia, \'Equivalents}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient $u,v$ deux fonctions positives, $u \sim v$, $u \to 0$.
Montrer que $\frac {u^v - v^u}{u-v} \sim -\ln(v)$.

(\'Ecrire $u = v + w$ avec $w/v \to 0$)}
  \reponse{}
}
