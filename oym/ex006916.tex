\contenu{
  \texte{On lance deux dés à 6 faces. Soit $X_1$ le résultat du premier dé,
$X_2$ le résultat du deuxième dé, et $S=X_1+X_2$.}
\begin{enumerate}
  \item \question{Calculer $E(X_1)$ et $Var(X_1)$.}
  \item \question{En déduire $E(S)$ et $\text{Var}(S)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$E(X_1)=3.5$, $\text{Var}(X_1)=\frac{35}{12}\simeq 2,92$.}
  \item \reponse{$X_1,X_2$ sont indépendantes de même loi.
D'où $E(S)=2E(X_1)=7$ et $\text{Var}(S)=2\text{Var}(X_1)=\frac{35}{6}$.}
\end{enumerate}
}
