\contenu{
  \texte{}
  \question{Soit $\cal P$ la parabole d'équation $y^2 = 2px$.
Chercher l'enveloppe des cordes $[A,B]$ de $\cal P$ de hauteur $h > 0$
donnée.}
  \reponse{$y_A = t  \Rightarrow  \begin{cases} 2px = t^2 + ht + h^2/2 \cr y = t + h/2\cr \end{cases}
                   \Rightarrow  \text{parabole } y^2 + h^2/4 = 2px$.}
}
