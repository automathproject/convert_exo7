\contenu{
  \texte{Soit $R_n$ le nombre de relations d'{\'e}quivalence sur un ensemble {\`a} $n$ {\'e}l{\'e}ments.}
\begin{enumerate}
  \item \question{Trouver une relation de r{\'e}currence entre $R_n$ et les $R_k$, $k < n$
    \par
    (fixer un {\'e}l{\'e}ment, et raisonner sur la classe d'{\'e}quivalence de cet
    {\'e}l{\'e}ment).}
  \item \question{Calculer $R_n$ pour $n \le 6$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$R_n = \sum_{k=0}^{n-1}C_{n-1}^kR_k$ avec $R_0 = 1$.}
  \item \reponse{1,1,2,5,15,52,203.}
\end{enumerate}
