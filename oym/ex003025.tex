\contenu{
  \texte{Soit $A$ un anneau non nul, commutatif et int{\`e}gre.}
\begin{enumerate}
  \item \question{Montrer que si $A$ est fini, alors c'est un corps.}
  \item \question{Montrer que si $A$ n'a qu'un nombre fini d'id{\'e}aux, alors c'est un corps
    (consid{\'e}rer les id{\'e}aux $I_n = x^nA$ pour $x\in A$ non nul).}
\end{enumerate}
