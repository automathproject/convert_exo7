\contenu{
  \texte{Soit $E$ un ev de dimension finie et $f \in \mathcal{L}(E)$.
Montrer que les propriétés suivantes sont équivalentes :}
\begin{enumerate}
  \item \question{$\mathrm{Ker} f^2 = \mathrm{Ker} f$.}
  \item \question{$\Im  f^2 = \Im  f$.}
  \item \question{$\mathrm{Ker} f \oplus \Im f = E$.}
  \item \question{$\mathrm{Ker} f \cap \Im f = \{\vec 0\}$.}
  \item \question{$\mathrm{Ker} f + \Im f = E$.}
\end{enumerate}
