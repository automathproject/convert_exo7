\titre{Propriété de parité}

\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f:\R \to \R$ de classe $\mathcal{C}^1$ telle que la fonction
$t  \mapsto 2f(t) - tf'(t)$ est paire. $f$ est-elle paire~?}
  \reponse{Contre-exemple~: $f(t) = t^2$ si $t\ge 0$ et $f(t) = -t^2$ si $t<0$.}
}
