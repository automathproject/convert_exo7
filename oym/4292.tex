\titre{$\int_0^\infty f(t)/t^2\,d t$}

\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {\R^+} \to {\R^+}$ une fonction de classe $\mathcal{C}^1$ vérifiant :
$\exists\ \alpha > 0 \text{ tel que } \forall\ x \ge 0,\ f'(x) \ge \alpha$.
Montrer que $ \int_{t=1}^{+\infty} \frac{f(t)}t\,d t$ diverge.}
  \reponse{}
}
