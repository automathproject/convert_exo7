\contenu{
  \texte{}
  \question{Déterminer l'ensemble des centres de courbure en $O$ aux courbes intégrales
de l'équation dif\-fé\-ren\-tielle $(1-x^2)y''-xy'-2y=1$ telles que $y(0)=0$.}
  \reponse{$y'(0)=\lambda  \Rightarrow  I = (-\lambda-\lambda^3,1+\lambda^2)$.}
}
