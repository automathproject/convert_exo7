\titre{Droites non coplanaires}
\theme{Exercices de Michel Quercia, Sous-espaces affines}

\date{2010/03/17}
\organisation{exo7}
\contenu{
  \texte{Soit $\cal E$ un espace affine de dimension 3, et $D,D',D''$ trois droites
parallèles à un même plan $\cal P$, mais deux à deux non coplanaires.}
\begin{enumerate}
  \item \question{Montrer que par tout point $A$ de $D$, il passe une unique droite $\Delta_A$
    rencontrant $D'$ et $D''$.}
  \item \question{Montrer que les droites $\Delta_A$ sont toutes parallèles à un même plan
    $\cal Q$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Le plan passant par $A$ et $D'$ n'est pas parallèle à $D''$.}
  \item \reponse{On doit pouvoir s'en sortir avec un repère adéquat \dots}
\end{enumerate}
