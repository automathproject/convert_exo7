\titre{$u_n/(1+u_n)$}

\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $\sum u_n$ une série à termes positifs et $v_n = \frac{u_n}{1+u_n}$.
Montrer que $\sum u_n$ et $\sum v_n$ ont même nature.}
  \reponse{Si $u_n \to 0$, alors $v_n \sim u_n$;
         sinon, $v_n$ \hbox to 0pt{\hskip 3mm$/$\hss} $\longrightarrow 0$.}
}
