  \texte{Soit $A\in M(4,\Rr)$ semi-simple, et soient $\pm
ai$, $\pm bi$, $a>0$, $b>0$ les valeurs propres.}
\begin{enumerate}
  \item \question{Montrer que si $a/b$ est rationnelle, alors
toutes les solutions de $x' = Ax$ sont périodiques.}
  \item \question{Montrer que si $a/b$ est irrationnelle, alors
il existe une solution non-périodique $x(t)$ telle que
$M < \vert x(t) \vert < N$ pour certaines constantes
$M,N>0$.}
\end{enumerate}
