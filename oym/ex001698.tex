  \texte{}
  \question{Soit $E$ un $\Cc$-espace vectoriel de dimension finie $n$. Soit $f\in\mathcal{L}(E)$ un
endomorphisme nilpotent d'indice $N$ (le plus petit entier $p$ tel que $f^p = 0$).
Montrer que
$$N=n \Leftrightarrow \text{rang} f = n-1.$$}
  \reponse{}
}
