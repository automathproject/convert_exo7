\contenu{
  \texte{Chercher les extrémums des fonctions $f(x,y)$ suivantes :}
\begin{enumerate}
  \item \question{$3xy - x^3 - y^3$}
  \item \question{$-2(x-y)^2+x^4+y^4$}
  \item \question{$x^2y^2(1+3x+2y)$}
  \item \question{$2x+y-x^4-y^4$}
  \item \question{$\frac {xy}{(x+y)(1+x)(1+y)},\ x,y> 0$}
  \item \question{$xe^y + ye^x$}
  \item \question{$x(\ln^2x + y^2)$, $x>0$}
  \item \question{$\sqrt{x^2+(1-y)^2} + \sqrt{y^2+(1-x)^2}$}
  \item \question{$MA+MB-MO$,\quad $O = \text{mil}(A,B)$}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$(0,0)$ : non extrémal\par
             $(1,1)$ : maximum local}
  \item \reponse{$(0,0)$ : non extrémal\par
             $\pm(\sqrt2, -\sqrt2)$ : minimum absolu}
  \item \reponse{$(-2/15,-1/5)$ : maximum local \par
             $(x,0)$ : max. local pour $x<-1/3$, min. local pour $x > -1/3$\par
             $(0,y)$ : max. local pour $y<-1/2$, min. local pour $y > -1/2$}
  \item \reponse{$(2^{-1/3},4^{-1/3})$ : maximum absolu}
  \item \reponse{prendre le log. $(1,1)$ : maximum absolu}
  \item \reponse{$(-1,-1)$ : non extrémal}
  \item \reponse{$(1,0)$ : minimum absolu\par
             $(e^{-2},0)$ : non extrémal}
  \item \reponse{$= MA + MB  \Rightarrow $ minimum absolu sur $[A,B]$}
  \item \reponse{$M \in \text{med}(A,B),\ \overline{(\vec{MA}, \vec{MB})} = \pm \frac {2\pi}3$\par
             $M = A,B$ : minimum absolu\par
             $M = O$}
\end{enumerate}
}
