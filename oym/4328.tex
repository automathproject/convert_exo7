\contenu{
  \texte{}
  \question{Chercher un équivalent pour $n\to\infty$ de $ \int_{t=1}^{1+1/n} \sqrt{1+t^n}\,d t$.}
  \reponse{$u=t^n  \Rightarrow  \sim \frac 1n  \int_{u=1}^e \frac{\sqrt{1+u}}u\,d u$.}
}
