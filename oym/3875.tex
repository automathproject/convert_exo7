\titre{$f$ croissante et $x  \mapsto f(x)/x$ est décroissante}
\theme{Exercices de Michel Quercia, Fonctions monotones}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {]0,+\infty[} \to {\R}$ croissante telle que
$g : {]0,+\infty[} \to {\R}, x \mapsto {\frac {f(x)}x}$ est décroissante.
Montrer que $f$ est continue.}
