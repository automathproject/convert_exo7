\titre{$ \int_{t=0}^1 t^t\,d t$}
\theme{}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
\begin{enumerate}
  \item \question{A l'aide d'un développement en série entière, montrer que
    $ \int_{t=0}^1 t^t\,d t = \sum_{n=1}^\infty \frac{(-1)^{n-1}}{n^n}$.}
  \item \question{Calculer la valeur commune des deux membres à $10^{-5}$ près.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$t^t = \exp(t\ln t) = \sum_{k=0}^\infty \frac{t^k\ln^k t}{k!}$.}
  \item \reponse{0.78343}
\end{enumerate}
