\contenu{
  \texte{}
  \question{Peut-on empiler 100 pièces de 1F de sorte que la dernière soit complètement
en porte à faux ?
(c'est-à-dire que sa projection sur un plan horizontal ne rencontre pas la projection de
la première pièce)}
  \reponse{Le déport maximal entre la première pièce et la dernière pour une
pile de $n$ pièces est $\frac 12 + \frac 14 + \dots + \frac 1{2(n-1)}$
(en diamètre d'une pièce).
Il dépasse 1 pour $n > 4$.}
}
