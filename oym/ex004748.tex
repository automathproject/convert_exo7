\titre{Suite de Cauchy non convergente}
\theme{}
\auteur{quercia}
\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $E = \R[X]$ muni de la norme : $\|\sum a_kX^k \| = \max(|a_k|,\ k\in \N)$.
On note $P_n = 1 + X + \frac{X^2}2 + \dots + \frac{X^n}n$.
Montrer que la suite $(P_n)$ est de Cauchy, mais ne converge pas.}
  \reponse{}
