\contenu{
  \texte{}
  \question{Soit $f:\R \to \R$ de classe $\mathcal{C}^\infty$ telle que pour tous $n\in\N$ et $x\in\R$,
on a $f^{(n)}(x) > 0$.

Montrer que pour tout entier $n$, $\frac{f(x)}{x^n}\to +\infty$ lorsque $x\to+\infty$.}
  \reponse{}
}
