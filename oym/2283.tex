\contenu{
  \texte{}
  \question{Montrer que dans un anneau fini tout id\'eal premier est maximal.}
  \reponse{Soit $A$ un anneau fini, et $I$ un idéal premier. Alors $A/I$ est
  intègre, et fini (!), donc $A/I$ est un corps (voir exercice \ref{ex:bar5}). Donc $I$ est maximal.}
}
