  \texte{}
  \question{Soit $I =  \int_{t=0}^\pi \ \frac {t\sin t}{1 + \cos^2t}\,d t$.
Effectuer dans $I$ le changement de variable $u = \pi - t$, et en déduire
la valeur de~$I$.}
  \reponse{$\frac {\pi^2}4$.}
}
