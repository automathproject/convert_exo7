\begin{enumerate}
  \item \question{Soit $M \in \mathcal{M}_n(\Z)$. Montrer que $M \in GL_n(\Z)$ si et seulement si
    $|\det M| = 1$.}
  \item \question{Soit $X = \begin{pmatrix} x_1 \cr \vdots \cr x_n \cr \end{pmatrix} \in \mathcal{M}_{n,1}(\Z)$ et
    $d$ le pgcd de $x_1, \dots, x_n$.
    Montrer qu'il existe $A \in GL_n(\Z)$ telle que
    $AX = \begin{pmatrix}d \cr 0\cr \vdots \cr 0 \cr \end{pmatrix}$ (par récurrence sur $n$).}
  \item \question{Soient $X,Y \in \mathcal{M}_{n,1}(\Z)$. CNS pour qu'il existe $A \in GL_n(\Z)$
    telle que $AX = Y$ ?}
