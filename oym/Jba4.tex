\contenu{
  \texte{Soit $P \in {\Z[X]}$ et $a,b \in \Z^*$ premiers entre eux tels que
$P\Bigl(\frac ab\Bigr) = 0$.}
\begin{enumerate}
  \item \question{Montrer que $a$ divise le coefficient constant de $P$.}
  \item \question{Montrer que pour tout $k\in\Z$, $a-kb$ divise $P(k)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Appliquer le { 1)} {\`a} $P(X+k)$.}
\end{enumerate}
