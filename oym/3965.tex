\titre{Dérivée $n$-ème de $\arctan x$ et $e^{x^3}$}
\theme{Exercices de Michel Quercia, Dérivation}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{\'Etablir une formule de récurrence pour les dérivées successives des fonctions :

$f : x \mapsto \arctan x$ et $g : x \mapsto e^{x^3}$.}
  \reponse{$(1+x^2)f^{(n+1)} + 2nxf^{(n)} + n(n-1)f^{(n-1)} = 0$ pour $n \ge 1$.\par
         $g^{(n+1)} = 3x^2g^{(n)} + 6nxg^{(n-1)} + 3n(n-1)g^{(n-2)}$ pour $n \ge 0$.}
}
