  \texte{Soit $A$ et $B$ deux parties born\'ees de $\Rr$.
\textbf{Vrai} ou \textbf{faux} ?}
\begin{enumerate}
  \item \question{$A \subset B \Rightarrow \sup A \leqslant \sup B$,}
  \item \question{$A \subset B \Rightarrow \inf A \leqslant \inf B$,}
  \item \question{$\sup (A\cup B) = \max(\sup A,\sup B)$,}
  \item \question{$\sup(A+B) < \sup A + \sup B$,}
  \item \question{$\sup(-A) = -\inf A$,}
  \item \question{$\sup A +\inf B \leqslant \sup(A+B)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Vrai.}
  \item \reponse{Faux. C'est vrai avec l'hypothèse $B \subset A$ et non $A \subset B$.}
  \item \reponse{Vrai.}
  \item \reponse{Faux. Il y a égalité.}
  \item \reponse{Vrai.}
  \item \reponse{Vrai.}
\end{enumerate}
}
