\titre{Projection $p + q - q\circ p$}
\theme{}
\auteur{quercia}
\date{2010/03/10}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient $p,q$ deux projections telles que $p \circ q = 0$.
Montrer que $p + q - q \circ p$ est une projection, et déterminer ses éléments.}
  \reponse{Direction = $\mathrm{Ker} p \cap \mathrm{Ker} q$ et Base = $\Im p \oplus \Im q$.}
}
