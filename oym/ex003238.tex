\titre{Polyn{\^o}me minimal de $2\cos(2\pi/7)$}

\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Montrer que $x = 2\cos\frac {2\pi}7$ est racine de $X^3 + X^2 - 2X - 1$.
Quelles sont les autres racines ?}
  \reponse{$x = z + \frac 1z$ avec $z^6 + z^5 + \dots + 1 = 0$.
\\
Autres racines : $2\cos\frac{4\pi}7$ et $2\cos\frac{6\pi}7$.}
}
