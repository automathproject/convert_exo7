  \item \question{Calculer le ppcm des nombres  : 108 et 144 ; 128 et 230 ; 6, 16 et 50.}
  \item \question{Montrer que si $a\ge 1$ et $b\ge 1$ sont des entiers
    de pgcd $d$ et, si on pose $a=da' ; b=db' $, le ppcm de $a$ et $b$ est $da'b'$.}
  \item \question{Montrer que si $a,b,c$ sont des entiers sup\'erieurs \`a  $1$,
    on a :
    $$\text{ppcm}(a,b,c)=\text{ppcm}(\text{ppcm}(a,b),c).$$}
\end{enumerate}
