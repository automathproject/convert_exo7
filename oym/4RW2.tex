\titre{Approximation de $f$ et $f'$}
\theme{Exercices de Michel Quercia, Suites et séries de fonctions}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{Soit $f : {[a,b]} \to \R$ de classe $\mathcal{C}^1$.}
\begin{enumerate}
  \item \question{Montrer qu'il existe
une suite de polynômes $(P_n)$ telle que $P_n$ converge uniformément
vers $f$ et $P_n'$ converge uniformément vers $f'$.}
  \item \question{Si $f$ est $\mathcal{C}^\infty$, peut-on trouver une suite de polynômes
$(P_n)$ telle que pour tout $k$ la suite $(P_n^{(k)})$ converge uniformément
vers $f^{(k)}$~?}
\end{enumerate}
\begin{enumerate}
  \item \reponse{oui, par convolution.}
\end{enumerate}
