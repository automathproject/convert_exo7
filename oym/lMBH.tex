\titre{$f$ et $g$ colin{\'e}aires}
\theme{Exercices de Michel Quercia, Fonctions vectorielles}
\auteur{quercia}
\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{Soient ${f,g} : {I\subset \R} \to E$ deux fonctions vectorielles de classe $\mathcal{C}^1$.}
\begin{enumerate}
  \item \question{On suppose : $\forall\ t \in I,\ f(t)$ et $g(t)$ sont colin{\'e}aires.
    Est-ce que $f'(t)$ et $g'(t)$ sont colin{\'e}aires ?}
  \item \question{On suppose : $\forall\ t \in I,\ f'(t)$ et $g'(t)$ sont colin{\'e}aires.
    Existe-t-il $\vec c \in E$ tel que $f-\vec c$ et $g$ soient colin{\'e}aires~?}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Non : $f(t) = (t,t^2)$, $g(t) = (1,t)$.}
  \item \reponse{Non : $f(t) = \left(\frac{t^2}2, \frac{t^3}3\right)$,
                   $g(t) = \left(t,\frac{t^2}2\right)$.}
\end{enumerate}
}
