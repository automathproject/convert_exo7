  \texte{}
  \question{Soit $ E$ un espace euclidien et $ f:E\rightarrow E$ tel que $ f(0)=0 $ et :
$$\forall (x,y)\in E^{2},\left\| f(x)-f(y)\right\| =\left\| x-y\right\| . $$
Montrer que $ f$ est lin\'{e}aire.}
