\titre{Groupe fini ?}
\theme{Exercices de Michel Quercia, Groupes}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $G$ un groupe ayant un nombre fini de sous-groupes. Montrer que $G$ est fini.}
  \reponse{Si $a\in G$ est d'ordre infini alors il engendre un sous-groupe
isomorphe {\`a}~$\Z$, qui a une infinit{\'e} de sous-groupes~; c'est exclu.
Donc tous les sous-groupes monog{\`e}nes de~$G$ sont finis, et $G$ est
la r{\'e}union de ces sous-groupes.}
}
