  \texte{}
  \question{Déterminer la développante de la chainette $\mathcal{C}$ d'équation $y = a\ch(x/a)$ qui
rencontre $\mathcal{C}$ pour $x=0$. (Tractrice) Dessiner les deux courbes.}
  \reponse{$x=a\ln\left|\tan\left(\frac t2+\frac\pi4\right)\right|$,\quad
         $y=a\cos t$.}
