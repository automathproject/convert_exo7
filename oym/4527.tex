\titre{Limite simple de fonctions convexes}
\theme{Exercices de Michel Quercia, Suites et séries de fonctions}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit ${f_n} : {[a,b]} \to \R$ des fonctions continues convexes convergeant simplement
vers une fonction continue $f$. Montrer que la convergence est uniforme.}
  \reponse{Prendre une subdivision régulière de $[a,b]$ et encadrer $f_n$ par
les cordes associées.}
}
