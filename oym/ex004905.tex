\contenu{
  \texte{}
  \question{On donne un cercle ${\cal C}$ de centre $O$ et $A \in {\cal C}$.
Pour $M \in {\cal C}$, on construit le projeté $N$ sur le diamètre
perpendiculaire à $(OA)$, et $I$, le point d'intersection de $(OM)$ et $(AN)$.
Quel est le lieu de $I$ ?}
  \reponse{Repère $(O,\frac{\vec{OA}}R,\vec j)   \Rightarrow $ parabole $\rho = \frac{R}{1+\sin\theta}$.}
