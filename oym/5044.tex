\contenu{
  \texte{}
  \question{Soit $s \mapsto M_s$ une courbe de l'espace de classe $\mathcal{C}^3$ paramétrée par une
abscisse curviligne,
et $P$ le point tel que $\vec{OP} = \frac {\vec{dM}}{ds}$.
Chercher les éléments de courbure de la trajectoire de $P$.}
  \reponse{$c_1 = \sqrt{1+\frac{\tau^2}{c^2}}$,
         $\tau_1 = \frac{c\tau'-\tau c'}{c(c^2+\tau^2)}$.}
}
