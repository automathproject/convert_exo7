\contenu{
  \texte{Soient $A, B \subset E$. R\'esoudre les \'equations \`a l'inconnue $X \subset E$}
\begin{enumerate}
  \item \question{$A \cup X = B$.}
  \item \question{$A \cap X = B$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$B\setminus A \subset X \subset B$.}
  \item \reponse{$B \subset X \subset B \cup \complement{A}$.}
\end{enumerate}
}
