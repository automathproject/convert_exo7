\contenu{
  \texte{Soit~$f(a) = \sum_{n=0}^\infty e^{-a^2n^2}$ sous réserve de convergence $(a\in\R$).}
\begin{enumerate}
  \item \question{Domaine de définition de~$f$~?}
  \item \question{Limite de~$af(a)$ quand $a\to0$~?}
  \item \question{Limite de~$f(a)$ quand $a\to+\infty$~?}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\R^*$.}
  \item \reponse{CSI~: $\frac{\sqrt\pi}{2a} =  \int_{x=0}^\infty e^{-a^2x^2}\,d x \le f(a) \le  \int_{x=0}^\infty e^{-a^2x^2}\,d x + 1 = \frac{\sqrt\pi}{2a} + 1$. Donc $af(a)\to \frac{\sqrt\pi}2$ lorsque $a\to0^+$.}
  \item \reponse{TCM~: $f(a)\to 1$ lorsque $a\to+\infty$.}
\end{enumerate}
