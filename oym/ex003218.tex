\titre{$X^2-2X\cos\theta+1$ divise $X^{n+1}\cos(n-1)\theta-X^n\cos n\theta-X\cos\theta+1$}
\theme{}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Montrer que $X^2-2X\cos\theta+1$ divise
$X^{n+1}\cos(n-1)\theta-X^n\cos n\theta-X\cos\theta+1$, puis
d{\'e}terminer le quotient.}
  \reponse{Division de proche en proche : $Q = \sum_{k=0}^{n-1} X^k\cos k\theta$.}
}
