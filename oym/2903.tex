  \texte{}
  \question{Soit $E$ un ensemble fini de cardinal $n$.
Calculer $\sum_{A \subset E} \mathrm{Card}\,(A)$,
         $\sum_{A,B \subset E} \mathrm{Card}\,(A\cap B)$,
         $\sum_{A,B \subset E} \mathrm{Card}\,(A\cup B)$.}
  \reponse{$n2^{n-1}$, $n4^{n-1}$, $3n4^{n-1}$.}
}
