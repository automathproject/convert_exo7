\titre{Matrice vérifiant $A^k = I$}

\auteur{quercia}
\date{2010/03/09}
\organisation{exo7}
\contenu{
  \texte{Soit $A \in \mathcal{M}_n(K)$ telle que $A^k = I\ (k \ne 0)$.
On pose $B = I + A + A^2 + \dots + A^{k-1}$.
Soient $u,v$ les endomorphismes de $ K^n$ matrices $A$ et $B$ dans la base
canonique.}
\begin{enumerate}
  \item \question{Montrer que : $\mathrm{Ker} (u-\mathrm{id}) = \Im v$,\quad $\Im(u-\mathrm{id}) = \mathrm{Ker} v$,\quad
    $\mathrm{Ker} v \oplus \Im v =  K^n$.}
  \item \question{En déduire : $\mathrm{tr} B = k\mathrm{rg} B$.}
\end{enumerate}
