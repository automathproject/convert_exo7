  \texte{}
  \question{Calculer $I = \iint_\Delta xy\,d xd y$ où
    $\Delta = \{(x,y) \text{ tel que } y\ge 0 \text{ et } (x+y)^2 \le 2x/3 \}$.}
  \reponse{Poser $u=x, v=x+y$. On obtient $I=\frac{2}{1701}$.}
