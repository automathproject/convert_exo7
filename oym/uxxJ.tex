\contenu{
  \texte{Soit $A \in \mathcal{M}_n(K)$ telle que $a_{ij} = (-1)^{n-j}C_{n-j}^{i-1}$.}
\begin{enumerate}
  \item \question{Interpréter $A$ comme la matrice d'un endomorphisme de $ K_{n-1}[X]$.}
  \item \question{En déduire $A^3$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\phi(P) = (-X-1)^{n-1}P\left(-\frac 1{X+1}\right)$.}
  \item \reponse{$I$.}
\end{enumerate}
