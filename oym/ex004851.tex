\titre{$f\circ f$ est contractante}
\theme{}
\auteur{quercia}
\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $E$ un espace vectoriel norm{\'e} complet et $f : E \to E$ telle que $f\circ f$
est contractante. Montrer que $f$ admet un unique point fixe.}
  \reponse{Soit $(a_n)$ d{\'e}finie par $a_{n+1} = f(a_n)$ : les sous-suites
$(a_{2n})$ et $(a_{2n+1})$ convergent vers le point fixe de $f\circ f$.}
}
