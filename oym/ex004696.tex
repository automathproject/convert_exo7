\titre{Fonction $\N \to \N$ injective}
\theme{}
\auteur{quercia}
\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {\N} \to {\N}$ injective.
Montrer que $\frac{f(1)}{1^2} + \frac{f(2)}{2^2} + \dots + \frac{f(n)}{n^2} \xrightarrow[n\to\infty]{} +\infty$.}
  \reponse{$u_{2n}-u_{n} \ge \frac {1 + \dots + n}{4n^2} \ge \frac 18$.}
}
