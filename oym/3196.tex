\contenu{
  \texte{}
  \question{Soit $A \in { K[X]}$ de degr{\'e} $> 0$.
Montrer que pour tout polyn{\^o}me $P \in { K_n[X]}$, il existe des polyn{\^o}mes $P_0,P_1,\dots,
P_n$ uniques v{\'e}rifiant :
$$\begin{cases}\deg P_i < \deg A\cr
P = P_0 + P_1A + \dots + P_nA^n.\end{cases}$$}
  \reponse{}
}
