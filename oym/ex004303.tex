\contenu{
  \texte{}
  \question{On admet que $ \int_{t=0}^{+\infty} e^{-t^2}\,d t =\frac{\sqrt\pi}2$.
Calculer les intégrales :
$I_n =  \int_{t=0}^{+\infty} e^{-t^2}t^{2n}\,d t$ pour $n \in \N$.}
  \reponse{$I_n = \frac{(2n)!\sqrt\pi}{2^{2n+1}n!}$.}
}
