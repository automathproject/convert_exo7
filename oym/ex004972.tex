\contenu{
  \texte{}
  \question{Soient $P$ un plan, $D$ une droite tels que
$\overline{(P,D)} \equiv \theta (\mathrm{mod}\,\pi)$.

Montrer que pour toute droite $\Delta \subset P$, on a
$\cos(D,\Delta) \ge \cos \theta$.
Quand y a-t-il égalité ?}
