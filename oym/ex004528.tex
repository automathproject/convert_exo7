\titre{Fonction définie par une série}
\theme{}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{On pose $f(x) = \sum_{n=0}^\infty \frac{\arccos(\cos nx)}{n!}$.}
\begin{enumerate}
  \item \question{Montrer que $f$ est définie sur $\R$, continue, paire et $2\pi$-périodique.}
  \item \question{Calculer $f(0)$, $f(\pi)$, $f\left(\frac\pi2\right)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$f(0) = 0$, $f(\pi) = \pi\sh 1$,
             $f\left(\frac\pi2\right) = \frac\pi2(e-\cos 1)$.}
\end{enumerate}
}
