\titre{Tangentes passant par un point donné}

\contenu{
  \texte{}
  \question{Soit $f : {[a,b]} \to \R$ dérivable, telle que $f(a) = f(b) = 0$.
Montrer que pour tout $d \in \R\setminus[a,b]$, il existe une tangente à ${\cal C}_f$
passant par le point $(d,0)$.}
