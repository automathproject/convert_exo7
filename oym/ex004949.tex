  \texte{}
  \question{Soit $ABC$ un triangle équilatéral de côté $a$.

Quels sont les points $M$ du plan $(ABC)$ tels que
$MA^2 + a^2 = 2(MB^2 + MC^2)$ ?}
  \reponse{Cercle circonscrit au triangle $A'BC$ symétrique de $ABC$ par
         rapport à $(BC)$.}
}
