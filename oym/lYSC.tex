\contenu{
  \texte{Chercher les courbes planes vérifiant l'équation intrinsèque :}
\begin{enumerate}
  \item \question{$R=s$.}
  \item \question{$Rs = 1$.}
  \item \question{$R^2 = 2as$, $a > 0$ donné.}
  \item \question{$R = 1+s^2$.}
  \item \question{$R^2+s^2 = a^2$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$x =  \int_{}^s \cos\ln|t|\,d t,\quad
              y =  \int_{}^s \sin\ln|t|\,d t,\quad
              \rho = \pm\frac1{\sqrt2}e^{\theta+\pi/4}$.}
  \item \reponse{$x =  \int_0^s \cos\frac{u^2}2\,du,\quad
              y =  \int_0^s \sin\frac{u^2}2\,du$
             (Clothoïde ou spirale de Cornu)}
  \item \reponse{$x = a(\varphi\sin\varphi + \cos\varphi),\quad
              y = a(-\varphi\cos\varphi + \sin\varphi)$.}
  \item \reponse{$x = \ln\Bigl|\tan\Bigr(\frac\varphi2+\frac\pi4\Bigr)\Bigr|,\quad
              y = \frac1{\cos\varphi} = \ch x$.}
  \item \reponse{$x=\frac a4(\sin2\varphi + 2\varphi),\quad
             y = \frac a4\cos2\varphi$ (cycloïde).}
\end{enumerate}
