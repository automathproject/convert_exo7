\titre{$(\int tf(t)\,d t)/x$}
\theme{}
\auteur{quercia}
\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {[0,+\infty[} \to \R$ continue telle que $ \int_{t=0}^{+\infty} f(t)\,d t$
converge. Montrer que $\frac1x \int_{t=0}^x tf(t)\,d t \to 0$ lorsque $x\to+\infty$.}
  \reponse{Soit $F(x) =  \int_{t=0}^x f(t)\,d t$ :
         $\frac1x \int_{t=0}^x tf(t)\,d t = F(x) - \frac1x \int_{t=0}^x F(t)\,d t$.}
}
