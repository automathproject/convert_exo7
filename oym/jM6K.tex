\titre{$\Arctan(n+a) - \Arctan n$}
\theme{Exercices de Michel Quercia, Séries numérique}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{Soit $a \in \R$.}
\begin{enumerate}
  \item \question{Montrer que la série de terme général $\Arctan(n+a) - \Arctan n$ est
    convergente.}
  \item \question{On pose $S(a) = \sum_{k=0}^\infty (\Arctan(k+a) - \Arctan k)$.
    Trouver $\lim_{a\to+\infty} S(a)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\sim \frac a{n^2}$.}
  \item \reponse{$S(a) \ge \sum_{k=0}^n \Arctan(k+a)-\Arctan k \to
              \frac\pi2 + \Arctan 1 + \Arctan\frac12 + \dots
              + \Arctan\frac1n
               \Rightarrow  S(a) \to +\infty$ lorsque $a\to+\infty$ .}
\end{enumerate}
}
