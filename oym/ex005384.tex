\contenu{
  \texte{}
  \question{Soient $f$ et $g$ deux fonctions continues en $x_0\in\Rr$. Montrer que $\mbox{Min}\{f,g\}$ et $\mbox{Max}\{f,g\}$ sont continues en $x_0$.}
  \reponse{$\mbox{Min}(f,g)=\frac{1}{2}(f+g-|f-g|)$ et $\mbox{Max}(f,g)=\frac{1}{2}(f-g+|f-g|)$ sont continues en $x_0$ en vertu de théorèmes généraux.}
}
