\contenu{
  \texte{\'Ecrire la n\'egation des assertions suivantes o\`u $P, Q, R, S$ sont des
propositions.}
\begin{enumerate}
  \item \question{$P \Rightarrow Q$,}
  \item \question{$P$ et non $Q$,}
  \item \question{$P$ et ($Q$ et $R$),}
  \item \question{$P$ ou ($Q$ et $R$),}
  \item \question{($P$ et $Q$) $\Rightarrow \ (R \Rightarrow S)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$P$ et non $Q$;}
  \item \reponse{``non P ou $Q$" ce qui la même chose que ``$P \Rightarrow Q$";}
  \item \reponse{(non $P$) ou ((non $Q$) ou (non $R$)) (on peut supprimer les parenthèses);}
  \item \reponse{non $P$ et (non $Q$ ou non $R$) (ici les parenthèses sont importantes);}
  \item \reponse{$P$ et $Q$ et $R$ et non $S$;}
\end{enumerate}
}
