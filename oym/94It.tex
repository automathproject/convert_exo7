  \item \question{Soit $f : {\R^+} \to {\R^+}$ continue ayant une limite finie en~$+\infty$.
    Montrer que $ \int_{t=0}^{+\infty} (f(t+a) - f(t))\,d t$ converge.}
  \item \question{Calculer $ \int_{t=0}^{+\infty} (\arctan(t+1) - \arctan(t))\,d t$.}
\end{enumerate}
