  \texte{Soient $E,F$, deux ensembles non vides. On d{\'e}finit deux relations sur $X = F^E$
par :
$$\begin{matrix}
  f \sim g &\iff &\exists\ \phi : F \to  F \text{ bijective tq } g=\phi\circ f,\hfill\cr
  f \equiv g &\iff &\bigl( \forall\ x,y \in E,\ f(x)=f(y)\iff g(x)=g(y)\bigr).\cr
\end{matrix}$$}
\begin{enumerate}
  \item \question{Montrer que ce sont des relations d'{\'e}quivalence.}
  \item \question{Montrer que $f \sim g  \Rightarrow  f \equiv g$.}
  \item \question{On suppose $f\equiv g$. Montrer que $f \sim g$ dans les cas suivants :
  \begin{enumerate}}
  \item \question{$F$ est fini et $f$ est surjective.}
  \item \question{$F$ est fini et $f$ est quelconque.}
  \item \question{$E$ est fini.}
\end{enumerate}
