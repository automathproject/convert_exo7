\titre{$\sum C_n^kx^{n-k}\cos(k\alpha) = 0$}
\theme{}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{R{\'e}soudre en $x$ :
$x^n + C_n^1x^{n-1}\cos\alpha + \dots + C_n^n\cos(n\alpha) = 0$.}
  \reponse{$\frac 12\Bigl( (x+e^{i\alpha})^n + (x+e^{-i\alpha})^n \Bigr) = 0
 \Leftrightarrow x = \mathrm{cotan}\left(\frac{(2k+1)\pi}{2n}\right)\sin\alpha - \cos\alpha$.}
}
