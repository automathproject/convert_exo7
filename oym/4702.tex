\titre{Suite r{\'e}curente}

\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $u_0\in\N^*$ et $(u_n)$ la suite d{\'e}finie par la relation
de r{\'e}currence~: $u_{n+1} = u_n^2+1$. Montrer qu'il exitste $a\in\R$ tel
que $u_n = [a^{2^n}]$ pour tout~$n$ o{\`u} $[\ ]$ d{\'e}signe la partie enti{\`e}re.}
  \reponse{Les suites $(\ln(u_n)/2^n)$ et $(\ln(1+u_n)/2^n)$ sont adjacentes.}
}
