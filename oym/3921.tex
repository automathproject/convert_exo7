\titre{$\arccos(\cos x) - \frac 12\arccos(\cos 2x)$}
\theme{Exercices de Michel Quercia, Fonctions circulaires inverses}

\contenu{
  \texte{}
  \question{Simplifier $\arccos(\cos x) - \frac 12\arccos(\cos 2x)$ pour $x \in {[0,2\pi]}$.}
  \reponse{$f(x)=0$ pour $x\in ]0,\pi/2[$ ;
$f(x)=2x-\pi$ pour $x\in ]\pi/2,\pi[$ ;
$f(x)=3\pi-2x$ pour $x\in ]\pi,3\pi/2[$ ;
$f(x)=0$ pour $x\in ]3\pi/2,2[$.}
}
