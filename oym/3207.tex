\contenu{
  \texte{}
  \question{Trouver $\lambda, \mu \in \C$ tels que $X^2+X+1$ divise
$X^5 + \lambda X^3 + \mu X^2 + 1$.}
  \reponse{$\lambda = \mu = -1$.}
}
