\titre{$\sup \circ \inf$ et  $\inf \circ \sup$}
\theme{Exercices de Michel Quercia, Relations d'ordre}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {\R^2} \to {\R}$ une fonction born{\'e}e. On d{\'e}finit les
fonctions :
$$g : {\R} \to {\R},  t \mapsto {\sup\{ f(t,y)\text{ tq } y\in \R\}}$$
 $$ h : {\R} \to {\R}, t \mapsto {\inf\{ f(x,t)\text{ tq } x\in \R\}}$$

Montrer que $g$ et $h$ sont born{\'e}es, puis comparer $\sup h$ et $\inf g$.}
  \reponse{}
}
