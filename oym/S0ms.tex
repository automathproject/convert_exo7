  \texte{}
\begin{enumerate}
  \item \question{Montrer que l'équation : $x^3+y^3-3xy=1$ définit au voisinage de $0$ une fonction
    implicite : $y=\varphi(x)$ telle que $\varphi(0)=1$.}
  \item \question{Donner le DL de $\varphi$ en $0$ à l'ordre $3$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\varphi(x) = 1 + x -\frac23x^3 + o (x^3)$.}
\end{enumerate}
