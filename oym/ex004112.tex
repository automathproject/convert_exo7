\begin{enumerate}
  \item \question{$xy'+y = xy^3$.}
  \item \question{$2xy'+y = \frac{2x^2}{y^3}$.}
  \item \question{$\sqrt x y' - y + (x+2\sqrt x)\sqrt y = 0$.}
  \item \question{$xy'+y = (xy)^{3/2}$.}
  \item \question{$x^3y' = y(3x^2+y^2)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$y = \frac{\pm1}{\sqrt{\lambda x^2+2x}}$ ou $y = 0$.}
  \item \reponse{$y = \pm\sqrt[4]{x^2+\frac\lambda{x^2}}$.}
  \item \reponse{$y = ( (\sqrt x + 2)^2 + \lambda e^{\sqrt x}\, )^2$.}
  \item \reponse{$y = \frac 1x\left(\frac2{\lambda-x}\right)^2$ ou $y = 0$.}
  \item \reponse{$y = \pm\frac{\sqrt2\,x^3}{\sqrt{2\lambda-x^4}}$ ou $y=0$.}
\end{enumerate}
