  \texte{}
  \question{Soient $d,m \in \N^*$. Donner une condition n{\'e}c{\'e}ssaire et suffisante sur
$d$ et $m$ pour qu'il existe $a,b \in \Z$ tels que $a\wedge b = d$
et $a\vee b = m$.

R{\'e}soudre ce probl{\`e}me pour $d = 50$ et $m = 600$.}
  \reponse{$\{a,b\} \in \bigl\{\{50,600\},\{150,200\}\bigr\}$.}
