  \texte{}
  \question{\'Etudier $ \int_{t=0}^{+\infty} \frac{e^{-tx}}{1+t^2}\,d t$.}
  \reponse{Fonction de $x$ de classe $\mathcal{C}^\infty$ sur $]0,+\infty[$, décroissante
de limite $\pi/2$ en $0^+$ et $0$ en $+\infty$. Demi-tangente verticale en~$0^+$,
\'Equivalente à $1/x$ en $+\infty$ (par IPP). \'Equation différentielle~:
$f(x) + f''(x) = 1/x$.}
}
