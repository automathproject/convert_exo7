\titre{$u^2 = 0$ en dimension 3}

\contenu{
  \texte{}
  \question{Soit $E$ un ev de dimension 3 et $u\in \mathcal{L}(E)$ tel que $u^2 = 0$.
Montrer qu'il existe $f \in E^*$ et $\vec a \in E$ tels que :
$\forall\ \vec x \in E,\ u(\vec x\,) = f(\vec x\,)\vec a$.}
