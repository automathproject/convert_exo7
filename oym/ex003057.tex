\contenu{
  \texte{On note $a_n = 25^n + 2^{3n+4}$.}
\begin{enumerate}
  \item \question{Trouver $a,b\in\Z$ tels que :
     $\forall\ n\in\N,\ a_{n+2} = a.a_{n+1} + b.a_n$.}
  \item \question{En d{\'e}duire que : $\forall\ n \in \N,\ a_n$ est divisible par $17$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$a=33$, $b=-200$.}
\end{enumerate}
