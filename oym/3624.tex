  \texte{}
  \question{On donne une matrice carrée réelle $M$ d'ordre $n$. Soient $\alpha,\beta$ les
multiplicités de zéro dans $\chi_M$ et $\mu_M$. Montrer que $\dim(\mathrm{Ker} M) = \alpha$
si et seulement si $\beta = 1$.}
  \reponse{Trigonaliser fortement $M$.}
