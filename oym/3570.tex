\titre{$f  \mapsto p\circ f\circ p$}
\theme{Exercices de Michel Quercia, Réductions des endomorphismes}
\auteur{quercia}
\date{2010/03/10}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $p \in \mathcal{L}(E)$ une projection et
$\Phi : {\mathcal{L}(E)} \to {\mathcal{L}(E)}, f \mapsto {p\circ f\circ p.}$

Déterminer les éléments propres de $\Phi$.}
  \reponse{$\lambda = 1$ : Dir$(p) \subset \mathrm{Ker} f$,\quad $\Im f \subset \text{Base}(p)$.

$\lambda = 0$ : $f\bigl(\text{Base}(p)\bigr) \subset \text{Dir}(p)$.}
}
