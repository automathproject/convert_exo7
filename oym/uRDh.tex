\titre{$\sum\sin(n)/n$}
\theme{Exercices de Michel Quercia, Suites et séries de fonctions}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{Pour $n\in\N^*$ et $x\in[-1,1]$ on pose $u_n(x)=\frac{x^n\sin(nx)}n$.}
\begin{enumerate}
  \item \question{Montrer que la série $\sum_{n=1}^\infty u_n(x)$ converge uniformément
    sur $[-1,1]$ vers une fonction continue, $f$.}
  \item \question{Justifier la dérivabilité de~$f$ sur $]-1,1[$ et calculer $f'(x)$.
    En déduire $f(x)$.}
  \item \question{En déduire la valeur de $\sum_{n=1}^\infty \frac{\sin n}n$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Transformation d'Abel.}
  \item \reponse{$f(x) = \Arctan\left(\frac{x\sin x}{1-x\cos x}\right)$.}
  \item \reponse{$\frac{\pi-1}2$.}
\end{enumerate}
