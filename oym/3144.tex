\contenu{
  \texte{}
  \question{Soit $x=\frac pq \in \Q^*$ avec $p,q$ entiers, $q\ge 1$, $p\wedge q = 1$.
Calculer $\sum_{k=0}^{q-1} \text{frac}(kx)$.}
  \reponse{$=\frac{q-1}2$.}
}
