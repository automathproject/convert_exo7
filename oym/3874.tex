\titre{$f$ croissante et $f\circ f = \mathrm{id}$}
\theme{Exercices de Michel Quercia, Fonctions monotones}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : \R \to \R$ croissante telle que :
$\forall\ x \in \R,\ f\circ f(x) = x$.
Montrer que : $\forall\ x \in \R,\ f(x) = x$.}
  \reponse{}
}
