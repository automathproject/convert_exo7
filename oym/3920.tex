\contenu{
  \texte{}
  \question{Simplifier $\cos(3\arctan x)$ et $\cos^2\left(\frac 12\arctan x\right)$.}
  \reponse{$\cos(3\arctan x) = \frac {1-3x^2}{(1+x^2)^{3/2}}$.
\qquad
$\cos^2\left(\frac 12\arctan x\right) = \frac 12\left(1+\frac1{\sqrt{1+x^2}}\right)$.}
}
