\theme{Exercices de Michel Quercia, Fonctions circulaires inverses}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Existe-t-il une fonction $f : D \to \R$ telle que :
$\forall\ x \in D,\ 2\arcsin x + \arcsin f(x) = \frac \pi6$ ?}
  \reponse{$D = \left[-\frac 12, \frac{\sqrt3}2\right]$,\quad
   $f(x) = \frac12 - x^2- x\sqrt3\sqrt{1-x^2}$.}
}
