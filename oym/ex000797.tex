  \texte{}
  \question{Soit $f:[0,1]\rightarrow {\Rr}$ une application continue par morceaux,
continue en $0,$ trouver une suite $(g_{n})_{n\in \Nn}$ de fonctions en
escaliers telle que :
$$\lim_{n\rightarrow \infty }\int_{0}^{1}f(t)g_{n}(t)dt=f(0). $$}

}
