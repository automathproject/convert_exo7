\contenu{
  \texte{Soit $S\subset \N$ fini et $P=\sum_{s\in S} a_sX^s\in\C[X]$.}
\begin{enumerate}
  \item \question{On suppose que les $a_s$ sont r{\'e}els. Montrer que $P$ a moins de racines strictement positives distinctes que la suite 
$(a_s)$ n'a de changement de signe.}
  \item \question{On suppose que $P$ v{\'e}rifie~: $\forall\ s\in S,\ P(s)=0$. Montrer que $P$ est nul.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{R{\'e}currence sur $\mathrm{Card}\,(S)$ en mettant le terme de plus bas degr{\'e} en
facteur et en d{\'e}rivant le quotient.}
  \item \reponse{Appliquer la question pr{\'e}c{\'e}dente aux suites $(\Re(a_s))$ et $(\Im(a_s))$.}
\end{enumerate}
