\titre{Expression d'une primitive $n$-ème de $f$}

\contenu{
  \texte{}
  \question{Soit $f : {[a,b]} \to \R$ continue et
$g(x) =  \int_{t=a}^x \frac {(x-t)^{n-1}}{(n-1)!} f(t)\,d t$.
Montrer que $g^{(n)} = f$.}
