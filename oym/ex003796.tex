\titre{$(u(x)\mid x)$ est réel}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{Soit $E$ un espace vectoriel hermitien et $u\in\mathcal{L}(E)$.}
\begin{enumerate}
  \item \question{Montrer que $u=u^*$ si et seulement si pour tout~$x\in E$,
    $(u(x)\mid x)$ est réel.}
  \item \question{On suppose $u$ autoadjoint positif. Montrer~:
    $\forall\ x\in E,\ \|u(x)\|^4 \le (x\mid u(x))\times(u(x)\mid u^2(x))$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$((u-u^*)(x)\mid x) = 0$.}
  \item \reponse{Orthodiagonaliser et appliquer l'inégalité de Cauchy-Schwarz.}
\end{enumerate}
