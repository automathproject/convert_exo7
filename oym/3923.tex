  \texte{}
  \question{Simplifier
$\frac x2 - \arcsin\sqrt{\frac{1+\sin x}2}$ pour $x \in {[-\pi,\pi]}$.}
  \reponse{$f(x)=x+\pi/4$ pour $x\in ]-\pi,-\pi/2[$ ; \\
$f(x)=-\pi/4$ pour $x\in ]-\pi/2,\pi/2[$ ; \\
$f(x)=x-3\pi/4$ pour $x\in ]\pi/2,\pi[$.}
}
