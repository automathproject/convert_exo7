\titre{Angles $> 2\pi/3$}
\theme{Exercices de Michel Quercia, Produit scalaire}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $E$ un espace euclidien de dimension supérieure ou égale à 3.
Existe-t-il trois vecteurs $\vec u_1,\vec u_2,\vec u_3$ unitaires
faisant entre eux deux à deux des angles strictement supérieurs à
$\frac {2\pi}3$ ?}
  \reponse{Non, $\|\vec u_1+\vec u_2+\vec u_3\|^2 < 0$.}
}
