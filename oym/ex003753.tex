\titre{$A{}^t\!A + A + {}^t\!A = 0$}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
\begin{enumerate}
  \item \question{Trouver les matrices $A \in \mathcal{M}_n(\R)$ telles que :
    $A{}^t\!A + A + {}^t\!A = 0$.}
  \item \question{Montrer que pour une telle matrice, $|\det A| \le 2^n$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$A = P-I$, $P \in {\cal O}(n)$.}
  \item \reponse{Hadamard.}
\end{enumerate}
}
