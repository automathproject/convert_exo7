  \question{Chercher $\lim_{n\to\infty}  \int_0^1\frac{t^n-t^{2n}}{1-t}\,d t$.}
  \reponse{$ \int_0^1\frac{t^n-t^{2n}}{1-t}\,d t = \frac1{n+1} + \dots + \frac1{2n}
              \to \ln 2$ lorsque $n\to\infty$.}
