\titre{$\arcsin \circ \arctan - \arctan \circ \arcsin$}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
\begin{enumerate}
  \item \question{Soient $P(X) = X + aX^3 + bX^5 + cX^7$ et $Q(X) = X + \alpha X^3 + \beta X^5 + \gamma X^7$.
    Chercher la partie de degré inférieur ou égale à 7 de $P\circ Q - Q \circ P$.}
  \item \question{Application : Donner le DL à l'ordre 7 en 0 de
    $\arcsin(\arctan x) - \arctan(\arcsin x)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$(3a\alpha(\alpha-a) + 2(b\alpha-a\beta))X^7$.}
  \item \reponse{$-x^7/30 + o_{x\to0}(x^7)$.}
\end{enumerate}
}
