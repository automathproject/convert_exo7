\contenu{
  \texte{}
  \question{Soient $A,B,C,D \in \mathcal{M}_n(K)$ avec $A$ inversible et $AC=CA$. On considère
$M = \begin{pmatrix} A & B \cr C & D \cr \end{pmatrix} \in \mathcal{M}_{2n}(K)$.
Montrer que $\det M = \det(AD-CB)$.}
