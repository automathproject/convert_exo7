\contenu{
  \texte{Déterminer les séries de Laurent et les
résidus à l'origine des fonctions suivantes:}
\begin{enumerate}
  \item \question{$f(z) = \frac1z$}
  \item \question{$f(z) = \frac1{z^2+1}$}
  \item \question{$f(z) = \frac1{z(z^2+1)}$}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Le r\'esidu est $a_{-1} =1$.}
  \item \reponse{$f(z) =\frac{1}{1+z^2}=\sum_{k=0}^\infty (-1)^k z^{2k}$  pour $ |z|< 1$.
C'est une s\'erie enti\`ere car $f$ est holomorphe dans le disque unit\'e et $\mathrm{Res} (f ,0)=0$.}
  \item \reponse{$f(z) = \frac{1}{z}\frac{1}{1+z^2} =\sum_{k=0}^\infty (-1)^k z^{2k-1} =\frac{1}{z}-z +z^3 ....$.et $\mathrm{Res} (f ,0)=0$}
\end{enumerate}
}
