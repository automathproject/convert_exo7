\contenu{
  \texte{Factoriser $X^n-1$ sur $\C$.}
\begin{enumerate}
  \item \question{En d{\'e}duire $\prod_{k=1}^{n-1} \sin\left(\frac {k\pi}n\right)$.}
  \item \question{Calculer {\'e}galement $\prod_{k=0}^{n-1} \sin\left(\frac {k\pi}n + \theta\right)$.}
  \item \question{On note $\omega = e^{2i\pi/n}$. Calculer
    $\prod_{0 \le k,\ell < n, k \ne \ell} (\omega^k - \omega^\ell)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\frac n{2^{n-1}}$.}
  \item \reponse{$\frac {\sin(n\theta)}{2^{n-1}}$.}
  \item \reponse{$-(-n)^n$.}
\end{enumerate}
