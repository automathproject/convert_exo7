  \texte{}
  \question{Nature de la série de terme général $u_n=\sum_{k=1}^{n}\frac{1}{(n+k)^p}$, $p\in]0,+\infty[$.}
  \reponse{Pour tout entier naturel non nul $n$, $0<\frac{1}{2^pn^{p-1}}=\sum_{k=1}^{n}\frac{1}{(2n)^p}\leqslant\sum_{k=1}^{n}\frac{1}{(n+k)^p}\leqslant\sum_{k=1}^{n}\frac{1}{n^p}=\frac{1}{n^{p-1}}$ et la série de terme général $u_n$ converge si et seulement si $p > 2$.}
}
