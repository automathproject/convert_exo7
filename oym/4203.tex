\titre{$x\frac{\partial f}{\partial x} = y\frac{\partial f}{\partial y}$}
\theme{Exercices de Michel Quercia, \'Equations aux dérivées partielles}
\auteur{quercia}
\date{2010/03/11}

\contenu{
  \texte{}
  \question{Résoudre sur $(\R^{+*})^2$ : $x\frac{\partial f}{\partial x} = y\frac{\partial f}{\partial y}$,
en posant $\begin{cases} u = xy \cr v = \frac xy. \end{cases}$}
  \reponse{$f(x,y) = g(xy)$.}
}
