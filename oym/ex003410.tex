\contenu{
  \texte{}
  \question{\'Etudier l'existence de solutions du système :
$$\left\{\begin{array}{ccccccc} x     &+& my &+& z      &=& 1  \cr
           (m+1)x &+& 2y &+& (m-3)z &=& -1 \cr
           (m-1)x & &    &-& 3z     &=& -1.\cr \end{array}\right.$$}
  \reponse{Système de Cramer ssi $m\ne 1,\pm2i$; compatible ssi $m\ne 1$.}
}
