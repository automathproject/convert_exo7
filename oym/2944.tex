  \texte{}
  \question{Simplifier $x = \prod_{p=2}^n\frac {p^3-1}{p^3+1}$ en utilisant $1,j,j^2$.}
  \reponse{$x = \frac {2(n^2+n+1)}{3n(n+1)}$.}
