  \texte{}
  \question{Soit $f : {\R} \to {\R}$ non identiquement nulle telle que :
$\forall\ x,y \in \R,\ \begin{cases} f(x+y) &= f(x) + f(y) \cr
                               f(xy)  &= f(x)f(y).   \cr\end{cases}$

Montrer que $f$ est croissante, puis $f = \mathrm{id}_{\R}$.}
  \reponse{}
