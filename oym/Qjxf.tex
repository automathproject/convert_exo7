\contenu{
  \texte{Soit $\mathbb{A}$ une $\R$-algèbre commutative, intègre et de dimension finie.}
\begin{enumerate}
  \item \question{Montrer que $\mathbb{A}$ est un corps.}
  \item \question{Si $\dim{\mathbb{A}} > 1$ montrer que tout élément de $\mathbb{A}$ est
    algébrique de degré 1 ou 2 sur $\R$. En déduire qu'alors $\mathbb{A}$
    est isomorphe à~$\C$.}
\end{enumerate}
