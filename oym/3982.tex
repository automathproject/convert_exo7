\contenu{
  \texte{}
  \question{Soient $x_1, x_2, \dots, x_n > 0$.
Montrer que $\frac {x_1}{x_2} + \frac{x_2}{x_3} + \dots + \frac{x_n}{x_1} \ge n$.}
  \reponse{$y_i = \frac{x_i}{x_{i+1}}  \Rightarrow 
          \frac n {y_1\cdots y_n} \le \frac{y_1 + \dots + y_n}n$.}
}
