\titre{Fronti{\`e}re d'un ouvert}

\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $U$ un ouvert d'un espace vectoriel norm{\'e}. Montrer que la fronti{\`e}re de $U$ est d'int{\'e}rieur vide.}
  \reponse{Soit $a$ int{\'e}rieur {\`a} $\text{Fr}(U)$ : Il existe $B(a,r) \subset \overline U\setminus U$.
$B(a,r) \cap U = \varnothing  \Rightarrow  a \notin \overline U$, contradiction.}
}
