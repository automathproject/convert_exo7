\titre{$f(x+y)f(x-y) \le f^2(x)$}

\contenu{
  \texte{}
  \question{Soit $f : \R \to \R$ de classe $\mathcal{C}^2$ telle que :
$\forall\ x,y \in \R,\ f(x+y)f(x-y) \le f^2(x)$.

Montrer que : $\forall\ t \in \R,\ f(t)f''(t) \le f'^2(t)$.}
