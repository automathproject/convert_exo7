\titre{Centre du cercle circonscrit}
\theme{Exercices de Michel Quercia, Nombres complexes}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient $a,b,c \in \C$, affixes de points $A,B,C$ non align{\'e}s.
Calculer l'affixe du centre du cercle circonscrit {\`a} $ABC$ en fonction
de $a,b,c$.}
  \reponse{$\omega =
          \frac {a(c\bar c-b\bar b) + b(a\bar a-c\bar c) + c(b\bar b-a\bar a)}
                 {a(\bar c-\bar b) + b(\bar a-\bar c) + c(\bar b-\bar a)}$.}
}
