  \texte{Soit $A$ une partie de l'espace métrique $(\Cc,|~~|)$ et $z$ un nombre complexe.}
\begin{enumerate}
  \item \question{Montrer que $z$ appartient $\mathring{A}$ si et seulement si il existe un nombre réel $\epsilon>0$ 
 tel que $B(z,\epsilon)$ soit incluse dans $A$.}
  \item \question{Montrer que $z$ appartient à $\overline{A}$ si et seulement si pour tout réel $\epsilon>0$,
 la boule $B(z,\epsilon)$ rencontre~$A$.}
