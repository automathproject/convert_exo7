  \texte{}
  \question{On pose $E=C([0,1],\R)$ et on le munit de la norme $N_{\infty}$. Soit 
$F=\{f\in E \, | \, f(0)=f(1)\}$.
D{\'e}terminer l'adh{\'e}rence et l'int{\'e}rieur de $F$.}
  \reponse{$\overline F = F$, $\mathring F = \varnothing$.}
}
