  \texte{}
  \question{Résoudre le système différentiel suivant :
$$
\left\{
\begin{array}{rcrcrcr}
\frac{dx}{dt} &=& 2x(t) &+& 2y(t) &+& 2z(t) \\
\frac{dy}{dt} &=&  x(t) &+& 3y(t) &+& 2z(t) \\
\frac{dz}{dt} &=& -x(t) &-&  y(t) &-&  z(t)
\end{array}
\right.
$$


Donner toutes les solutions qui satisfont $x(0)=1$, $y(0)=2$, $z(0)=-1$.}
  \reponse{}
}
