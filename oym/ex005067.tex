  \question{Calculer $\cos\frac{\pi}{8}$ et $\sin\frac{\pi}{8}$.}
  \reponse{$\cos^2\frac{\pi}{8}=\frac{1}{2}\left(1+\cos(2\times\frac{\pi}{8})\right)=\frac{1}{2}\left(1+\frac{\sqrt{2}}{2}\right)=\frac{2+\sqrt{2}}{4}$,
et puisque $\cos\frac{\pi}{8}>0$,

\begin{center}
\shadowbox{$\cos\frac{\pi}{8}=\frac{1}{2}\sqrt{2+\sqrt{2}}$.}
\end{center}
