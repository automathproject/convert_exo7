\contenu{
  \texte{Soit $ABC$ un triangle.
On note : $a = BC$, $b = CA$, $c = AB$,}
\begin{enumerate}
  \item \question{Soit $A'$ le pied de la bissectrice intérieure issue de $A$.
    Montrer que $\frac{A'B}{A'C} = \frac cb$.}
  \item \question{En déduire les coordonnées barycentriques de $I$, centre du cercle inscrit.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Soit $\alpha' = \overline{(\vec{A'A},\vec{A'B})}$ :
             $\frac{A'B}{\sin(\alpha/2)} = \frac{AB}{\sin(\alpha')}$,
             $\frac{A'C}{\sin(\alpha/2)} = \frac{AC}{\sin(\pi-\alpha')}$.}
  \item \reponse{$I=\text{Bar}(A:a,B:b,C:c)$.}
\end{enumerate}
