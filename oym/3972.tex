\titre{Dérivées de $e^{-1/x^2}$}

\contenu{
  \texte{}
  \question{Soit $f : \R \to \R$
$$ \begin{cases}x \ne 0 &f(x) = \exp\left(-\frac 1{x^2}\right)\cr
                            &f(0)=0.\cr\end{cases}$$
Montrer que $f$ est de classe $\mathcal{C}^\infty$ en $0$ et : $\forall\ k \in \N,\ f^{(k)}(0) = 0$.}
  \reponse{}
