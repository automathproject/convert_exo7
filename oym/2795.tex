\contenu{
  \texte{}
  \question{Déterminer les séries de Taylor à l'origine de
$\frac1{1-z}$, $\frac1{(1-z)^2}$, $\frac1{(1-z)^3}$,
$\frac1{(1-z)^4}$.}
  \reponse{$$\frac{1}{1-z} =\sum_{k=0}^\infty z^k \quad \quad |z|<1.$$
$$\frac{1}{(1-z)^2}=\left( \frac{1}{1-z}\right)' = \sum_{k=1}^\infty k z^{k-1} \quad \text{pour} \quad |z|<1.$$
etc.}
}
