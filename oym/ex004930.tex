\contenu{
  \texte{}
\begin{enumerate}
  \item \question{Soit $D$ la droite d'équations $\begin{cases} y = 1\cr x = \lambda z\cr\end{cases}$ où
    $\lambda$ est un réel non nul fixé.
    Déterminer une équation cartésienne et la nature de la surface ${\cal S}$
    engendrée par la rotation de $D$ autour de $Oz$.}
  \item \question{En déduire que tout hyperboloïde de révolution à une nappe est réunion
    d'une famille de droites ({\it surface réglée\/}).}
  \item \question{Généraliser à un hyperboloïde à une nappe quelconque.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$x^2+y^2 = 1 + \lambda^2 z^2$.}
\end{enumerate}
