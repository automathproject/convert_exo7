\contenu{
  \texte{}
  \question{Soit $P \in { K[X]}$ tel que les restes des divisions de $P$ par $X^2+1$
et $X^2-1$ valent respectivement $2X-2$ et $-4X$.
Quel est le reste de la division de $P$ par $X^4-1$ ?}
  \reponse{$-3X^3 + X^2 - X - 1$.}
}
