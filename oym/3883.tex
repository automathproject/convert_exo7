\titre{Point fixe pour une application croissante}

\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {[0,1]} \to {[0,1]}$ croissante. Montrer qu'il existe $x \in {[0,1]}$
tel que $f(x) = x$.

(\'Etudier $A = \{ x \in [0,1] \text{ tq } f(x) \le x \}$)}
  \reponse{$\alpha = \inf(A)  \Rightarrow  f(\alpha^+) \le \alpha \text{ et } f(\alpha^-) \ge \alpha$.}
}
