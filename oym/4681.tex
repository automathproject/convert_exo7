\titre{Suite $a^n/\prod(1+a^k)$}
\theme{Exercices de Michel Quercia, Suites convergentes}
\auteur{quercia}
\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $a \in \C\setminus \mathbb{U}$. {\'E}tudier la suite de terme g{\'e}n{\'e}ral :
$u_n = \frac {a^n}{(1+a)(1+a^2)\dots(1+a^n)}$.}
  \reponse{Si $|a| < 1$, {\`a} partir d'un certain rang, $\left|\frac a{1+a^n} \right|<\alpha < 1$
$ \Rightarrow  \lim = 0$.

Si $|a| >1$, $\frac{u_{n+1}}{u_n} \to 0  \Rightarrow  u_n \to 0$.}
}
