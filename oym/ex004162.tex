  \texte{}
  \question{Soit $f : {\R^3} \to {\R^3}, {(x,y,z)} \mapsto {(e^{2y}+e^{2z}, e^{2x}-e^{2z}, x-y).}$
Montrer que $f$ induit un $\mathcal{C}^1$-difféomorphisme de $\R^3$ sur un ouvert à préciser.}
  \reponse{$f(\R^3) = \{ (u,v,w) \text{ tq } u+v > 0,\ ue^{2w}-v > 0\}$.}
