  \texte{}
  \question{Établir la convergence puis calculer la valeur de
$ \int_{t=0}^1 \frac{\ln(t^2)\ln(1-t^2)}{t^2}\,d t$.}
  \reponse{Développer en série entière $\ln(1-t^2)$.
$I = \frac{\pi^2}2 - 4\ln 2$.}
