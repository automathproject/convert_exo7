\titre{$\int_0^\infty \text{périodique}/t\,d t$}

\contenu{
  \texte{}
  \question{Soit $f:\R \to \R$ continue, périodique de période $T > 0$.
On note $m = \frac1T \int_{t=0}^T f(t)\,d t$. Montrer que
$ \int_{t=T}^{+\infty} \frac{f(t)}t\,d t$ converge si et seulement
si $m=0$.}
  \reponse{}
}
