\titre{$\ln(n) + a\ln(n+1) + b\ln(n+2)$}
\theme{Exercices de Michel Quercia, Séries numérique}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Pour quelles valeurs de $a,b\in \R$ la série de terme général
$\ln(n) + a\ln(n+1) + b\ln(n+2)$ est-elle convergente~?
Calculer alors la somme de la série.}
  \reponse{$\begin{cases} a+b = -1 \cr a+2b = 0\cr \end{cases} \Leftrightarrow a=-2$, $b=1$, $S = -\ln 2$.}
}
