\contenu{
  \texte{}
  \question{\'Etudier l'existence de solutions du système :
 $$\left\{\begin{array}{ccccccc} x      &+&  y &+& (1-m)z  &=& m+2  \cr
           (1+m)x  &-&  y &+& 2z      &=& 0    \cr
           2x      &-& my &+& 3z      &=& m+2. \cr\end{array}\right.$$}
  \reponse{Système de Cramer ssi $m\ne 0,\pm2$; compatible ssi $m\ne2$.}
}
