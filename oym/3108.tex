\titre{$q\mid2^p-1$}
\theme{Exercices de Michel Quercia, Congruences}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient $p,q$ premiers impairs tels que $q\mid2^p-1$.
Montrer que $q\equiv 1 (\mathrm{mod}\,{2p})$.}
  \reponse{L'ordre de $\dot 2$ dans $(\Z/q\Z)^*$ divise $p$ donc est
         {\'e}gal {\`a}~$p$ et cet ordre divise $\varphi(q) = q-1$.}
}
