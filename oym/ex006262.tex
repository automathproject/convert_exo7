  \texte{Soient $E,F$ des espaces de Banach, $\Omega$ un ouvert connexe de $E$
et $f_n: \Omega \to F$ une suite d'applications différentiables.
On suppose que cette suite vérifie :}
\begin{enumerate}
  \item \question{[(i)] Il existe $x_0\in \Omega$ tel que $(f_n (x_0))$ converge dans $F$.}
  \item \question{[(ii)] La suite $(Df_n)$ converge uniformément sur toute boule fermée
$B_F (a,r) \subset \Omega$.}
\end{enumerate}
