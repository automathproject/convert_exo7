\contenu{
  \texte{}
  \question{Soient $A,B\in \mathcal{M}_n(\R)$. Montrer que :
$\exp(A) - \exp(B) =  \int_{s=0}^1\exp(sA) (A-B) \exp((1-s)B)\,d s$.}
  \reponse{Développer en séries sous l'intégrale, multiplier, permuter
avec l'intégrale puis simplifier.}
}
