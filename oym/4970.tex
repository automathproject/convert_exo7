\contenu{
  \texte{}
  \question{Dans un \emph{rond} on donne les plans
$P : x+y+z = 1$ et $Q : 2x-y+z=1$.
Chercher une équation cartésienne du plan $Q'$ symétrique de $Q$ par rapport
à $P$.}
  \reponse{$2x-7y-z=-1$.}
}
