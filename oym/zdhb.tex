\contenu{
  \texte{Résoudre dans $\Rr$ puis dans $[0,2\pi]$ les équations suivantes~:}
\begin{enumerate}
  \item \question{$\sin x=0$,}
  \item \question{$\sin x=1$,}
  \item \question{$\sin x=-1$,}
  \item \question{$\cos x=1$,}
  \item \question{$\cos x=-1$,}
  \item \question{$\cos x=0$,}
  \item \question{$\tan x=0$,}
  \item \question{$\tan x=1$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\sin x=0\Leftrightarrow x\in\pi\Zz$. De plus, $\mathcal{S}_{[0,2\pi]}=\{0,\pi,2\pi\}$.}
  \item \reponse{$\sin x=1\Leftrightarrow x\in\frac{\pi}{2}+2\pi\Zz$. De plus, $\mathcal{S}_{[0,2\pi]}=\left\{\frac{\pi}{2}\right\}$.}
  \item \reponse{$\sin x=-1\Leftrightarrow x\in-\frac{\pi}{2}+2\pi\Zz$. De plus, $\mathcal{S}_{[0,2\pi]}=\left\{\frac{3\pi}{2}\right\}$.}
  \item \reponse{$\cos x=1\Leftrightarrow x\in2\pi\Zz$. De plus, $\mathcal{S}_{[0,2\pi]}=\{0,2\pi\}$.}
  \item \reponse{$\cos x=-1\Leftrightarrow x\in\pi+2\pi\Zz$. De plus, $\mathcal{S}_{[0,2\pi]}=\{\pi\}$.}
  \item \reponse{$\cos x=0\Leftrightarrow x\in\frac{\pi}{2}+\pi\Zz$. De plus,
$\mathcal{S}_{[0,2\pi]}=\left\{\frac{\pi}{2},\frac{3\pi}{2}\right\}$.}
  \item \reponse{$\tan x=0\Leftrightarrow x\in\pi\Zz$. De plus, $\mathcal{S}_{[0,2\pi]}=\left\{0,\pi,2\pi\right\}$.}
  \item \reponse{$\tan x=1\Leftrightarrow x\in\frac{\pi}{4}+\pi\Zz$. De plus,
$\mathcal{S}_{[0,2\pi]}=\left\{\frac{\pi}{4},\frac{5\pi}{4}\right\}$.}
\end{enumerate}
