\contenu{
  \texte{D{\'e}terminer les nombres $z \in \C$ tels que $\dots$}
\begin{enumerate}
  \item \question{$z,z^2,z^4$ sont align{\'e}s.}
  \item \question{$1,z,z^2$ forment un triangle rectangle.}
  \item \question{$z,\frac 1z,-i$ sont align{\'e}s.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$z \in \R$ ou $z \in -\frac 12 + i\R$.}
  \item \reponse{$z \in -1 + i\R$ ou $z \in i\R$
             ou $\left|z+\frac 12\right| = \frac 12$.}
  \item \reponse{$z\in i\R$ ou $|z-i| = \sqrt2$.}
\end{enumerate}
