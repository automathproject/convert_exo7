  \question{\'Etudier en $ + \infty$ et $-\infty$ la fonction
$f (x) = \sqrt[3]{x^3 + 1}
 + \sqrt{x^2 + x + 1}$.}
  \reponse{$$\lim _{x\rightarrow \infty }\sqrt [3]{{x}^{3}+1}-\sqrt {{x}^{2}+x+1}=-
1/2.$$}
