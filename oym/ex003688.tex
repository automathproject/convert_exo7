\contenu{
  \texte{On munit $\R_2[X]$ du produit scalaire : $(P\mid Q) =  \int_{t=0}^1 PQ(t)\,d t$.}
\begin{enumerate}
  \item \question{Vérifier que c'est effectivement un produit scalaire.}
  \item \question{Soit $\varphi : {\R_2[X]} \to {\R}, P \mapsto {P(0).}$
    Trouver le polynôme $A$ tel que : $\forall\ P \in \R_2[X],\
    \varphi(P) = (A\mid P)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$30X^2-36X+9$.}
\end{enumerate}
