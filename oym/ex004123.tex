\titre{Intervalle maximal pour $y' = f(y)$}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f:\R \to \R$ de classe $\mathcal{C}^1$ strictement positive et $y$ la solution
maximale définie sur $]\alpha,\beta[$ du problème de Cauchy : $y' = f(y)$,
$y(x_0) = y_0$.
Montrer que $\beta = x_0 +  \int_{t=y_0}^{+\infty} \frac{d t}{f(t)}$ et que
    $y \to +\infty$ lorsque $x\to\beta^-$.}
  \reponse{}
