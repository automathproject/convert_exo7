\contenu{
  \texte{Soit $\sum a_nz^n$ une série entière de rayon de convergence $R > 0$.
Déterminer les rayons de convergence des séries :}
\begin{enumerate}
  \item \question{$\sum a_n^2 z^n$.}
  \item \question{$\sum \frac {a_n}{n!} z^n$.}
  \item \question{$\sum \frac{n!\,a_n}{n^n}z^n$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$R' = R^2$.}
  \item \reponse{$R'=\infty$.}
  \item \reponse{$R' = eR$.}
\end{enumerate}
