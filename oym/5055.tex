\contenu{
  \texte{}
  \question{Soit $\mathcal{C}$ le cercle intersection de la sphère d'équation $x^2+y^2+z^2 = 1$
et du plan d'équation $x+y=1$, et $S=(1,1,1)$.
Déterminer l'équation cartésienne du cône de sommet $S$ s'appuyant sur $\mathcal{C}$.}
  \reponse{$x^2+y^2+z^2-2xz-2yz+2z=1$.}
}
