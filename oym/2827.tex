\contenu{
  \texte{}
  \question{Soit $F$ une fonction entière telle que $|F(z)|\leq \frac1n$
pour $|z| = n$, $n\geq1$. Montrer que $F$ est identiquement
nulle.}
  \reponse{Supposons qu'il existe $z\in \C$ avec $F(z) \neq 0$ et notons $\alpha = |F(z)|>0$. Si $n>1$ tel que $\frac{1}{n} <\alpha $, alors le principe du maximum affirme que
$$|F(z) | < \frac{1}{n} < \alpha \quad \text{pour tout} \;\; |z| < n.$$
Contradiction.}
}
