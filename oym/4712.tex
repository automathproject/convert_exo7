\contenu{
  \texte{}
  \question{{\'E}tudier la convergence des suites $(x_n)$, $(y_n)$ d{\'e}finies par :
$0 < y_0 < x_0$ et $\begin{cases}x_{n+1} = \frac{x_n+y_n}2 \cr
                           y_{n+1} = \frac{2x_ny_n}{x_n+y_n} .\cr\end{cases}$}
  \reponse{$x_ny_n = \text{c}^{\text{te}}  \Rightarrow  x_n,y_n \to \sqrt{x_0y_0}$.}
}
