\titre{{\'E}quation $X(X-1)P' + P^2 -(2X+1)P + 2X = 0$}
\theme{}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Trouver tous les polyn{\^o}mes $P \in { K[X]}$ tels que :
$X(X-1)P' + P^2 -(2X+1)P + 2X = 0$.}
  \reponse{$\deg P < 2  \Rightarrow  P \in \{1,X,X+1\}$.}
}
