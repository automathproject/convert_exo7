\contenu{
  \texte{}
  \question{Pour $\theta \in \R$, on note $A_\theta = (\cos\theta,\sin\theta)$.
Chercher l'enveloppe des droites $D_\theta = (A_\theta\ A_{2\theta})$.}
  \reponse{$3x = \cos2\theta + 2\cos\theta$, $3y = \sin2\theta + 2\sin\theta$ :
         cardioïde à rebroussement en $(-1/3,0)$.}
}
