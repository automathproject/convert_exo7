\contenu{
  \texte{}
  \question{Soit $X$ une variable aléatoire réelle dans $L^2$.
Montrer que $\displaystyle \text{Var}(X)=\min_{t\in\Rr}E((X-t)^2)$.}
  \reponse{Soit $g(t)=E((X-t)^2)=E(X^2)-2tE(X)+t^2$. C'est une parabole dont le
minimum est en $t=E(X)$ (on peut aussi dériver $g$ et trouver le minimum
de cette manière).}
}
