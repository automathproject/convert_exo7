  \item \question{Effectuer la division euclidienne de $A$ par $B$ :
\begin{enumerate}}
  \item \question{$A=3X^5+4X^2+1,\ B=X^2+2X+3$}
  \item \question{$A=3X^5+2X^4-X^2+1,\ B=X^3+X+2$}
  \item \question{$A=X^4-X^3+X-2,\ B=X^2-2X+4$}
  \item \question{$A=X^5-7X^4-X^2-9X+9,\ B=X^2-5X+4$}
\end{enumerate}

  \item \reponse{\begin{enumerate}}

  \item \reponse{$X^4-X^3+X-2=(X^2-2X+4)(X^2+X-2)-7X+6$}
  \item \reponse{$X^5-7X^4-X^2-9X+9$\\
