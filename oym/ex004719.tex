\titre{$\sqrt m - \sqrt n$}
\theme{}
\auteur{quercia}
\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Montrer que l'ensemble $A = \{ \sqrt m - \sqrt n \text{ tq } m,n \in \N \}$ est dense dans
$\R$.}
  \reponse{Soient $x<y$ : 
         Il existe $a \in \N$ tel que $n\ge a  \Rightarrow  \sqrt{n+1} - \sqrt{n} < y-x$.\par
         Il existe $b \in \N$ tel que $\sqrt a - \sqrt b < x$.\par
         Alors il existe $c \ge a$ tel que $x < \sqrt c - \sqrt b < y$.\par}
}
