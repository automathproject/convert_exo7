  \texte{}
\begin{enumerate}
  \item \question{Les groupes $\Z/9\Z$ et $\Z/3\Z\times \Z/3\Z$ sont-ils isomorphes ?}
  \item \question{Les groupes $\mathbb{F}_7$ et $\Z/2\Z\times \Z/3\Z$ sont-ils isomorphes ? Justifier.}
  \item \question{Les groupes $(\mathbb{F}_7)^\star$ et $\Z/2\Z\times \Z/3\Z$ sont-ils isomorphes ? Justifier.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Non, car $\Z/3\Z\times \Z/3\Z$ ne contient aucun élément d'ordre 9.}
  \item \reponse{Non, car ils n'ont pas le même ordre.}
  \item \reponse{Les groupes $(\mathbb{F}_7)^\star$ et $\Z/2\Z\times \Z/3\Z$ sont-ils isomorphes ? Par le théorème sur le groupe des inversibles d'un corps fini et par le théorème chinois, ils sont tous les deux cycliques d'ordre $6$, donc isomorphes.}
\end{enumerate}
