  \texte{}
  \question{Soient $a_1,\dots,a_n \in \R$.
CNS pour que $\sum_{k=1}^n a_k \mathrm{cotan}(kx)$ ait une limite finie en 0 ?}
  \reponse{$\sum_{k = 1}^n \frac {a_k}k = 0 \Rightarrow  L = \frac 13 \sum_{k = 1}^n a_k$.}
