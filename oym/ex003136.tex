\begin{enumerate}
  \item \question{Montrer que : $M_n$ est premier $ \Rightarrow $ $n$ est premier.}
  \item \question{V{\'e}rifier que $M_{11}$ n'est pas premier.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$M_{11} = 23\times 89$.}
\end{enumerate}
