  \texte{}
  \question{Déterminer $\{M \in {\cal{M}}_n(\R) \text{ tq } M(^tMM)^2=I_n\}$.}
  \reponse{$M = (^tMM)^{-2}$ est symétrique définie positive, donc diagonalisable
en base orthonormale. En examinant la forme diagonale on trouve $M=I$.}
}
