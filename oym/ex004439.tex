\contenu{
  \texte{Par comparaison à une intégrale, donner un équivalent de :}
\begin{enumerate}
  \item \question{$\sum_{k=n+1}^{2n} \frac 1{\sqrt k}$.}
  \item \question{$\sum_{k=2}^{n} \frac 1{k\ln k}$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$2(\sqrt 2-1)\sqrt n$.}
  \item \reponse{$\ln(\ln n)$.}
\end{enumerate}
}
