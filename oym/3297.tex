\titre{Sev de $ K^3$ engendrés par deux vecteurs}
\theme{Exercices de Michel Quercia, Espaces vectoriels}
\auteur{quercia}
\date{2010/03/09}
\organisation{exo7}
\contenu{
  \texte{}
  \question{On considère les vecteurs de $ K^3$ :
$\vec a = (1, 2, 1)$,
$\vec b = (1, 3, 2)$,
$\vec c = (1, 1, 0)$,
$\vec d = (3, 8, 5)$.

Soient $F = \text{vect}(\vec a,\vec b\,)$ et
       $G = \text{vect}(\vec c,\vec d\,)$.
Comparer $F$ et $G$.}
  \reponse{$F = G$.}
}
