\contenu{
  \texte{}
  \question{Soit $f : {[a,b]} \to \R$, $K$-lipchitzienne.

Montrer que $\left| \int_{t=a}^b f(t)\,d t - \frac{b-a}n\sum_{k=1}^n
f\Bigl(a+k\frac{b-a}n\Bigr)\right| \le \frac{K(b-a)^2}{2n}$.}
  \reponse{}
