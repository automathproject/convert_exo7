\titre{Division de $(X+1)^n - X^n - 1$ par $X^2 + X + 1$}
\theme{Exercices de Michel Quercia, Division euclidienne}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Chercher le reste de la division euclidienne de
$(X+1)^n - X^n - 1$ par $X^2 + X + 1$.}
  \reponse{Substituer $j$ {\`a} $X$ $ \Rightarrow $
         $R = \begin{cases}(-1)^n - 2           &\text{ si }n \equiv 0 (\mathrm{mod}\, 3) \hfill\cr
                     ((-1)^{n+1}-1)(X+1)  &\text{ si }n \equiv 1 (\mathrm{mod}\, 3) \hfill\cr
                     ((-1)^n+1)X          &\text{ si }n \equiv 2 (\mathrm{mod}\, 3).\end{cases}$}
}
