  \texte{Soit $G$ un groupe additif et $A = \{ \text{morphismes } f : G \to G\}$.}
\begin{enumerate}
  \item \question{Montrer que $(A,+,\circ)$ est un anneau.}
  \item \question{On prend $G = \Z/n\Z$, $n \ge 2$.
    Montrer que $A$ est l'ensemble des applications
    $ G \to G,  x \mapsto {kx}$ avec $k \in G$,
    et que $A$ est isomorphe {\`a} l'anneau $\Z/n\Z$.}
\end{enumerate}
