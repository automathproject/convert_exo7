  \texte{}
  \question{Soient $m,n \in \N^*$. Chercher pgcd$( X^n-1,X^m-1)$.}
  \reponse{$n = qm+r  \Rightarrow  X^n-1 \equiv X^r-1 (\mathrm{mod}\,{X^m-1})$.
On applique la m{\'e}thode des divisions euclidiennes entre $n$ et $m$
$ \Rightarrow $ \hbox{pgcd$ {}= X^{n \wedge m} - 1$.}}
}
