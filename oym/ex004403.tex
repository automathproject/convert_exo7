  \texte{}
  \question{Quels sont les complexes $z$ tels que $\cos z \in {[-1,1]}$ ?}
  \reponse{$\cos(x+iy) = \cos x \ch y - i\sin x \sh y  \Rightarrow  \cos z \in {[-1,1]}$
          si et seulement si $z \in \R$.}
}
