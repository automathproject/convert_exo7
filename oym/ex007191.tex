  \texte{}
  \question{Soit $\leq$ une relation d'ordre sur un ensemble $E$, et $<$ la relation d'ordre strict associée, c'est-à-dire par définition : $x<y \iff x\leq y \text{ et } x\neq y$. Est-ce que le contraire de $x\leq y$ est $y<x$ ?
% non. C'est vrai si l'ordre est total}
  \reponse{Le contraire de $2|n$ n'est pas que $n$ divise strictement $2$.}
}
