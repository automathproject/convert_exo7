\titre{ $1/(\cos x + \ch a)$}
\theme{}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{Soit $a > 0$.}
\begin{enumerate}
  \item \question{Développer en série entière : $f(x) = \frac1{x+e^a}$.}
  \item \question{En déduire le développement en série de Fourier de
    $g(x) = \frac1{\cos x + \ch a}$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$g(x) = \frac1{\sh a}\bigl(e^af(e^{ix}) - e^{-ix}f(e^{-ix})\bigr)
                   = \frac1{\sh a}\Bigl(1 + 2\sum_{k=1}^\infty
                      (-1)^ke^{-ka}\cos kx \Bigr)$.}
\end{enumerate}
}
