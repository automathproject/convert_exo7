\titre{Trigonalisation de matrices}
\theme{}
\auteur{quercia}
\date{2010/03/10}
\organisation{exo7}
\contenu{
  \texte{Soit $A = \begin{pmatrix} -1 &\phantom-2 & 0 \cr 2 &2 &-3 \cr -2 &2 &1 \cr \end{pmatrix}$
et $\varphi$ l'endomorphisme de $\R^3$ canoniquement associé à $A$.}
\begin{enumerate}
  \item \question{Vérifier que $A$ n'est pas diagonalisable.}
  \item \question{Chercher deux vecteurs propres de $A$ linéairement indépendants.}
  \item \question{Compléter ces vecteurs en une base de $\R^3$.}
  \item \question{\'Ecrire la matrice de $\varphi$ dans cette base.}
  \item \question{Résoudre le système différentiel : $X' = AX$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{1 est valeur propre double, $d_1 = 1$.}
  \item \reponse{$\begin{pmatrix} 1\cr 1 \cr 1\end{pmatrix}$, $\begin{pmatrix} 2 \cr 1 \cr 2\end{pmatrix}$.}
  \item \reponse{$\begin{pmatrix} 1 \cr 0 \cr 0\end{pmatrix}$.}
  \item \reponse{$\begin{pmatrix} 1 &0 &6 \cr 0 &0 &-4 \cr 0 &0 &1 \cr\end{pmatrix}$.}
  \item \reponse{$X = \begin{pmatrix} (6\alpha t + \gamma)e^t           + 2\beta \cr
                            (6\alpha t + \gamma + 3\alpha)e^t + \beta  \cr
                            (6\alpha t + \gamma -  \alpha)e^t + 2\beta \cr \end{pmatrix}$.}
\end{enumerate}
}
