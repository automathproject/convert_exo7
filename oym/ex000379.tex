\contenu{
  \texte{Calculer pgcd$(P,Q)$ lorsque :}
\begin{enumerate}
  \item \question{$P=X^3-X^2-X-2$ et $Q=X^5-2X^4+X^2-X-2$,}
  \item \question{$P=X^4+X^3-2X+1$ et $Q=X^3+X+1$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\pgcd(X^3-X^2-X-2,X^5-2 X^4+X^2-X-2) = X-2$.}
  \item \reponse{$\pgcd(X^4+X^3-2 X+1,X^3+X+1) = 1$.}
\end{enumerate}
}
