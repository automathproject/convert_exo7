  \question{Soient $a,b \in \R$. Simplifier $\arctan a + \arctan b$.}
  \reponse{$\arctan a + \arctan b \equiv \arctan\left(\frac {a+b}{1-ab}\right)(\mathrm{mod}\, \pi)$.}
