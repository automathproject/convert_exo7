  \question{Soient $(u_{n})_{n\in \Nn} $ et $(v_{n})_{n\in \Nn}$ deux suites r\'{e}elles
telles que :
$$\lim_{n\rightarrow \infty }u_{n}+v_{n}=\lim_{n\rightarrow \infty}u_{n}v_{n}=0.$$
Montrer que
$$\lim_{n\rightarrow \infty }v_{n}=\lim_{n\rightarrow \infty }u_{n}=0.$$}
  \reponse{}
