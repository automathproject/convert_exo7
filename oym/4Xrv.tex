  \texte{}
\begin{enumerate}
  \item \question{Soient $a\in\Z$ et $b\in\N^*$.
    Montrer que : $\f{a} + \f{a+1} + \dots + \f{a+b-1} = a$.}
  \item \question{Soient $a\in\R$ et $b\in\N^*$.
    Montrer que : $\f{a} + \f{a+1} + \dots + \f{a+b-1} = [a]$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$a=bq+r \Rightarrow \sum = \underbrace{q + q + \dots + q}_{b-r}
  + \underbrace{(q+1) + \dots + (q+1)}_r = bq+r = a$.}
\end{enumerate}
