  \texte{}
  \question{Soit $f:\R \to \R$ dérivable telle que $f(0) = 0$.
Chercher $\lim_{n\to\infty}\sum_{k=1}^n f\left(\frac k{n^2}\right)$.}
  \reponse{$\frac {f'(0)}2$.}
}
