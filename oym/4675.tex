\titre{Croissance logarithmique compar{\'e}e}

\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient $(a_n)$, $(b_n)$ deux suites strictement positives telles que :
$\forall\ n \in \N,\ \frac {a_{n+1}}{a_n} \le \frac {b_{n+1}}{b_n}$.


Montrer que si $b_n \xrightarrow[n\to\infty]{} 0$, alors $a_n \xrightarrow[n\to\infty]{} 0$.}
  \reponse{}
}
