\contenu{
  \texte{Soit $\alpha \in \R$ et $f_n(x) = n^\alpha x(1-x)^n$ pour $x \in {[0,1]}$.}
\begin{enumerate}
  \item \question{Trouver la limite simple des fonctions $f_n$.}
  \item \question{Y a-t-il convergence uniforme ?}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\|f_n\|_\infty = f_n\left(\frac1{n+1}\right)
             \sim en^{\alpha-1}$.}
\end{enumerate}
}
