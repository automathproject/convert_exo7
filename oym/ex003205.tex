\titre{Division de $(X-2)^2n + (X-1)^n - 1$ par $(X-1)(X-2)$}
\theme{}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{Soit $P_n = (X-2)^{2n} + (X-1)^n - 1$.}
\begin{enumerate}
  \item \question{Montrer que $P_n$ est divisible par $X-1$ et par $X-2$.
    On note $Q_1$ et $Q_2$ les quotients correspondant.}
  \item \question{Montrer que $P_n$ est divisible par $(X-1)(X-2)$ et que le quotient est $Q_2-Q_1$.}
  \item \question{Montrer que ce quotient est {\'e}gal {\`a} :
    $$\Bigl((X-2)^{2n-2} - (X-2)^{2n-3} + \dots - (X-2) + 1\Bigr)
    + \Bigl((X-1)^{n-2} + (X-1)^{n-3} + \dots + (X-1) + 1\Bigr).$$}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Faire le produit.}
\end{enumerate}
}
