  \texte{Soit $A$ une matrice inversible donn\'ee.}
\begin{enumerate}
  \item \question{si $(A+\delta A)$ est une matrice inversible, d\'emontrer
$$\frac{\|(A+\delta A)^{-1}-A^{-1}\|}{\|(A+\delta A)^{-1}\|} \leq \mbox{cond} (A) \frac{\|\delta A\|}{\|A\|}$$}
  \item \question{D\'emontrer que
$$\frac{\|(A+\delta A)^{-1}-A^{-1}\|}{\|A^{-1}\|} \leq \mbox{cond} (A) \frac{\|\delta A\|}{\|A\|} (1+{ \cal O} (\|A\|)) $$}
\end{enumerate}
