\contenu{
  \texte{Soit $(f_n)_{n\in\N^*}$ une suite de fonctions définie par~:
$\forall\ n\in\N^*,\ \forall\ x\in{[0,1]},\ f_n(x) = \Bigl(\frac{x+x^n}2\Bigr)^n$.}
\begin{enumerate}
  \item \question{Montrer que~$(f_n)$ converge simplement vers une fonction~$\varphi$.}
  \item \question{\begin{enumerate}}
  \item \question{La convergence est-elle uniforme~?}
  \item \question{La convergence est-elle monotone~?}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$0\le f_n(x)\le x^n$ et $f_n(1) = 1$ donc  lorsque $n\to\infty$ $f_n(x)\to \begin{cases}0 & \text{ si }  x<1 \cr 1 & \text{ si }x=1.\cr\end{cases}$}
  \item \reponse{\begin{enumerate}}
  \item \reponse{Non, la continuité n'est pas conservée.}
  \item \reponse{Oui, il y a décroissance évidente.}
\end{enumerate}
