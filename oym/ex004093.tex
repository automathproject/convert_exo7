\titre{ $y'' + ay = 0$, $a > 0  \Rightarrow  y$ s'annule}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{Soit $a : \R \to {\R^{+*}}$ une fonction continue.}
\begin{enumerate}
  \item \question{Soit $y$ une solution de l'équation $ y'' + a(x)y = 0$.
    Montrer que $y$ s'annule au moins une fois sur $\R$.}
  \item \question{Soit $z$ une solution de l'équation $ z'' - a(x)z = 0$.
    Montrer que $z=0$ ou bien $z$ s'annule au plus une fois sur $\R$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Sinon, la convexité de $y$ est incorrecte.}
  \item \reponse{S'il existe $x$ tel que $z(x) = z'(x) = 0$, alors $z=0$ ce qui
             est absurde.\par
             S'il existe $x$ tel que $z(x) = 0 \ne z'(x)$, alors par convexité,
             $z$ ne peut s'annuler ailleurs.}
\end{enumerate}
}
