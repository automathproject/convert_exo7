\titre{$n^2 + 3n + 5$ mod 121}
\theme{Exercices de Michel Quercia, Congruences}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Montrer que : $\forall\ n \in \Z$, $n^2 + 3n + 5$ n'est pas divisible par 121.}
  \reponse{$n^2 + 3n + 5 \equiv (n-59)^2 - 88 (\mathrm{mod}\, {121})$.

Si $121$ divise $n^2 + 3n + 5$, alors $11\mid n-59  \Rightarrow $
contradiction.}
}
