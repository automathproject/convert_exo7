\contenu{
  \texte{}
  \question{Par comparaison à une intégrale, donner un équivalent de
$u_n = \sum_{k=1}^n \ln^2k $. La série de terme général $\frac1{u_n}$
est-elle convergente ?}
  \reponse{$u_n \sim n\ln^2 n  \Rightarrow $ CV.}
}
