\contenu{
  \texte{}
  \question{\'Etudier l'existence de solutions du système :
$$\left\{\begin{array}{l}ax-by=p\cr by-cz=q\cr cz-ax=r.\cr\end{array}\right.$$}
  \reponse{CN d'existence de solution : $p+q+r = 0$. C'est une CNS si la liste
$(a,b,c)$ comporte au plus un zéro.}
