\titre{Indépendance du polynôme minimal par rapport au corps}
\theme{Exercices de Michel Quercia, Réductions des endomorphismes}
\auteur{quercia}
\date{2010/03/10}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient $ K\subset L$ deux corps et $A\in \mathcal{M}_n(K)$.
On note $\mu_ K(A)$ et $\mu_{L}(A)$ les polynômes minimaux
de~$A$ en tant que matrice à coefficients dans $ K$ ou dans $L$.
Montrer que ces polynômes sont égaux.}
