\contenu{
  \texte{}
  \question{Montrer que pour tous entiers $m$ et $n$, le nombre
$mn(m^{60} - n^{60})$ est divisible par 56786730.}
  \reponse{$56786730 = 2 \times 3 \times 5 \times 7 \times 11 \times 13 \times 31 \times 61$.
         Pour tous ces facteurs premiers, on a $\varphi(p) \mid 60$.}
}
