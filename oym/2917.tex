\titre{Couples $(A,B)$ tels que  $A\cup B = E$}
\theme{Exercices de Michel Quercia, Ensembles finis}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $E$ un ensemble fini {\`a} $n$ {\'e}l{\'e}ments, et
${\cal E} = \{ (A,B) \in ({\cal P}(E))^2$ tq $A \cup B = E \}$.
Chercher card(${\cal E}$).}
  \reponse{$3^n$.}
}
