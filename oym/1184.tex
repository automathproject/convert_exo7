  \texte{}
  \question{Soit $F$ le sous-espace 
vectoriel de $\Rr^4$ des \'el\'ements $(x,y,z,t)$ qui satisfont :
$$\left\{
\begin{array}{rcl}
x+y+z+3t&=&0\\
2x+3y+4t&=&0\\
2x+5y-4z&=&0\\
\end{array}\right.$$
Donner une base de $F$ et sa dimension.}
  \reponse{}
