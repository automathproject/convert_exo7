  \texte{}
  \question{Chercher
$\mathrm{rg}\begin{pmatrix} 2&\phantom-4&-5&-7 \cr -1&3&1&2 \cr 1&a&-2&b \end{pmatrix}$ en fonction de
$a$ et $b$.}
  \reponse{$\mathrm{rg} = 3$ si $a \ne \frac 13$ ou $b \ne -3$, $\mathrm{rg}=2$ sinon.}
}
