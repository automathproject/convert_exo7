\contenu{
  \texte{}
  \question{Soit $A \in \mathcal{M}_n(K)$ inversible et $B$ la matrice obtenue en échangeant dans $A$
les colonnes $i$ et $j$. Montrer que $B$ est aussi inversible. Comment passe-t-on de
$A^{-1}$ à $B^{-1}$ ?}
  \reponse{Échange des lignes $i$ et $j$.}
}
