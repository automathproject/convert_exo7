  \item \question{$(2+x)y' = 2-y$.}
  \item \question{$xy'+y = \cos x$.}
  \item \question{$(1+x)y' + y = (1+x)\sin x$.}
  \item \question{$x^3y' -x^2y = 1$.}
  \item \question{$3xy' - 4y = x$.}
  \item \question{$y'+y = \sin x + 3\sin 2x$.}
  \item \question{$2x(1-x)y' + (1-2x)y = 1$.}
  \item \question{$x(x+1)y' + y = \arctan x$. \label{arctan}}
  \item \question{$x(x^2-1)y' + 2y = x\ln x - x^2$.
      

pour \ref{arctan} : \'Etudier les problèmes de raccordement.}
\end{enumerate}
