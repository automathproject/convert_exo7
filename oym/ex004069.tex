\titre{$y'' + \frac {2y'}{\tanh x} + y = 0$}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{On considère l'équation différentielle :
$(*) \iff y'' + \frac {2y'}{\tanh x} + y = 0$.}
\begin{enumerate}
  \item \question{On pose $z(x) = y'(x) + \frac {y(x)}{\tanh x}$.
     \'Ecrire l'équation différentielle (d'ordre 1) sur $z$ déduite de $(*)$.}
  \item \question{Résoudre sur $]-\infty,0[$ et $]0,+\infty[$ l'équation en $z$, puis
     $(*)$.}
  \item \question{Parmi les solutions trouvées, quelles sont celles prolongeables en 0 ?

     On note $y_0$ la solution de $(*)$ telle que $y_0(x) \to 1$ lorsque $x\to0$.}
  \item \question{Démontrer que $y_0$ est de classe $\mathcal{C}^1$ et que $\frac {y_0'(x)}{\tanh x}$
     admet une limite finie en $0$.

     En déduire que $y_0$ est de classe $\mathcal{C}^2$ sur $\R$.}
  \item \question{Est-ce que l'aire comprise entre la courbe de $y_0$
     et l'axe des abscisses est finie~?}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$z'+ \frac z{\tanh x} = 0$.}
  \item \reponse{$y = \frac {ax+b}{\sh x}$.}
\end{enumerate}
