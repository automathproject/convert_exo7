  \texte{Soit $E$ un ev de dimension finie et $f,g \in \mathcal{L}(E)$. \'Etablir :}
\begin{enumerate}
  \item \question{$\dim\mathrm{Ker}(f\circ g) \le \dim\mathrm{Ker} f + \dim\mathrm{Ker} g$.}
  \item \question{$\dim(\Im f \cap \mathrm{Ker} g) = \mathrm{rg}(f) - \mathrm{rg}(g\circ f)$.}
  \item \question{$\mathrm{rg}(f) + \mathrm{rg}(g) - \dim E \le \mathrm{rg}(f\circ g) \le \min(\mathrm{rg}(f),\mathrm{rg}(g))$.}
\end{enumerate}
