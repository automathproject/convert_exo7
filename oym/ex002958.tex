\titre{Calcul de $\tan(nx)$}

\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $n \in \N$, et $x \in \R$.
Exprimer $\tan(nx)$ en fonction de $\tan x$.}
  \reponse{$\tan(nx) = \frac {C_n^1\tan x - C_n^3\tan^3 x + \cdots}
                            {C_n^0       - C_n^2\tan^2 x + \cdots}$.}
}
