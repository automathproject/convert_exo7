\titre{$\sum nu_n$ converge}
\theme{Exercices de Michel Quercia, Séries numérique}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
  \question{On considère une suite $(u_n)_{n\ge 1}$ telle que la
série $\sum_{n\ge 1} n\, u_n$ converge. Montrer
que la série $\sum_{n\ge 1} u_n$ converge.}
  \reponse{$S_n = \sum_{k=0}^n ku_k  \Rightarrow 
\sum_{k=0}^n u_k = \sum_{k=1}^{n-1}\frac{S_k}{k(k+1)} - S_0 + \frac{S_n}n$.}
}
