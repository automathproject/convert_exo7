\titre{$e^{x^2}/(1-x)$}
\theme{}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Développer en série entière $\frac{e^x}{1-x}$ puis $\frac{e^{x^2}}{1-x}$.}
  \reponse{$\frac{e^{x^2}}{1-x} = (1+x)\frac{e^{x^2}}{1-x^2}
          = \sum_{n=0}^\infty \left(1+\frac1{1!} + \dots
          + \frac1{n!}\right)(x^{2n}+x^{2n+1})$.}
}
