  \texte{}
  \question{Soit $G$ un groupe multiplicatif et $a \in G$ d'ordre $np$ avec $n \wedge p = 1$.
\par
Montrer qu'il existe $b,c \in G$ uniques tels que
    $b$ est d'ordre $n$,
    $c$ est d'ordre $p$,
    $a = bc = cb$.
\par}
