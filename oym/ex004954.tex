\contenu{
  \texte{}
  \question{Trois figures $F_1,F_2,F_3$ se déduisent l'une de l'autre par rotations.
Montrer qu'il existe une figure $F$ dont $F_1,F_2,F_3$ se déduisent par
symétries axiales.}
  \reponse{Décomposer les rotations en symétries.}
}
