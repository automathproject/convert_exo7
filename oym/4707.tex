  \texte{}
  \question{On consid{\`e}re la suite $(u_n)$ d{\'e}finie par :
$u_0 > 0$, $u_{n+1} = \sqrt{ u_0 + \dots + u_n }$.
Montrer que $u_n \xrightarrow[n\to\infty]{} +\infty$.}
  \reponse{$u_{n+1} = \sqrt{ u_n^2 + u_n } \ge u_n$. Il n'y a pas de point fixe.}
