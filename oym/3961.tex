\titre{Dérivabilité uniforme}

\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {[a,b]} \to \R$ de classe $\mathcal{C}^1$. Démontrer que :
    $$\forall\ \varepsilon>0, \exists\ \delta>0 \text{ tq }
      \forall\ x,y \in [a,b], \text{ si } 0 < |x-y| < \delta, \text{ alors }
      \left|\frac {f(x)-f(y)}{x-y} - f'(x)\right| \le \varepsilon.$$}
  \reponse{}
