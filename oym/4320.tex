\contenu{
  \texte{}
  \question{Donner un équivalent pour $x\to+\infty$ de
$ \int_{t=0}^{+\infty}\frac{\sin t}{x^2+t^2}\,d t$.}
  \reponse{$t=ux$ puis intégration par parties $ \Rightarrow \sim \frac 1{x^2}$.}
}
