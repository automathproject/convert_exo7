  \texte{Soit $a$ et $b$ deux entiers non tous nuls et $<a,b>$ le sous-groupe de $(\Z,+)$ engendré par $\{a,b\}$.}
\begin{enumerate}
  \item \question{Montrer que $a\Z$ est inclus dans $<a,b>$.}
  \item \question{Montrer que $a\Z+b\Z$ est inclus dans $<a,b>$.}
  \item \question{Montrer que $a\Z+b\Z$ est un sous-groupe de $(\Z,+)$, contenant $\{a,b\}$.}
  \item \question{Montrer que $<a,b>$ est inclus dans $a\Z+b\Z$.}
  \item \question{Montrer que $<a,b>$ est $a\Z+b\Z$.}
  \item \question{Montrer que $<a,b>=a\Z+b\Z=\pgcd(a,b)\Z$.}
\end{enumerate}
