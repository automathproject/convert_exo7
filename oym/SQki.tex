  \texte{On consid{\`e}re le groupe $G = \Z^2$. Une {\it base\/} de $G$ est une
famille $(\alpha=(a,a'), \beta=(b,b'))$ engendrant $G$.}
\begin{enumerate}
  \item \question{\begin{enumerate}}
  \item \question{Montrer que $(\alpha,\beta)$ est une base de $G$ si et seulement
    si $\det(\alpha,\beta) = \pm 1$.}
  \item \question{Montrer que $\alpha = (a,a')$ appartient {\`a} une base de $G$ si et
    seulement si $a\wedge a' = 1$.}
