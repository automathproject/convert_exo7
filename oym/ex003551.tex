\titre{$A^3 = A + I$}
\theme{}
\auteur{quercia}
\date{2010/03/10}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $A \in \mathcal{M}_n(\R)$ telle que $A^3 = A + I$. Montrer que $\det(A) > 0$.}
  \reponse{$A$ est $\C$-diagonalisable et les valeurs propres sont $\alpha > 0$
et $\beta,\overline \beta$ avec la même multiplicité.}
}
