  \texte{Soit $(G,.)$ un groupe. On appelle conjugaison par $a\in G$, l'application $f_{a}$ de $G$ dans $G$ définie par $f_{a}(x)=a.x.a^{-1}$.}
\begin{enumerate}
  \item \question{Montrer que $f_{a}$ est un automorphisme de $G$.}
  \item \question{Soit $\Gamma=\{f_{a}:a\in G\}$. Montrer que $(\Gamma,\circ)$ est un groupe.}
  \item \question{Soit $\Phi:G\rightarrow\Gamma,a\mapsto f_{a}$. Vérifier que $\Phi$ est un morphisme.
Est-il injectif ? (indication: préciser ce morphisme lorsque $G$ est abélien).}
