\titre{$\int e^{-t}/t,d t$}
\theme{}
\auteur{quercia}
\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{On pose $f(x) =  \int_{t=x}^{+\infty} \frac{e^{-t}}t \,d t$.}
\begin{enumerate}
  \item \question{Chercher $\lim_{x\to{+\infty}} f(x)$.}
  \item \question{A l'aide d'une intégration par parties, donner un équivalent de $f(x)$
      pour $x\to{+\infty}$.}
  \item \question{Donner un équivalent de $f(x)$ pour $x\to0^+$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$e^{-x}/x$.}
  \item \reponse{$-\ln x$.}
\end{enumerate}
}
