\titre{Dérivée $n$-ème de $f(1/x)$}
\theme{Exercices de Michel Quercia, Dérivation}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f$ une fonction $n$ fois dérivable sur un intervalle $I$ ne contenant
pas $0$, et $g(x) = f\left(\frac 1x \right)$.

\'Etablir : $g^{(n)}(x) = (-1)^n\sum_{p=0}^{n-1}
           \frac {(n-1)(n-2)\dots(n-p)}{x^{2n-p}} C_n^p f^{(n-p)}\left(\frac 1x \right)$.}
  \reponse{}
}
