\contenu{
  \texte{}
  \question{On suppose que les séries $\sum a_{2n}z^n$ et $\sum a_{2n+1}z^n$ ont
pour rayons de convergence $R$ et $R'$. Déterminer le rayon de convergence de
$\sum a_nz^n$.}
  \reponse{$\min(\sqrt R,\sqrt {R'}\,)$.}
}
