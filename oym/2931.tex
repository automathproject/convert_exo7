\titre{$a+b+c=1$}
\theme{Exercices de Michel Quercia, Nombres complexes}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Trouver $a,b,c \in \mathbb{U}$ tels que $\begin{cases} a+b+c = 1\cr abc = 1.\end{cases}$}
  \reponse{$(0,a,a+b,a+b+c = 1)$ forme un losange donc l'un des nombres vaut $1$
         et les deux autres sont oppos{\'e}s $ \Rightarrow  \{a,b,c\} = \{1,i,-i\}$.}
}
