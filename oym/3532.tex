\contenu{
  \texte{}
  \question{Soit $A \in \mathcal{M}_n(K)$ inversible et $B = A^{-1}$, $C = A^2$.
Exprimer les polynômes caractéristiques $\chi_B$ et $\chi_C$ en fonction de $\chi_A$.}
  \reponse{$\chi_B(X) = \frac{(-X)^n}{\det(A)}\chi_A(1/X)$,\quad$\chi_C(X^2) = \chi_A(X)\chi_A(-X)$.}
}
