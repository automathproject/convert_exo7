  \texte{}
  \question{Montrer, pour $x > 0$ : $\sum_{n=0}^\infty \frac{(-1)^n}{n+x}
=  \int_{t=0}^1 \frac{t^{x-1}}{t+1}\,d t$.}
  \reponse{$\frac1{t+1} = \sum_{n=0}^\infty (-1)^nt^n$.}
}
