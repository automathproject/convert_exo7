  \texte{}
\begin{enumerate}
  \item \question{Soit $F$ un morphisme de groupes de $(\Z,+)$ dans lui-même.
Montrer qu'il suffit de connaître $F(1)$ pour connaître l'image de chaque entier par $F$.}
  \item \question{Existe-t-il un morphisme de groupes de $(\Z,+)$ dans lui-même tel que $F(2)=3$ ?}
  \item \question{Déterminer tous les morphismes de groupes de $(\Z,+)$ dans lui-même tels que $F(2)=~4$.}
  \item \question{Montrer qu'une condition nécessaire et suffisante sur $(a,b)\in \Z^2$ pour qu'il existe un morphisme de groupes de $(\Z,+)$ dans lui-même tel que $F(2)=a$ et $F(5)=b$ est : $5a=2b$.}
\end{enumerate}
