  \texte{}
  \question{Montrer que le produit de quatre entiers consécutifs, augmenté de 1, est un carré parfait.}
  \reponse{Soit $n$ un entier naturel.

$$n(n+1)(n+2)(n+3)+1=n^4+6n^3+11n^2+6n+1=(n^2+3n+1)^2,$$

avec $n^2+3n+1$ entier naturel.}
}
