\titre{Droites coplanaires}
\theme{}
\auteur{quercia}
\date{2010/03/17}
\organisation{exo7}
\contenu{
  \texte{Dans ${\cal E}_3$ muni d'un repère $(O,\vec i,\vec j,\vec k\,)$, on donne :
$D : \begin{cases} x - 2z = 1 \cr y - z = 2 \cr\end{cases}$
et $D' : \begin{cases} x + y + z = 1 \cr x - 2y + 2z = a.\cr\end{cases}$}
\begin{enumerate}
  \item \question{Pour quelles valeurs de $a$, $D$ et $D'$ sont-elles coplanaires ?}
  \item \question{Donner alors l'équation du plan contenant $D$ et $D'$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$a=-4$}
  \item \reponse{$x-5y+3z = -9$}
\end{enumerate}
