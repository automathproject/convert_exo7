\contenu{
  \texte{}
  \question{Soit $\mathcal{C}$ la courbe d'équation cartésienne $ x^4+y^4+x^3+y^3=2$.
En utilisant le théorème des fonctions implicites, calculer la courbure de $\mathcal{C}$
en $A = (-1,1)$.}
  \reponse{$-\frac{156}{125\sqrt2}$.}
}
