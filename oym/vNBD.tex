  \texte{Factoriser dans $\Rr[X]$ :}
\begin{enumerate}
  \item \question{$X^6 + 1$.}
  \item \question{$X^9 + X^6 + X^3 + 1$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$X^6+1 = -\left( {X}^{2}+1 \right) \left( {X}^{2}+X\sqrt {3}+1 \right) \left(-{X}^{2}+X\sqrt {3}-1 \right)$.}
  \item \reponse{$X^9+X^6+X^3+1= -\left( {X}^{2}+1 \right) \left({X}^{2}-X+1 \right) \left( {X}^{2}+X \sqrt {3}+1 \right) \left(-{X}^{2}+X\sqrt {3}-1 \right)
\left( X+1 \right)$.}
\end{enumerate}
