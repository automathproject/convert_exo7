\contenu{
  \texte{Construire les courbes représentatives des fonctions suivantes :}
\begin{enumerate}
  \item \question{$f(x) =  \int_{t=-\pi/2}^{\pi/2} |x+t|\sin t\,d t$.\par}
  \item \question{$f(x) =  \int_{t=x}^{x^2} \frac{d t}{\ln t}$.}
  \item \question{$f(x) =  \int_{t=x}^{x^2} \frac{d t}{\sqrt{t^4+t^2+1}}$.}
  \item \question{$f(x) =  \int_{t=0}^1 \frac{e^t\,d t}{t+x}$.}
  \item \question{$f(x) =  \int_{t=0}^{\pi/2} x\exp\Bigl(\frac{\sin t}x\Bigr)d t$.}
  \item \question{$f(x) =  \int_{t=0}^x \frac{\ln(1+xt)}{1+t^2}\,d t$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$-2$ de $-\infty$ à $-\frac\pi2$,\quad
             $2\sin x$ de $-\frac\pi2$ à $\frac\pi2$,\quad
             $2$ de $\frac\pi2$ à $+\infty$.}
\end{enumerate}
