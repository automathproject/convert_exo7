  \texte{}
  \question{Soit~$n\in\N^*$. Montrer l'existence de~$P_n\in\R[X]$ tel que $1+X-P_n^2$ est divisible
par~$X^n$.}
  \reponse{Prendre pour~$P_n$ la partie r{\'e}guli{\`e}re du d{\'e}veloppement limit{\'e} {\`a} l'ordre~$n$
de~$\sqrt{1+x}$.}
}
