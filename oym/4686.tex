\titre{$x_n-ax_{n-1} \to 0$}
\theme{Exercices de Michel Quercia, Suites convergentes}
\auteur{quercia}
\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $(x_n)$ une suite r{\'e}elle et $\alpha \in\, ]0,1[$.
On pose $$\left\{\begin{aligned}y_0 &=& x_0\hfill \cr y_n &=& x_n - \alpha x_{n-1}\text{ pour }n\ge 1.\end{aligned}\right.$$
\par
Montrer que : $(x_n \xrightarrow[n\to\infty]{} 0) \Leftrightarrow (y_n \xrightarrow[n\to\infty]{} 0)$.}
  \reponse{}
}
