\contenu{
  \texte{Trouver les restes des divisions euclidiennes :}
\begin{enumerate}
  \item \question{de $X^{50}$ par $X^2-3X+2$.}
  \item \question{de $\bigl(X+\sqrt3\bigr)^{17}$ par $X^2+1$.}
  \item \question{de $X^8 - 32X^2 + 48$ par $\bigl(X-\sqrt2\bigr)^3$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$(2^{50}-1)X + 2-2^{50}$.}
  \item \reponse{$2^{16}\bigl(X-\sqrt3\bigr)$.}
  \item \reponse{$192\bigl(X-\sqrt2\bigr)^2$.}
\end{enumerate}
