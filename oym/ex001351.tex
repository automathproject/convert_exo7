  \texte{Soit $G$ un groupe.}
\begin{enumerate}
  \item \question{Montrer que l'ensemble des automorphismes de $G$ muni
de la loi de composition des applications est un
groupe. Ce groupe est not\'e $\text{Aut}\,(G).$}
  \item \question{V\'erifier que l'application $\phi:G\to \text{Aut}\,(G)$
qui associe \`a $g\in G$ l'application
$\phi_g:G\to G, x\mapsto gxg^{-1}$ est un
morphisme de groupes. D\'eterminer son noyau
$Z(G),$ dit \textit{centre} de $G.$}
  \item \question{D\'eterminer $\text{Aut}\,(\Q)$ et $\text{Aut}\,(\Zz).$}
\end{enumerate}
