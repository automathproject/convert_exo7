  \texte{Soit $f:\R \to \R$ dérivable et $a \in \R$ tel que $f'(a) \ne 0$.}
\begin{enumerate}
  \item \question{Montrer qu'il existe un voisinage $V$ de $a$ tel que
    $\forall\ x \in V\setminus\{a\},\ f(x) \ne f(a)$.}
  \item \question{Si $f'$ est continue au point $a$, montrer qu'il existe un voisinage $V$ de $a$
    tel que $f_{|V}$ soit injective.}
