\titre{$f\circ g \circ f = f$ et $g \circ f\circ g = g$}
\theme{}
\auteur{quercia}
\date{2010/03/09}
\organisation{exo7}
\contenu{
  \texte{Soient $f,g \in \mathcal{L}(E)$ tels que $f\circ g \circ f = f$ et $g \circ f\circ g = g$.}
\begin{enumerate}
  \item \question{Montrer que $E = \mathrm{Ker} f \oplus \Im g$.}
  \item \question{Montrer que $f(\Im g) = \Im f$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\vec u = (\vec u - g\circ f(\vec u)) + g\circ f(\vec u)$.}
  \item \reponse{$f(\Im g) \subset \Im f$.\par
             $f = (f\circ g) \circ f  \Rightarrow  \Im f \subset \Im(f\circ g)= f(\Im g)$.}
\end{enumerate}
}
