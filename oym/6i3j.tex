\contenu{
  \texte{Soit $P = X^3 + pX + q$ de racines $a,b,c$.}
\begin{enumerate}
  \item \question{CNS pour ces racines soient aux sommets d'un carr{\'e} ?}
  \item \question{CNS pour que $a^2 + b^2 = 1 + c^2$ ?}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$50p^3 = 27q^2$.}
  \item \reponse{$a^2+b^2+c^2 = 2c^2 + 1 = -2p  \Rightarrow $ l'une des racines de l'{\'e}quation
      aux carr{\'e}s ($-Y^3 -2pY^2 -p^2Y + q^2$) doit {\^e}tre $-p - \frac 12$.
      CNS $\Leftrightarrow 2p+1 + 8q^2 = 0$.}
\end{enumerate}
