\titre{$\lim_\infty f'(x) = \lim_\infty f(x)/x$ }
\theme{Exercices de Michel Quercia, Dérivation}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f:\R \to \R$ dérivable telle que $f'(x) \to \ell$ lorsque $x\to +\infty$.
Montrer que $\frac {f(x)}x \to \ell$ lorsque $x\to +\infty$.

Chercher un contrexemple pour la réciproque.}
  \reponse{}
}
