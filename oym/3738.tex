\titre{Condition pour que deux symétries commutent}
\theme{Exercices de Michel Quercia, Transformations orthogonales}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $E$ un espace euclidien de dimension $n$.
Soient $H,K$ deux hyperplans de $E$, et $s_H$, $s_K$ les symétries associées.

Démontrer que $s_H$ et $s_K$ commutent si et seulement si $H = K$ ou
$H^\perp \subset K$.}
