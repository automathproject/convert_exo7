\contenu{
  \texte{}
  \question{Le langage ``Pascal'' ne dispose pas des fonctions $\arcsin$ et $\arccos$.
Définir $\arcsin x$ et $\arccos x$ à l'aide de la fonction $\arctan$.}
  \reponse{$\arcsin x = \arctan\left( \frac x{\sqrt{1-x^2}} \right)$.}
}
