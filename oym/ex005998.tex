  \texte{Dans une population 40\% des individus ont les yeux bruns, 25\% des
individus ont les cheveux blonds, 15\% des individus ont les yeux bruns
et les cheveux blonds.

On choisit un individu au hasard. Calculez :}
\begin{enumerate}
  \item \question{La probabilité de l'événement : si un individu a les yeux
bruns d'avoir les cheveux blonds.}
  \item \question{La probabilité de l'événement : si un individu a les cheveux
blonds d'avoir les yeux bruns.}
  \item \question{La probabilité de l'événement : si un individu a les cheveux
blonds, de ne pas avoir les yeux bruns.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Probabilité conditionnelle : si un individu a les yeux bruns d'avoir
les cheveux blonds. 
C'est $P(CB/YB)=P(YB/CB)P(CB)/P(YB)$=$P(YB\cap CB)/P(YB)=\frac{0.15}{0.4}=0.375$.}
  \item \reponse{La probabilité de l'événement : si un individu a les cheveux
blonds d'avoir les yeux bruns.
C'est $P(YB/CB)=P(YB\cap CB)/P(CB)$=$\frac{0.15}{0.25}=0.6$.}
  \item \reponse{La probabilité de l'événement : si un individu a les cheveux
blonds,de ne pas avoir les yeux bruns.
C'est $P(\text{non}YB/CB)=1-P(YB/CB)=0.4$.}
\end{enumerate}
}
