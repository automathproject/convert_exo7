\titre{$P$ {\`a} racines $x_i$ simples $ \Rightarrow \sum x_i^k/P'(x_i) = 0$}
\theme{}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{Soit $P \in {\C_n[X]}$ $(n \ge 2)$ ayant $n$ racines distinctes :
$x_1,\dots,x_n$.}
\begin{enumerate}
  \item \question{D{\'e}montrer que $\sum_{i=1}^n \frac 1{P'(x_i)} = 0$.}
  \item \question{Calculer $\sum_{i=1}^n \frac {x_i^k}{P'(x_i)}$ pour $0 \le k \le n-1$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{D{\'e}composer $1/P$ en {\'e}l{\'e}ments simples, et prendre $x \to \infty$.}
  \item \reponse{Idem avec  $X^k/P  \Rightarrow  \Sigma = \begin{cases}0 &\text{ si } 0 \le k < n-1 \cr
                                                   1 &\text{ si } k = n-1.\end{cases}$}
\end{enumerate}
}
