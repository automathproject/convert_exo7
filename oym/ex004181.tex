\titre{Partiellement dérivable $ \Rightarrow $ continue~?}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{Soit $U$ un ouvert de~$\R^2$.}
\begin{enumerate}
  \item \question{Donner un exemple de fonction $f : U \to \R$ ayant en tout point des
dérivées partielles premières, mais discontinue en au moins un point.}
  \item \question{Soit $f : U \to \R$ ayant en tout point des
dérivées partielles premières {\it bornées\/} sur~$U$. Montrer que~$f$
est continue.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$xy/(x^2+y^2)$.}
  \item \reponse{$f$ est lipschitzienne pour $\|\ \|_\infty$.}
\end{enumerate}
}
