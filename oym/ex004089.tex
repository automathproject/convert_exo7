\contenu{
  \texte{}
  \question{Soit $f : \R \to \R$ de classe $\mathcal{C}^2$ telle que :
$\forall\ x \in \R,\ f(x) + f''(x) \ge 0$.

Montrer que : $\forall\ x \in \R,\ f(x) + f(x+\pi) \ge 0$.}
  \reponse{$f(x) =  \int_{t=0}^x g(t)\sin(x-t)\,d t + \lambda\cos x + \mu\sin x$
         avec $g = f+f''$.}
}
