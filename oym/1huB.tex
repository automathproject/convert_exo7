  \texte{}
\begin{enumerate}
  \item \question{Soit $(u_n)$ une suite réelle
    telle que $\sum |u_n|$ et $\sum n|u_n|$ convergent.
    On note $v_n = \sum_{k=n}^\infty u_k$.
 \begin{enumerate}}
  \item \question{Montrer que $nv_n \to 0$ lorsque $n\to\infty$.}
  \item \question{Montrer que $\sum_{n=1}^\infty v_n = \sum_{n=1}^\infty nu_n$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\frac r{(1-r)^2}$.}
\end{enumerate}
