\contenu{
  \texte{}
  \question{Soit $E$ un espace vectoriel et $f,g \in \mathcal{L}(E)$ tels que $f\circ g = \mathrm{id}_E$.
Montrer que $g\circ f$ est une projection et déterminer ses éléments.}
  \reponse{Direction = $\mathrm{Ker} f$ et Base = $\mathrm{Im} g$.}
}
