\contenu{
  \texte{\'Etudier la convergence des séries de terme général :}
\begin{enumerate}
  \item \question{$\left(1+\frac 1n\right)^n - e$.}
  \item \question{$\ch^\alpha n - \sh^\alpha n$.}
  \item \question{$2\ln(n^3+1) - 3\ln(n^2+1)$.}
  \item \question{$\sqrt[n] {n+1} - \sqrt[n] n$.}
  \item \question{$\arccos\left(\frac {n^3+1}{n^3+2} \right)$.}
  \item \question{$\frac {a^n}{1+a^{2n}}$.}
  \item \question{$\frac{(-1)^n}{\sqrt{n^2+n}}$.}
  \item \question{$\frac{(-1)^n}{\ln n}$.}
  \item \question{$\frac{1+(-1)^n\sqrt n}n$.}
  \item \question{$\frac{2.4.6\dots(2n)}{n^n}$.}
  \item \question{$\frac{1!+2!+\dots+n!}{(n+2)!}$.}
  \item \question{$\frac{1!-2!+\dots\pm n!}{(n+1)!}$.}
  \item \question{$\frac{(-1)^n}{\ln n + \sin(2n\pi/3)}$.}
  \item \question{$\sqrt{1+\frac{(-1)^n}{\sqrt n}}-1$.}
  \item \question{$\frac{(-1)^n}{\sqrt n + (-1)^n}$.}
  \item \question{$\frac{(-1)^{[\sqrt n\,]}}n$.}
  \item \question{$\frac{(\ln n)^n}{n^{\ln n}}$.}
  \item \question{$\frac1{(\ln n)^{\ln n}}$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\sim -\frac e{2n}  \Rightarrow $ DV.}
  \item \reponse{$\sim \frac \alpha{2^{\alpha-1}}e^{n(\alpha-2)}  \Rightarrow $
             CV ssi $\alpha < 2$.}
  \item \reponse{$\sim -\frac 3{n^2}  \Rightarrow $ CV.}
  \item \reponse{$\sim \frac 1{n^2}  \Rightarrow $ CV.}
  \item \reponse{$\sim \sqrt{\frac 2{n^3}}  \Rightarrow $ CV.}
  \item \reponse{cv ssi $|a| \ne 1$.}
  \item \reponse{Série alternée $ \Rightarrow $ CV.}
  \item \reponse{Série alternée $ \Rightarrow $ CV.}
  \item \reponse{Harmonique + alternée $ \Rightarrow $ DV.}
  \item \reponse{d'Alembert $ \Rightarrow $ CV.}
  \item \reponse{$\le \frac{(n-1)(n-1)!+n!}{(n+2)!} \le \frac 2{(n+1)(n+2)}  \Rightarrow $ CV.}
  \item \reponse{$= \frac{(-1)^{n-1}}{n+1} +  O\Bigl(\frac1{n^2}\Bigr)  \Rightarrow $ CV.}
  \item \reponse{Décomposition en 3 séries alternées $ \Rightarrow $ CV.}
  \item \reponse{$=\frac{(-1)^n}{2\sqrt n} - \frac1{8n} +  O(n^{-3/2})  \Rightarrow $ DV.}
  \item \reponse{Regroupement de termes $ \Rightarrow $ DV.}
  \item \reponse{Regroupement par paquets + CSI $ \Rightarrow $ CV.}
  \item \reponse{Terme général ne tend pas vers zéro, DV.}
  \item \reponse{$= \frac1{n^{\ln\ln n}}  \Rightarrow $ CV.}
\end{enumerate}
