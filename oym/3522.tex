\contenu{
  \texte{}
  \question{Soit $E$ l'espace vectoriel des suites réelles $u=(u_n)_{n\ge 1}$
et $f$ l'endomorphisme de~$E$ défini par~:
$$(f(u))_n = \frac{u_1+2u_2+\dots+nu_n}{n^2}.$$
Quelles sont les valeurs propres de~$f$~?}
  \reponse{$1/k$, $k\ge 1$.}
}
