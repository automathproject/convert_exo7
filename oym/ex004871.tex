\titre{$1$ non valeur propre $ \Rightarrow $ un pt fixe unique}

\contenu{
  \texte{}
  \question{Soit ${\cal E}$ un espace affine de dimension finie et
$f : {\cal E} \to {\cal E}$ affine.
Montrer que $f$ admet un unique point fixe si et seulement si $1$ n'est pas
valeur propre de $\vec f$.}
