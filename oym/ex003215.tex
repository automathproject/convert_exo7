  \texte{}
  \question{Montrer que $\prod_{k=0}^{n-1} (\omega ^{2k}-2\omega ^k \cos \theta +1)=2(1-\cos (n\theta))$ avec
$\omega = e^{\frac{2i\pi}{n}}$.}
  \reponse{$\omega ^{2k}-2\omega ^k \cos \theta +1 = (\omega^k-e^{i\theta})(\omega^k-e^{-i\theta})$
et $\prod_{k=0}^{n-1}(\omega^k-x) = (-1)^n(x^n-1)$.}
}
