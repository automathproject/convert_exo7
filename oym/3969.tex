\titre{Dérivées $n$-èmes de $t^{n-1}\ln(t)$ et $t^{n-1}e^{1/t}$}
\theme{Exercices de Michel Quercia, Dérivation}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Calculer $\frac {d^n}{d t^n}\left(t^{n-1}\ln t\right)$, et
$\frac {d^n}{d t^n}\left(t^{n-1}\exp(1/t)\right)$
(essayer $n=1,2,3$).}
  \reponse{$\frac {(n-1)!}t$,
  \qquad
    $\frac {(-1)^n}{t^{n+1}}\exp(1/t)$.}
}
