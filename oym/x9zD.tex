\contenu{
  \texte{Calculer le pgcd de $P$ et $Q$ pour :}
\begin{enumerate}
  \item \question{$P=X^4 + X^3 - 3X^2 - 4X - 1$\par
    $Q=X^3 + X^2 - X - 1$}
  \item \question{$P=X^4 - 10X^2 + 1$\par
    $Q=X^4 -4X^3 + 6X^2 - 4X + 1$}
  \item \question{$P=X^5 - iX^4 + X^3 - X^2 + iX - 1$\par
    $Q=X^4 - iX^3 + 3X^2 -2iX + 2$}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$X+1$}
  \item \reponse{$1$}
  \item \reponse{$X^2-iX+1$}
\end{enumerate}
