\titre{pgcd $\times$ ppcm}
\theme{Exercices de Michel Quercia, Factorisation en nombres premiers}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient $a_1, \dots, a_n \in \N^{*}$ et $b_i = \prod_{j \ne i} a_j$.

Montrer que : \\$\text{pgcd}( a_1, \dots, a_n ) \times \text{ppcm}( b_1, \dots, b_n) =
       \text{ppcm}( a_1, \dots, a_n ) \times \text{pgcd}( b_1, \dots, b_n) =
       \prod a_i$.}
  \reponse{D{\'e}composer en facteurs premiers.}
}
