\titre{$ \int_{x=0}^{\pi/2} \frac{\ln(1+\cos x)}{\cos x}d x$}
\theme{}
\auteur{quercia}
\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{}
\begin{enumerate}
  \item \question{Montrer l'existence de $I =  \int_{x=0}^{\pi/2} \frac{\ln(1+\cos x)}{\cos x}d x$.}
  \item \question{Montrer que $I = \iint_D^{} \frac{\sin y}{1+\cos x\cos y} d xd y$ où $D = [0,\frac\pi2]^2$.}
  \item \question{En déduire la valeur de $I$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{Fubini $ \Rightarrow  I=\frac{\pi^2}8$.}
\end{enumerate}
