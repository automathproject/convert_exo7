\titre{Calcul de $(f\mid g)$}
\theme{Exercices de Michel Quercia, Séries de Fourier}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient ${f,g} : \R \to \C$ $2\pi$-périodiques, continues par morceaux.
On note $c_n(f)$ et $c_n(g)$ les coefficients de Fourier exponentiels de
$f$ et $g$.
Montrer que :  $\sum_{n=-\infty}^{+\infty} \overline{c_n(f)}c_n(g) =
        \frac1{2\pi} \int_{t=0}^{2\pi} \overline{f(t)}g(t)\,d t.$}
  \reponse{}
}
