\titre{$q$ positive $ \Rightarrow  q(x) = \|u(x)\|^2$}

\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $E$ un espace euclidien et $q$ une forme quadratique positive.
Montrer qu'il existe un endomorphisme $u$ auto-adjoint tel que :
$\forall\ \vec x \in E$, $q(\vec x) = \|u(\vec x)\|^2$.}
