\titre{$f  \mapsto f(2x)$}
\theme{Exercices de Michel Quercia, Réductions des endomorphismes}
\auteur{quercia}
\date{2010/03/10}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $E = \{ f : \R \to  \R \text{ continues}$ tq $f(x)\to 0$ lorsque $x\to\pm\infty \}$,
$\varphi : \R \to \R,  x \mapsto {2x}$ et $u : E \to  E,  f \mapsto {f \circ \varphi.}$

Montrer que $u$ n'a pas de valeurs propres
(si $u(f) = \lambda f$, étudier les limites de $f$ en $0$ ou $\pm\infty$).}
  \reponse{}
}
