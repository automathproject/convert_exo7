\titre{$FMT$ est rectangle en $F$}
\theme{Exercices de Michel Quercia, Coniques}
\auteur{quercia}
\date{2010/03/17}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit ${\cal E}$ une ellipse de foyer $F$, directrice $D$.
Soit $M \in {\cal E}$ hors de l'axe focal, et $T$ le point d'intersection
de la tangente en $M$ et de la directrice $D$.
Montrer que $FMT$ est rectangle en $F$.}
  \reponse{$M : \begin{pmatrix}X\cr Y\end{pmatrix}  \Rightarrow  T:\begin{pmatrix}a/e\cr b^2(e-X)/d\end{pmatrix}
                                \Rightarrow  \vec{FM}\cdot\vec{FT} = 0$.}
}
