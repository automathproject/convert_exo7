\titre{Signature en fonction du nombre d'orbites}
\theme{Exercices de Michel Quercia, Permutations}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $\sigma \in {\cal S}_n$. On note $c$ le nombre de cycles {\`a} supports disjoints
constituant $\sigma$, et $f$ le nombre de points fixes.

Calculer $\epsilon(\sigma)$ en fonction de $n$, $c$, et $f$.}
  \reponse{$\epsilon(\sigma) = (-1)^{n+c+f}$.}
}
