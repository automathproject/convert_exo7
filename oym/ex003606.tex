\contenu{
  \texte{}
  \question{Soit $E$ un $\C$-ev de dimension finie et $f\in\mathcal{L}(E)$.
Montrer que $f$ est diagonalisable si et seulement si
pour tout $\lambda\in\C$ on a $rg(f-\lambda\mathrm{id}) = rg(f-\lambda\mathrm{id})^2$.}
