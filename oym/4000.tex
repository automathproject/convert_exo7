\titre{Fonction convexe non dérivable sur un sous ensemble dénombrable}
\theme{Exercices de Michel Quercia, Fonctions convexes}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $(a_n)$ une suite bornée de réels.
On pose $f(x) = \sum_{n=0}^\infty \frac {|x-a_n|}{3^n}$.


Montrer que $f$ est convexe, et n'est pas dérivable aux points $a_n$.}
  \reponse{Pour $a_0$ : $\bigl| f(a_0+h) - f(a_0) - |h| \bigr| \le \frac {|h|}2$.}
}
