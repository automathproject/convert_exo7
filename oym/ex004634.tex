\contenu{
  \texte{On considère la fonction $2 \pi$-périodique sur $\R$ définie par
$f(x) = x \sin {\frac x 2}\quad \text{ si } 0 \le x < 2\pi$.}
\begin{enumerate}
  \item \question{Calculer les coefficients de Fourier de $f$.}
  \item \question{Quelle est la nature de la série de Fourier $S_f$ de $f$ ?}
  \item \question{En déduire la somme de la série
$\sum^\infty_{n=1}\, \frac{(-1)^{n-1}}{4 n^2 - 1}$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$a_n = -\frac{4}{4n^2-1}$,
         $b_n = -\frac{32n}{\pi(4n^2-1)^2}$.}
  \item \reponse{$\frac{\pi-2}4$.}
\end{enumerate}
}
