\titre{$f(a)=f(b)=0$, $f'(a)f'(b) > 0  \Rightarrow $ il existe un autre zéro}
\theme{Exercices de Michel Quercia, Dérivation}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {[a,b]} \to {\R}$ dérivable telle que
$f(a) = f(b) = 0$, et $f'(a) > 0$, $f'(b) > 0$.

Montrer qu'il existe $c \in {]a,b[}$ tel que $f(c) = 0$, et $f'(c) \le 0$.}
  \reponse{}
}
