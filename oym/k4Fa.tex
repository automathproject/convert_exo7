\contenu{
  \texte{}
\begin{enumerate}
  \item \question{Combien existe-t-il d'op{\'e}rations internes sur un ensemble {\`a} $n$ {\'e}l{\'e}ments ?}
  \item \question{Combien sont commutatives ?}
  \item \question{Combien ont un {\'e}l{\'e}ment neutre ?}
  \item \question{Combien sont commutatives et ont un {\'e}l{\'e}ment neutre ?}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$n^{n^2}$.}
  \item \reponse{$n^{n(n+1)/2}$.}
  \item \reponse{$n\times n^{(n-1)^2}$.}
  \item \reponse{$n\times n^{n(n-1)/2}$.}
\end{enumerate}
