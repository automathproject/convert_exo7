\contenu{
  \texte{}
  \question{Résoudre l'équation $2\frac{\partial f}{\partial x} + 3\frac{\partial f}{\partial y} = 4f$ avec la {\it condition aux limites\/}
: $f(t,t) = t\ (t \in \R)$.

(\'Etudier $\varphi : t  \mapsto f(a+bt,a+ct)$ avec $a,b,c$ bien choisis)}
  \reponse{$f(x,y) = (3x-2y)e^{4(x-y)}$.}
}
