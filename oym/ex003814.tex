\titre{Noyaux de $A$ et ${}^tA$}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit~$A\in\mathcal{M}_n(\R)$ telle que pour tout~$X\in\mathcal{M}_{n,1}(\R)$ on a
$\mathrm{tr}({}^tXAX)\ge 0$. Comparer les noyaux de~$A$ et ~${}^tA$.}
  \reponse{Ils sont égaux (décomposer $A$ en symétrique $+$ antisymétrique).}
}
