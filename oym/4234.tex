\titre{Intégrale de $|f|$}
\theme{Exercices de Michel Quercia, Intégrale de Riemann}
\auteur{quercia}
\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {[a,b]} \to \R$ continue. Pour $n \in \N^*$, on pose
$I_n = \sum_{k=0}^{n-1} \left|  \int_{t=a_k}^{a_{k+1}} f(t)\,d t \right|$
où $a_k = a + k\frac {b-a}n$.

Montrer que $I_n \to  \int_{t=a}^b |f(t)|\,d t$ lorsque $n\to\infty$.}
  \reponse{}
}
