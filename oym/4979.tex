\titre{Isométries transformant un triangle en un triangle donné}

\contenu{
  \texte{}
  \question{Soient $ABC$ et $A'B'C'$ deux triangles tels que
$AB = A'B'$, $AC = A'C'$, $BC = B'C'$.
Combien y a-t-il d'isométries transformant $ABC$ en $A'B'C'$ ?

Indication : si $f$ et $g$ sont deux telles isométries, alors $f\circ g^{-1}$
est une isométrie conservant $ABC$.}
  \reponse{$ABC$ équilatéral : 12, \quad $ABC$ isocèle : 4,\quad $ABC$ scalène : 2.}
}
