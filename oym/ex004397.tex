\contenu{
  \texte{}
  \question{Calculer le volume intérieur au paraboloïde d'équation $x^2+y^2 = 2pz$
    et extérieur au cône d'équation $x^2+y^2 = \lambda^2z^2$
    ($p>0$, $\lambda>0$).}
  \reponse{$V=\frac{4\pi p^3}{3\lambda^4}$.}
}
