  \texte{Etudier la continuit\'e en $(0,0)$ des fonctions suivantes :}
\begin{enumerate}
  \item \question{$$f(x,y) = \left\{
         \begin{array}{cc}
         \frac{(x+y)^4}{x^4+y^4} & \mbox{ si }(x,y) \neq (0,0) \\
         1 & \mbox{ sinon. }
         \end{array}
         \right .$$}
  \item \question{$$f(x,y) = \left\{
         \begin{array}{cc}
         \frac{|x|^3|y|^5}{(x^2+y^2)^2} & \mbox{ si }(x,y) \neq (0,0) \\
         0 & \mbox{ sinon. }
         \end{array}
         \right .$$}
  \item \question{$$f(x,y) = \left\{
         \begin{array}{cc}
         \frac{e^{xy}-1}{x^2+y^2} & \mbox{ si }(x,y) \neq (0,0) \\
         0 & \mbox{ sinon. }
         \end{array}
         \right .$$}
\end{enumerate}
