  \texte{}
  \question{Soit $A \in \mathcal{M}_n(\R)$. \'Etudier le rang de $\text{com}(A)$ en fonction du
    rang de $A$.}
  \reponse{Si $\mathrm{rg}(A) = n$, $\mathrm{rg}(\text{com}(A)) = n$.\quad
	     Si $\mathrm{rg}(A) = n-1$, $\mathrm{rg}(\text{com}(A)) = 1$.\quad
	     Si $\mathrm{rg}(A) \le n-2$, $\mathrm{rg}(\text{com}(A)) = 0$.}
}
