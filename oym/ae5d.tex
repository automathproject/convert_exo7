  \texte{Compl\'eter les pointill\'es par le connecteur logique qui s'impose : $\Leftrightarrow ,\ \Leftarrow,\ \Rightarrow.$}
\begin{enumerate}
  \item \question{$  x\in \Rr\ \ x^2=4 \ \ldots\ldots \ x=2$ ;}
  \item \question{$  z\in \Cc\ \ z=\overline{z} \ \ldots\ldots \ z\in\Rr$ ;}
  \item \question{$ x\in \Rr\ \ x=\pi \ \ldots\ldots \ e^{2ix}=1$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\Leftarrow$}
  \item \reponse{$\Leftrightarrow$}
  \item \reponse{$\Rightarrow$}
\end{enumerate}
