\begin{enumerate}
  \item \question{Montrer que le polynôme $X^3+X+1$ est irréductible dans $\mathbb{F}_2[X]$.}
  \item \question{Quelle est alors la structure de l'ensemble quotient $A=\mathbb{F}_2[X]/(X^3+X+1)$ ?}
  \item \question{Quel est le cardinal de $A$ ? Soit $\alpha$ la classe du polynôme $X$ dans ce quotient $A$. Donner la liste des éléments de $A$.}
  \item \question{Sans calculs, mais en justifiant votre réponse, dire ce que valent les quantités suivantes 
$$\alpha+\alpha, \ \ \alpha^3+\alpha+1, \ \ \ \alpha^7.$$}
  \item \question{Déterminer l'ordre multiplicatif de $\alpha$ dans $A^\times$.}
  \item \question{Etablir la table des puissances de $\alpha$.}
  \item \question{Calculer $(1+\alpha) (1+\alpha^2)$.}
  \item \question{Calculer $\alpha^3+\alpha^2$ comme puissance de $\alpha$.}
  \item \question{Calculer $(1+\alpha)^{-1}$.}
\end{enumerate}
