\titre{$1/2 + 1/3 + \dots + 1/n$ n'est pas entier}
\theme{}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $n \in \N$, $n \ge 2$.
Montrer que $x_n = 1 + \frac 12 + \frac 13 + \dots + \frac 1n$ est de la forme :
$\frac {p_n}{2q_n}$ avec $p_n,q_n \in \N^*$ et $p_n$ impair.}
  \reponse{$H_n  \Rightarrow  H_{2n}  \Rightarrow  H_{2n+1}$.}
}
