\titre{Approximation polynomiale de la racine carrée}
\theme{Exercices de Michel Quercia, Suites et séries de fonctions}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
  \question{On considère la suite $(f_n)$ de fonctions sur $[0,1]$ définie par les relations :
$f_0 = 0$, $f_{n+1}(t) = f_n(t) + \frac{t-f_n^2(t)}2$.

\'Etudier la convergence simple, uniforme, des fonctions $f_n$.}
  \reponse{$f_n(t) \to \sqrt t$ par valeurs croisantes, il y a convergence uniforme.}
}
