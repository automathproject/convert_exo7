\contenu{
  \texte{}
  \question{Pour $k\in\N$, $k\ge 2$, appliquer le théorème des accroissements finis à
$x \mapsto\ln(\ln x)$ sur $[k,k+1]$.
En déduire que la série de terme général $\frac1{k\ln k}$ est divergente.}
