\titre{$f(t^n)$}

\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Donner les deux premiers termes du DL pour $n\to\infty$ de
$I_n =  \int_{t=0}^1 \sqrt{1+t^n}\,d t$.}
  \reponse{$1+\frac1n \int_{t=0}^1\frac{d t}{\sqrt{1+t}+1} +  o\Bigl(\frac1n\Bigr)
        = 1+\frac{2\sqrt2-2+2\ln(2\sqrt2-2)}n +  o\Bigl(\frac1n\Bigr)$.}
}
