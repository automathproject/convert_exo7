\contenu{
  \texte{}
  \question{Déterminer les fonctions $f \in \mathcal{C}^0(\R,\R)$ telles que :
$\forall\ x\in \R,\ f(x) +  \int_0^x (x-t)f(t)\,dt = 1$.}
  \reponse{$f(x) = \cos x$.}
}
