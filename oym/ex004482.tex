  \texte{}
  \question{Soit $(u_n)$ une suite de complexes qui converge au sens
de Césaro vers zéro.

Étudiez la suite de terme général
$v_n=\sum_{k=0}^n \frac{u_k}{n+k+1}\cdotp$}
  \reponse{Transformation d'Abel $+$ découpage, $v_n\to0$ lorsque $n\to\infty$.}
}
