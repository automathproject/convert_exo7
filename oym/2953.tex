\contenu{
  \texte{}
  \question{Soit $a \in \R$. R{\'e}soudre :
$$\begin{cases}\cos(a) + \cos(a+x) + \cos(a+y) = 0\cr
        \sin(a) + \sin(a+x) + \sin(a+y) = 0.\end{cases}$$}
  \reponse{$x \equiv -y \equiv \pm \frac {2\pi}3 (\mathrm{mod}\, {2\pi})$.}
}
