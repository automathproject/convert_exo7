\titre{$(u,x,f(x))$ liée}

\date{2010/03/09}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $E$ un ev de dimension supérieure ou égale à 3 et
$\vec u \in E \setminus \{\vec 0\}$.
Trouver tous les endomorphismes $f \in \mathcal{L}(E)$ tels que :
$\forall\ \vec x \in E,\text{ la famille }(\vec u, \vec x, f(\vec x))
 \text{ est liée.}$}
  \reponse{$f(\vec x) = \alpha\vec x + \beta(\vec x)\vec u$, $\beta \in E^*$.}
}
