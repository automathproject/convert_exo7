\contenu{
  \texte{}
  \question{Soit $f : {[0,+\infty[} \to \R^+$ une fonction continue telle que
$ \int_{t=0}^{+\infty} f(t) \,d t$ converge.

Montrer que $ \int_{t=0}^{+\infty} f(t) \,d t =
     \int_{u=0}^{+\infty} f\Bigl(\Bigl| u - \frac1u \Bigr|\Bigr) \,d u$.}
  \reponse{}
