\titre{Nombre de couples $(a,b)$ tel que $a^2+b^2 \le n$}
\theme{}
\auteur{quercia}
\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Pour $n \in \N$ on pose
$E_n = \{ (p,q)\in \N^2 \text{ tel que } p^2+q^2 \le n \}$ et $C_n = \mathrm{Card}\,(E_n)$.

Interpréter $C_n$ comme une aire et donner un équivalent de $C_n$ lorsque
$n\to\infty$.}
  \reponse{}
