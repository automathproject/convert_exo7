\titre{$\mathrm{tr}(AB) \le \mathrm{tr}(A)\mathrm{tr}(B)$}
\theme{}

\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient $A,B \in \mathcal{M}_n(\R)$ symétriques positives.
Montrer que $0 \le \mathrm{tr}(AB) \le \mathrm{tr}(A)\mathrm{tr}(B)$.}
  \reponse{Se ramener au cas où $A$ est diagonale.}
}
