\titre{$\int f + \int f^{-1}$}
\theme{Exercices de Michel Quercia, Intégrale de Riemann}
\auteur{quercia}
\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {[a,b]} \to {[c,d]}$ continue, bijective, strictement croissante.

Calculer $ \int_{t=a}^b f(t)\,d t +  \int _{u=c}^d f^{-1}(u)\,d u$
(faire un dessin, et commencer par le cas où $f$ est de classe $\mathcal{C}^1$).}
  \reponse{}
}
