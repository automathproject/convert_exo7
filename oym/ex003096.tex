\titre{$3^x = 2^y + 1$}
\theme{}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
\begin{enumerate}
  \item \question{Soient $x,y\in \N$, $y \ge 3$.
    Montrer par r{\'e}\-cur\-rence sur $y$ que :
    $3^x \equiv 1 (\mathrm{mod}\, {2^y}) \iff 2^{y-2}\mid x$.}
  \item \question{Trouver tous les couples d'entiers $x,y \in \N$ tels que
    $3^x = 2^y + 1$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$x=1,y=1$ ou $x=2,y=3$.}
\end{enumerate}
}
