\contenu{
  \texte{}
  \question{D\'{e}terminer les $f\in C^{2}({\Rr},{\Rr}) $
telles que :
$$\forall x\in \Rr,f''(x)+f(-x)=x\cos x. $$}
  \reponse{R\'{e}ponse : $\frac{1}{2}\left( -x\cos x+\sin x\right) +\lambda
\cos x+\mu \sinh x.$}
}
