\contenu{
  \texte{}
  \question{Soit $\sum u_n$ une série à termes positifs convergente.

\'Etudier la convergence de la série de terme général
$v_n = \frac1{n^2}\sum_{k=1}^n ku_k$.}
  \reponse{$\sum_{k=1}^n v_k = \sum_{k=1}^n ku_k\sum_{p=k}^n \frac 1{p^2} \le
         \sum_{k=1}^n \frac{ku_k}{k-1}  \Rightarrow $ CV.}
}
