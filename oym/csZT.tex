    du plan réalisant le minimum de :
  \begin{enumerate}}
  \item \question{$MA^2 + MB^2 + MC^2$.}
  \item \question{$MA   + MB   + MC  $.}
  \item \question{$MA\times MB\times MC$.}
\end{enumerate}
