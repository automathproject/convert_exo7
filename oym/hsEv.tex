\Delta~:&[a,b]&\rightarrow&\Rr\\
 &x&\mapsto&\left|
 \begin{array}{ccc}
 f(a)&f(b)&f(x)\\
 g(a)&g(b)&g(x)\\
 1&1&1
 \end{array}
 \right|
\end{array}$.}

  \item \question{Montrer que $\Delta$ est continue sur $[a,b]$, dérivable sur $]a,b[$ et calculer sa dérivée.}
