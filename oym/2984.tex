\titre{Ordre de $ab$}

\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soient $a,b$ deux {\'e}l{\'e}ments d'un groupe multiplicatif $G$ tels que :
$\begin{cases}a \text{ est d'ordre } \alpha\cr
        b \text{ est d'ordre } \beta\cr
        \alpha \wedge \beta = 1 \cr
        ab = ba.\end{cases}$

D{\'e}terminer l'ordre de $ab$.}
  \reponse{}
}
