  \texte{}
  \question{Soit~$E$ un ev de dimension finie et $(u_n)$ une suite d'endomorphismes
diagonalisables convergeant vers~$u\in\mathcal{L}(E)$.

$u$ est-il diagonalisable~?}
  \reponse{Non. Prendre $\mathrm{mat}(u_n) = \left(\begin{smallmatrix}0&1\cr0&1/n\cr\end{smallmatrix}\right)$.}
