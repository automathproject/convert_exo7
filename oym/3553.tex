\contenu{
  \texte{}
  \question{Soit $A \in \mathcal{M}_n(\C)$ telle que $A^n = I$ et $(I,A,\dots,A^{n-1})$ est libre. Montrer qu'alors on a $\mathrm{tr}(A) = 0$.}
  \reponse{$A$ est diagonalisable et a $n$ valeurs propres distinctes,
	     sinon il existerait un polynôme annulateur de degré inférieur ou
	     égal à $n-1$. Ces racines sont les $n$ racines $n$-èmes de 1 et
	     leur somme est nulle.}
}
