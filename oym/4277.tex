\contenu{
  \texte{}
  \question{On pose $I_n =  \int_{t=0}^{\pi/2} \cos(2nt)\ln(\sin t)\,d t$ ($n\in\N^*$).
Calculer $2nI_n - (2n+2)I_{n+1}$ et en déduire $I_n$ en fonction de~$n$.}
  \reponse{$2nI_n - (2n+2)I_{n+1}=0  \Rightarrow  I_n = -\frac\pi{4n}$.}
}
