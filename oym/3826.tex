\contenu{
  \texte{}
  \question{Soit $E$ un espace préhilbertien réel et $B$ une boule ouverte de $E$ ne
contenant pas $\vec 0$.
Montrer qu'il existe une forme linéaire $f \in E^*$ telle que :
$\forall\ \vec x \in B,\ f(\vec x) > 0$.}
