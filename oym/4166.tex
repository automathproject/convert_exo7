\contenu{
  \texte{}
  \question{Montrer que l'égalité $2e^{x+y} + y - x = 0$ définit $y = \varphi(x)$ au voisinage
    de $(1,-1)$. Calculer $\varphi'(1)$ et $\varphi''(1)$.}
  \reponse{$\varphi'(1) = -\frac13$, $\varphi''(1) = -\frac8{27}$.}
}
