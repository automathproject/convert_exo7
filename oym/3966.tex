\titre{Dérivée $n$-ème de $(x^3+2x^2-5)e^{-x}$}
\theme{Exercices de Michel Quercia, Dérivation}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Calculer la dérivée $n^{\text{ème}}$ de $x \mapsto (x^3+2x^2-5)e^{-x}$.}
  \reponse{$(-1)^ne^{-x}\bigl( x^3 + (2-3n)x^2 + (3n^2-7n)x + (-n^3+5n^2-4n-5)\bigr)$.}
}
