\contenu{
  \texte{}
  \question{Soit $A\in\mathcal{M}_n(\R)$ à coefficients strictement positifs.
Déterminer la signature de la forme quadratique sur~$\R^n$ définie par~:
$q(x_1,\dots,x_n) = \sum_{i,j} a_{i,j}(x_i-x_j)^2$.}
  \reponse{$(n-1,0)$.}
}
