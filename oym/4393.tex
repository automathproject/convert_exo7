  \texte{}
  \question{Soit $\cal P$ le plan rapporté au repère $(O,\vec i,\vec j)$.
    Calculer l'aire du domaine délimité par la courbe d'équation
    $x^{2/3}+y^{2/3} = a^{2/3}$.}
  \reponse{Formule de Green : ${\cal A} = \frac{3\pi a^2}8$.}
