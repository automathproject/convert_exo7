\contenu{
  \texte{}
  \question{Soit $E$ un espace vectoriel norm{\'e} et $A \subset E$ ferm{\'e}.
Montrer que si $\text{Fr}(A)$ est connexe, alors $A$ est connexe.}
  \reponse{Soient $F_1$, $F_2$ ferm{\'e}s non vides disjoints tels que
$F_1 \cup F_2 = A$ : Alors $\text{Fr}(A) = \text{Fr}(F_1) \cup \text{Fr}(F_2)$.}
}
