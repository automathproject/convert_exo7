\contenu{
  \texte{}
  \question{Soit $A\in\mathcal{M}_n(\C)$. Montrer que $A$ est nilpotente si et seulement
si pour tout $k\in\N^*$ on a $\mathrm{tr}(A^k)=0$.}
  \reponse{$0$ est valeur propre, se placer dans un hyperplan stable et récurer.}
}
