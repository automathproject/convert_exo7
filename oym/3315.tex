\contenu{
  \texte{}
  \question{Est-il vrai que $\mathcal{L}({E\times F})$ et $\mathcal{L}(E)\times \mathcal{L}(F)$ sont isomorphes~?
($E$ et $F$ espaces vectoriels de dimensions finies).}
  \reponse{Non, ils n'ont pas même dimension si $E\ne\{\vec 0\}$ ou
$F\ne\{\vec 0\}$.}
