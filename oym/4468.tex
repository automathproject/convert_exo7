\contenu{
  \texte{}
  \question{Soient $a \in {[0,1[}$. \'Ecrire $\frac1{(1-a)^2}$ comme produit de deux
séries. En déduire la somme de la série $\sum_{k=0}^\infty ka^k$.
Calculer par la même méthode $\sum_{k=0}^\infty k^2a^k$.}
  \reponse{$\frac{a}{(1-a)^2}$ et $\frac{a+a^2}{(1-a)^3}$.}
}
