\begin{enumerate}
  \item \question{$2x^3-x^2+x+1 = 0$.}
  \item \question{$6x^5+11x^4-x^3+5x-6 = 0$.}

\end{enumerate}
\begin{enumerate}
  \item \reponse{$x=-\frac 12$.}
  \item \reponse{$x=\frac 23$.}
  \item \reponse{Pas de solution.}
\end{enumerate}
