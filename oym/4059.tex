\titre{$y'' + k^2y = \sum_{n=1}^\infty \frac{\cos nx}{n^2}$}
\theme{Exercices de Michel Quercia, \'Equations différentielles linéaires (I)}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $k \in \R$.
Résoudre $y'' + k^2y = \sum_{n=1}^\infty \frac{\cos nx}{n^2}$.}
  \reponse{$k \notin\Z$ : $y = \sum_{k=1}^\infty \frac{\cos nx}{n^2(k^2-n^2)} + a\cos kx + b\sin kx$.\par
         $k\in\Z$ : remplacer $\frac{\cos kx}{k^2(k^2-k^2)}$ par $\frac{x\sin kx}{2k^3}$.}
}
