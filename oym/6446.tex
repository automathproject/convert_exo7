  \texte{}
  \question{Montrer que si un ensemble $E$ est muni d'une opération binaire qui vérifie les propriétés 
\begin{itemize}
\item $(P_1)$ $(ab)c=a(cb)$ ;
\item $(P_2)$ il existe $e\in E$ tel que $ea=a,\; \forall a$;
\item $(P_3)$ pour tout $a\in E$ il existe $b\in E$ tel que $ba=e$;
\end{itemize}
alors c'est un groupe abélien.}
