\titre{$f((x+y)/2) \le (f(x) + f(y))/2$}

\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f : {[a,b]} \to {\R}$ continue telle que :
$\forall\ x,y \in {[a,b]},\ f\left(\frac {x+y}2\right) \le \frac {f(x) + f(y)}2$.

Montrer que $f$ est convexe.}
