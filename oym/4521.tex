\titre{$f_n \circ g_n$}
\theme{Exercices de Michel Quercia, Suites et séries de fonctions}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit ${f_n} : {[a,b]} \to {[c,d]}$ et ${g_n} : {[c,d]} \to \R$
des fonctions continues convergeant uniformément vers les fonctions $f$ et $g$.
Montrer que $g_n\circ f_n$ converge uniformément vers $g\circ f$.}
  \reponse{$|g_n(f_n(x))-g(f(x))| \le |g_n(f_n(x))-g(f_n(x))|
                                  + |g(f_n(x))-g(f(x))|$
         et $g$ est uniformément continue.}
}
