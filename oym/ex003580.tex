\titre{$A^3$ est semblable à $A^4$}
\theme{}
\auteur{quercia}
\date{2010/03/10}
\organisation{exo7}
\contenu{
  \texte{Quelles sont les matrices $A \in \mathcal{M}_{3}(\C)$ telles que $A^3$ est semblable à $A^4$ ?
On étudiera séparément les cas :}
\begin{enumerate}
  \item \question{$A$ a 3 valeurs propres distinctes.}
  \item \question{$A$ a 2 valeurs propres distinctes}
  \item \question{$A$ a une seule valeur propre.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$A \sim \text{diag}(1,\alpha,\alpha^{-1})$ où $\alpha$ est une
             racine primitive $7^{\text{ème}}$ de $1$,\par
             $A \sim \text{diag}(\alpha,\alpha^{10},\alpha^{-11})$ où $\alpha$
             est une racine primitive $37^{\text{ème}}$ de $1$.}
  \item \reponse{pas de solution.}
  \item \reponse{vp${} = 0$ ou $1$.}
\end{enumerate}
}
