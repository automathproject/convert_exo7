  \texte{}
  \question{Soient $(u_{n})_{n\in \Nn} $ et $(v_{n})_{n\in \Nn}$ deux suites \`{a} valeurs
dans $[0,1]$ telles que :
$$\lim_{n\rightarrow \infty }u_{n}v_{n}=1.$$
Montrer que :
$$\lim_{n\rightarrow \infty }u_{n}=\lim_{n\rightarrow \infty }v_{n}=1.$$}
  \reponse{}
