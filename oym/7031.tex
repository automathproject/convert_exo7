\contenu{
  \texte{}
  \question{Résoudre dans $\R_+^*$ le système suivant:
\[
\begin{cases}
xyz=1\\
xy^2z^4=2\\
xy^3z^9=3
\end{cases}
\]}
  \reponse{On trouve une unique solution $\left(\frac{3}{8},\frac{16}{3\sqrt 3},\frac{1}{\sqrt 3}\right)$.}
}
