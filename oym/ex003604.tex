\titre{$u$ diagonalisable $ \Rightarrow  Ker(u-\lambda\mathrm{id}) + Im(u-\lambda\mathrm{id})$ est directe}
\theme{}
\auteur{quercia}
\date{2010/03/10}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $E$ un $ K$-ev de dimension finie et $u \in \mathcal{L}(E)$ diagonalisable.
Pour $\lambda \in \text{spec}(u)$, on note $E_\lambda = \mathrm{Ker}(u-\lambda\mathrm{id})$
et $F_\lambda = \Im(u-\lambda\mathrm{id})$.
Montrer que $E_\lambda \oplus F_\lambda = E$.}
  \reponse{}
}
