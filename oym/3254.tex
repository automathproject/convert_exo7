  \texte{}
  \question{Soient $a,b,c$ les racines de $X^3 + pX + q$, $q \ne 0$. Calculer :
$\sum_{\sigma \in S_3} \left( \frac {\sigma(a)}{\sigma(b)}
                      + \frac {\sigma(b)}{\sigma(c)}
                      + \frac {\sigma(c)}{\sigma(a)} \right)$.}
  \reponse{$3\Bigl(\frac ab + \frac ac + \frac ba + \frac bc + \frac ca
          + \frac cb\big) = -9$.}
}
