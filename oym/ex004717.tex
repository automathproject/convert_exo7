\titre{Partie {\`a} un seul point d'accumulation}
\theme{}
\auteur{quercia}
\date{2010/03/16}
\organisation{exo7}
\contenu{
  \texte{Soit $A$ une partie born{\'e}e de $\R$ ayant un seul point d'accumulation, $a$.}
\begin{enumerate}
  \item \question{Montrer que $A$ est d{\'e}nombrable.}
  \item \question{On num{\'e}rote les {\'e}l{\'e}ments de $A$ d'une mani{\`e}re quelconque :
    $A = \{ x_1, x_2, \dots, x_n, \dots \}$.

    Montrer que $x_n \xrightarrow[n\to\infty]{} a$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$A \setminus ]a-\frac 1n, a+\frac 1n[$ est fini.}
\end{enumerate}
}
