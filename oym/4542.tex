\titre{Convergence de $f^{(n)}$}

\contenu{
  \texte{}
  \question{Soit $f\in \mathcal{C}^{\infty}(\R)$. On définit la suite $(f_n)_{n\in\N^*}$ par
$f_n = {f}^{(n)}$ (dérivée $n$-ème).
On suppose que ${(f_n)}_{n\ge 1}$ converge uniformément vers $\varphi$.
Que peut-on dire de $\varphi$ ?}
