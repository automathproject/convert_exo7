  \texte{}
  \question{Existe-t-il une norme $N$ sur $\mathcal{M}_n(\Rr)$ ($n\geqslant2$) telle que $\forall(A,B)\in(\mathcal{M}_n(\Rr))^2$, $N(AB)=N(A)N(B)$.}
  \reponse{Non, car si $A=E_{1,1}\neq0$ et $B=E_{2,2}\neq0$ alors $AB=0$ puis $N(AB)<N(A)N(B)$.}
