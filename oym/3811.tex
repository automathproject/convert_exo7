  \texte{}
  \question{Soit $f \in \mathcal{L}(E)$ tel que $f\circ f^* = f^*\circ f$ et $f^2 = -$id.
Montrer que $f$ est orthogonal.}
  \reponse{$h = f\circ f^*$ : $h^2 = \text{id}$ et $h \ge 0  \Rightarrow  h = \text{id}$.}
}
