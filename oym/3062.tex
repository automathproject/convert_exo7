  \texte{}
  \question{Soit $A \subset \R$ v{\'e}rifiant :
$$\begin{cases}\forall\ x \in \R,\ \exists\ a,b\in A \text{ tq }a < x < b\cr
        \forall\ a,b \in A,\ \frac{a+b}2 \in A.\cr\end{cases}$$
Montrer que $A$ est dense dans $\R$.}
