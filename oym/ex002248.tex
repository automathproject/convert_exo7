  \texte{Soit $p>2$ un nombre premier.}
\begin{enumerate}
  \item \question{Soit $a$ premier \`a $p$. Supposons que la congruence
$x^2\equiv a\mod p$ poss\`ede une solution.
Montrer que 
$a^{(p-1)/2}\equiv 1 \mod p$.}
  \item \question{La congruence $x^2\equiv -1 \mod p$ a une solution 
ssi $p\equiv 1 \mod 4$.}
\end{enumerate}
