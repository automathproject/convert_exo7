\contenu{
  \texte{Résoudre dans $\C$ :}
\begin{enumerate}
  \item \question{$\cos z = 2$.}
  \item \question{$\ch z = -1$.}
  \item \question{$\sin z + \sin jz + \sin j^2z = 0$.}
  \item \question{$8\cos z + 4i\sin z = 7 + 5i$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$z \equiv \pm i\ln(2+\sqrt3\,) (\mathrm{mod}\,{2\pi})$.}
  \item \reponse{$z \equiv i\pi(\mathrm{mod}\,{2i\pi})$.}
  \item \reponse{$z \equiv 0 (\mathrm{mod}\, {2\pi})$ ou
             $z \equiv 0 (\mathrm{mod}\, {2j\pi})$ ou
             $z \equiv 0 (\mathrm{mod}\, {2j^2\pi})$.}
  \item \reponse{$\Leftrightarrow 6e^{2iz} -(7+5i)e^{iz} + 2 = 0 \Leftrightarrow
               \begin{cases} e^{iz} = 1+i\hfill:
                       & z \equiv \pi/4 - i\ln\sqrt2      (\mathrm{mod}\,{2\pi})\cr
                       e^{iz} = (1-i)/6  :
                       & z \equiv -\pi/4 - i\ln(\sqrt2/6) (\mathrm{mod}\,{2\pi}).\cr\end{cases}$}
\end{enumerate}
