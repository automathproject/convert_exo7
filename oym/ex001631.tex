  \texte{}
  \question{Soit $P_{0}$ un polynôme de $\R_{n}[X]$, et $f$ l'application
suivante :
$$
 f: \begin{array}{rcl}
      \R_{n}[X] & \rightarrow  & \R_{n}[X] \\[1ex]
      P         & \mapsto & R=
      \text{ reste de la division euclidienne de $P$ par $P_{0}$}
    \end{array}
$$
A l'aide d'un polynôme annulateur de $f$, montrer que $f$ est
diagonalisable.}
  \reponse{}
}
