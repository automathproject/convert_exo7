  \texte{}
\begin{enumerate}
  \item \question{$y-xy' = \sqrt{x^2+y^2}$.}
  \item \question{$y' = \frac{x-y}{x+y}$.}
  \item \question{$(x^2+y^2)y' = 2xy$.}
  \item \question{$(x+y)y' = 2x-y$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$y = \frac{1-\lambda^2x^2}{2\lambda}$, $\lambda > 0$.}
  \item \reponse{$y = -x\pm\sqrt{2x^2-\lambda}$ ou $y = x(-1\pm\sqrt2)$.}
  \item \reponse{$y = \frac {-1 \pm \sqrt{1+4\lambda^2x^2}}{2\lambda}$
               ou $y = \pm x$ ou $y=0$.}
  \item \reponse{$y = -x \pm \sqrt{\lambda+3x^2}$ et $y = x(-1\pm\sqrt3)$.}
\end{enumerate}
