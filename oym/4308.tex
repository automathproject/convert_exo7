\titre{$f$ et $f''$ de carrés sommables}
\theme{Exercices de Michel Quercia, Intégrale généralisée}
\auteur{quercia}
\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $f:\R \to \R$ de classe $\mathcal{C}^2$ telle que $ \int_{t=0}^{+\infty}f^2(t)\,d t$
et  $ \int_{t=0}^{+\infty}f''^2(t)\,d t$ convergent.
Montrer que $ \int_{t=0}^{+\infty}f'^2(t)\,d t$ converge.}
  \reponse{}
}
