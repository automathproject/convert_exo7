\titre{$e^z = z$}
\theme{Exercices de Michel Quercia, Fonction exponentielle complexe}
\auteur{quercia}
\date{2010/03/12}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Montrer qu'il existe une infinité de complexes $z$ tels que $e^z = z$
(on calculera $x$ en fonction de $y$, et on étudiera l'équation obtenue).}
  \reponse{$e^{x+iy} = x+iy \Leftrightarrow \begin{cases}x = y/\tan y \cr e^{-y/\tan y}
         = \sin y/y.\cr\end{cases}$
         Au voisinage de $2k\pi^+$, $e^{-y/\tan y} < \sin y/y$ (point plat)
         et au voisinage de $(2k+1)\pi^-$, $e^{-y/\tan y} > \sin y/y$
         (limite infinie).}
}
