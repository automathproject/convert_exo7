\contenu{
  \texte{Soit $A = \{ a + bi \text{ tq } a,b \in \Z\}$.}
\begin{enumerate}
  \item \question{Montrer que $A$ est un sous-anneau de $\C$. Quels sont les {\'e}l{\'e}ments
    inversibles ?}
  \item \question{Soient $u,v \in A$ avec $v \ne 0$. Montrer qu'il existe $q,r \in A$
    tels que $u = qv + r$ et $|r| < |v|$.
    A-t-on unicit{\'e} ?}
  \item \question{Montrer que $A$ est principal.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\pm1,\pm i$.}
  \item \reponse{On a : $1+i = 0\times2 + (1+i) = 1\times2 + (i-1)$.}
\end{enumerate}
