\titre{$(p-1)(p-2)...(p-n)/n!$}
\theme{Exercices de Michel Quercia, Congruences}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $p \in \N^*$ premier et $n \in \N^*$, $n < p$.
Montrer que $\frac {(p-1)(p-2) \dots (p-n)}{n!} - (-1)^n$ est  un entier
divisible par~$p$.}
