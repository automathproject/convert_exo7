\titre{$\sum e^{in\theta}/2^n$, Ensi P 91}
\theme{}
\auteur{quercia}
\date{2010/03/14}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Calculer $\sum_{n=1}^\infty \frac{\sin n\theta}{2^n}$ et
	     $\sum_{n=1}^\infty \frac{\cos n\theta}{n2^n}$.}
  \reponse{$\sum_{n=1}^\infty \frac{\sin n\theta}{2^n} =
		\frac{2\sin\theta}{5-4\cos\theta}$,\quad
	     $\sum_{n=1}^\infty \frac{\cos n\theta}{n2^n} =
		\ln2 - \frac12\ln(5-4\cos\theta)$.}
}
