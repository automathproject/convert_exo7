  \texte{}
  \question{Soit la suite de terme général~:
$u_n = (n^4+n^2)^{1/4}-P(n)^{1/3}$ où $P$ est un polynôme.
A quelle condition sur~$P$ la série $\sum u_n$ converge-t-elle~?}
  \reponse{$P(n) = n^3+\frac34n+C$.}
}
