\begin{enumerate}
  \item \question{R{\'e}soudre $\begin{cases} u_{n+2} = \frac 12(u_{n+1} + u_n)\cr
                      u_0 = a,\ u_1 = b.\cr\end{cases}$}
  \item \question{Si $a=0$, trouver $\lim u_n$.}
  \item \question{R{\'e}soudre : $v_{n+2} = \sqrt{v_{n+1} v_n}$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$u_n = \frac 13((a+2b) + 2(a-b)(-\frac12)^n)$.}
  \item \reponse{$v_n = \lambda\times\mu^{(-\frac12)^n}$.}
\end{enumerate}
