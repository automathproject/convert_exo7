\titre{$y'' + ay' = 0$ $a$ croissante positive $=> y$ est bornée}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $a : \R \to \R$ une fonction de classe $\mathcal{C}^1$ croissante strictement positive et
$y$ une solution de l'équation : $y'' + a(t)y = 0$.
Montrer que $y$ est bornée au voisinage de $+\infty$
(on étudiera $z = y^2 + y'^2/a$).}
  \reponse{}
}
