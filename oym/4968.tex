\contenu{
  \texte{}
  \question{Dans un \emph{rond} on donne le plan  $P : x + 2y + 3z = 4$.
Déterminer l'expression analytique de la projection orthogonale sur $P$.}
  \reponse{$\begin{cases} 14x' = 13x - 2y - 3z + 4\cr
                  14y' = -2x + 10y - 6z + 8\cr
                  14z' = -3x - 6y + 5z +12.\cr\end{cases}$}
}
