\titre{$1-ab$ et $1-ba$}
\theme{Exercices de Michel Quercia, Anneaux}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{}
  \question{Soit $A$ un anneau et $a,b\in A$. Montrer que $1-ab\in A^* \Leftrightarrow 1-ba \in A^*$.}
  \reponse{Si $(1-ab)c = 1 = c(1-ab)$ alors $abc  = c-1 = cab$
donc $babca = bca-ba = bcaba$ soit $ba(1+bca) = bca = (1+bca)ba$ donc
$1+bca$ est inverse de $1-ba$.}
}
