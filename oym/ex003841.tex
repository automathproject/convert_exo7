\titre{Produit scalaire}
\theme{}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{Soit $E = \mathcal{C}([a,b],\R)$ et $ u : {[a,b]} \to \R$ une fonction continue
par morceaux. On pose pour $f,g \in E$ :
$(f\mid g) =  \int_{t=a}^b u(t)f(t)g(t)\,d t$.}
\begin{enumerate}
  \item \question{A quelle condition sur $u$ définit-on ainsi un produit scalaire ?}
  \item \question{Soient $u,v$ deux fonctions convenables. A quelle condition les normes
    associées sont-elles équivalentes~?}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$u \ge 0$ et $u^{-1}(0)$ est d'intérieur vide.}
  \item \reponse{Il existe $\alpha,\beta > 0$ tels que $\alpha u \le v \le \beta u$.}
\end{enumerate}
