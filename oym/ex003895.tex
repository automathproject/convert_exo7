  \texte{}
  \question{Soit $x \in \R^*$. Vérifier que $\tanh x = 2\coth 2x - \coth x$.
En déduire la convergence et la somme de la série de terme général
$\frac1{2^n}\tanh\left(\frac{x}{2^n}\right)$.}
  \reponse{$2\coth 2x - \frac 1x$.}
}
