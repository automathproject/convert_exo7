  \question{Quelle est la nature de la série de terme général
$\ln\Bigl(1+\frac{(-1)^n}{\sqrt{n(n+1)}}\Bigr)$~?}
  \reponse{$=\frac{(-1)^n}{\sqrt{n(n+1)}}+O\Bigl(\frac1{n^2}\Bigr)  \Rightarrow $ converge.}
