\contenu{
  \texte{Résoudre dans $\Rr$ puis dans $[0,2\pi]$ les équations suivantes~:}
\begin{enumerate}
  \item \question{$\sin x=\frac{1}{2}$,}
  \item \question{$\sin x=-\frac{1}{\sqrt{2}}$,}
  \item \question{$\tan x=-1$,}
  \item \question{$\tan x=\frac{1}{\sqrt{3}}$,}
  \item \question{$\cos x=\frac{\sqrt{3}}{2}$,}
  \item \question{$\cos x=-\frac{1}{\sqrt{2}}$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$\sin x=\frac{1}{2} \Leftrightarrow x\in\left(\frac{\pi}{6}+2\pi\Zz\right)\cup\left(\frac{5\pi}{6}+2\pi\Zz\right)$. De plus,
$\mathcal{S}_{[0,2\pi]}=\left\{\frac{\pi}{6},\frac{5\pi}{6}\right\}$.}
  \item \reponse{$\sin x=-\frac{1}{\sqrt{2}}\Leftrightarrow x\in\left(-\frac{\pi}{4}+2\pi\Zz\right)\cup\left(-\frac{3\pi}{4}+2\pi\Zz\right)$. De plus,
$\mathcal{S}_{[0,2\pi]}=\left\{-\frac{\pi}{4},-\frac{3\pi}{4}\right\}$.}
  \item \reponse{$\tan x=-1\Leftrightarrow x\in-\frac{\pi}{4}+\pi\Zz$. De plus, $\mathcal{S}_{[0,\pi]}=\left\{\frac{3\pi}{4}\right\}$.}
  \item \reponse{$\tan x=\frac{1}{\sqrt{3}}\Leftrightarrow x\in\frac{\pi}{6}+\pi\Zz$. De plus,
$\mathcal{S}_{[0,\pi]}=\left\{\frac{\pi}{6}\right\}$.}
  \item \reponse{$\cos x=\frac{\sqrt{3}}{2}\Leftrightarrow x\in\left(-\frac{\pi}{6}+\pi\Zz\right)\cup\left(\frac{\pi}{6}+\pi\Zz\right)$. De
plus, $\mathcal{S}_{[0,2\pi]}=\left\{\frac{\pi}{6},\frac{11\pi}{6}\right\}$.}
  \item \reponse{$\cos x=-\frac{1}{\sqrt{2}}\Leftrightarrow x\in\left(-\frac{3\pi}{4}+\pi\Zz\right)\cup\left(\frac{3\pi}{4}+\pi\Zz\right)$. De
plus, $\mathcal{S}_{[0,2\pi]}=\left\{\frac{3\pi}{4},\frac{5\pi}{4}\right\}$.}
\end{enumerate}
}
