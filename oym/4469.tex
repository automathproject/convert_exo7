\contenu{
  \texte{}
  \question{Pour $n \in \N$ on note $T_n$ le nombre de manières de décomposer
$n$ francs avec des pièces de 1, 2, 5 et 10 francs ($T_0 = 1$).
Montrer que :
$$\forall\ x \in {[0,1[},\
\sum_{k=0}^\infty T_kx^k = \frac1{(1-x)(1-x^2)(1-x^5)(1-x^{10})}.$$}
  \reponse{}
