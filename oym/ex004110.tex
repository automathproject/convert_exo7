  \texte{}
\begin{enumerate}
  \item \question{$y' = y(1+y)$.}
  \item \question{$y' = \sin x\cos y$.}
  \item \question{$2yy'\sqrt x = \sqrt{y^2-1}$.}
  \item \question{$1+xy' = e^y$, condition initiale : $y(1)=1$.}
  \item \question{$y' = \sqrt{|y|}$ : étudier les problèmes de raccordements.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$y = -1 + \frac1{1-\lambda e^x}$ ou $y = -1$.}
  \item \reponse{$y \equiv 2\arctan(\lambda e^{-\cos x}) - \frac\pi2 (\mathrm{mod}\,\pi)$.}
  \item \reponse{$y = \pm\sqrt{1+(\sqrt x+\lambda)^2}$ ou $y = \pm1$.}
  \item \reponse{$y = -\ln\bigl(1-x(1-1/e)\bigr)$.}
  \item \reponse{$y=\left(\lambda+\frac x2\right)\left|\lambda+\frac x2\right|$ ou $y = 0$.}
\end{enumerate}
