\contenu{
  \texte{}
  \question{Soit $u_n$ une suite réelle telle que pour tout $n$ on ait 
$u_n^5+nu_n-1=0$. Trouver un développement asymptotique à deux termes de $u_n$.}
  \reponse{$u_n = \frac1n - \frac1{n^6} +  o\Bigl(\frac1{n^6}\Bigr)$.}
}
