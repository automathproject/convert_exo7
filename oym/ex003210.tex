\titre{Degr{\'e} minimal dans la formule de B{\'e}zout}
\theme{}
\auteur{quercia}
\date{2010/03/08}
\organisation{exo7}
\contenu{
  \texte{Soient $P,Q \in { K[X]}$ non nuls et $D = \text{pgcd}(P,Q)$.}
\begin{enumerate}
  \item \question{D{\'e}montrer qu'il existe $U,V \in { K[X]}$ uniques tels que :
      $\begin{cases} UP + VQ = D \cr
               \deg  U < \deg  Q - \deg  D \hfill\cr
               \deg  V < \deg  P - \deg  D .\end{cases}$}
  \item \question{Montrer que la m{\'e}thode des divisions euclidiennes fournit $U$ et $V$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{R{\'e}currence.}
\end{enumerate}
}
