  \texte{}
  \question{Effectuer la division selon les puissances
croissantes de :
$$X^{4}+X^{3}-2X+1\text{ par }X^{2}+X+1 \text{ \`a l'ordre }2.$$}
  \reponse{$X^{4}+X^{3}-2X+1 = (X^{2}+X+1)(2X^{2}-3X+1)+X^{3}(2-X)$.}
}
