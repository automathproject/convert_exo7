  \texte{A l'aide de d{\'e}composition en {\'e}l{\'e}ments simples, calculer :}
\begin{enumerate}
  \item \question{$\sum_{n=1}^\infty \frac 1{n(n + 1)}$.}
  \item \question{$\sum_{n=1}^\infty \frac 1{n(n + 1)(n+2)}$.}
  \item \question{$\sum_{n=1}^\infty \frac n{n^4 + n^2 + 1}$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$1$.}
  \item \reponse{$1/4$.}
  \item \reponse{$1/2$.}
\end{enumerate}
