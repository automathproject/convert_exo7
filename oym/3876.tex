\titre{\'Etude de $x(2+\sin(1/x))$}
\theme{Exercices de Michel Quercia, Fonctions monotones}
\auteur{quercia}
\date{2010/03/11}
\organisation{exo7}
\contenu{
  \texte{}
  \question{On pose : 
$$\begin{cases} f(x) = |x|\left({2+\sin\left(\frac1x\right)}\right) &\text { si } x \ne 0,\cr
                   f(0) = 0.\cr\end{cases}$$

Montrer que $f$ est continue, minimale en $0$, mais pour tout $\varepsilon > 0$,
$f_{|[0,\varepsilon]}$ n'est pas monotone.}
  \reponse{}
}
