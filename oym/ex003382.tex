\titre{Groupes de matrices}
\theme{}
\auteur{quercia}
\date{2010/03/09}
\organisation{exo7}
\contenu{
  \texte{Soit ${\cal G} \subset \mathcal{M}_n(K)$ tel que pour la multiplication, $\cal G$ soit un
groupe. On note $J$ l'élément neutre et pour $M \in \cal G$,
$\phi_M$ l'endomorphisme de $ K^n$ canoniquement associé à $M$.}
\begin{enumerate}
  \item \question{Montrer que $\phi_J$ est une projection.}
  \item \question{Montrer que : $\forall\ M \in \cal G$, ${\phi_M}_{|\mathrm{Ker}\phi_J} = 0$ et
    ${\phi_M}_{|\Im\phi_J}$ est un isomorphisme de $\Im\phi_J$.}
  \item \question{En déduire que $\cal G$ est isomorphe à un groupe $GL_k( K)$.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$J^2 = J$.}
  \item \reponse{$JM = MJ$.}
  \item \reponse{$k$ = $\mathrm{rg} J$.}
\end{enumerate}
}
