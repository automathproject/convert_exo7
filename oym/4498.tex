\contenu{
  \texte{}
  \question{Calculer les sommes suivantes : $A = \sum_{(p,q)\in(\N^*)^2}\frac1{p^2q^2}$,
$B = \sum_{(p,q)\in(\N^*)^2 ; p|q}\frac1{p^2q^2}$ et
$C = \sum_{(p,q)\in(\N^*)^2 ; p\wedge q=1}\frac1{p^2q^2}$.}
  \reponse{$A=\zeta(2)^2$, $B=\zeta(2)\zeta(4)$, $C=A/\zeta(4) = 5/2$.}
}
