\contenu{
  \texte{}
\begin{enumerate}
  \item \question{Construire la courbe $\mathcal{C}$ d'équations paramétriques :
    $x=3t-t^3$,\quad $y=3t^2$.}
  \item \question{Chercher les équations paramétriques des développantes de $\mathcal{C}$.}
  \item \question{Tracer la développante qui rencontre $\mathcal{C}$ à l'origine.}
\end{enumerate}
\begin{enumerate}
  \item \reponse{$x=\frac{4t^3 + a(1-t^2)}{1+t^2}$,\quad
             $y=\frac{t^4-3t^2+2at}{1+t^2}$.}
\end{enumerate}
