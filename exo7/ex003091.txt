\exercice{3091, quercia, 2010/03/08}

\enonce[Petit th{\'e}or{\`e}me de Fermat]
Soit $p \in \N$ premier. Montrer que pour $1 \le k \le p-1$, $p$ divise $C_p^k$.

En d{\'e}duire que $\forall\ n \in \Z$, $n^p \equiv n (\mathrm{mod}\, p)$.
\label{Fermat}
\finenonce

\noindication

\nocorrection

\finexercice
