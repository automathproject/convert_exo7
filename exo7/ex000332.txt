\exercice{332, cousquer, 2003/10/01}

\enonce
\begin{enumerate}
\item Trouver tous les points à coordonnées entières de la droite 
de l'espace d'équations 
$\left\{
\begin{array}{rcl}
    4x-2y-z-5 & = & 0  \\
    x+3y-4z-7 & = & 0
\end{array}\right.$.
\item Même question avec la droite 
$\left\{
\begin{array}{rcl}
    x+3y-5z -5 & = & 0  \\
    4x-2y+z+13 & = & 0
\end{array}\right.$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
