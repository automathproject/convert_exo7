\exercice{3445, quercia, 2010/03/10}
\enonce[$\prod a_{i\sigma(i)} =$ cste]

Soit $A = (a_{ij}) \in \mathcal{M}_n(K)$ telle qu'il existe $a \ne 0$ tel que :
$\forall\ \sigma \in S_n,\ \prod_{i=1}^n a_{i\sigma(i)} = a$. Montrer que $\mathrm{rg} A = 1$.
\finenonce

\noindication

\nocorrection

\finexercice
