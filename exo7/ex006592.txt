\exercice{6592, hueb, 2011/10/16}
\enonce Localiser les
singularités de chacune des fonctions suivantes et les
caractériser.
\begin{enumerate}
\item $\frac {z^2}{(z+1)^3}$,
\item $\frac {2z^3-z+1}{(z-4)^2 (z-i)(z-1+2i)}$,
\item $\frac {\sin mz}{z^2 +2z +2}$,
\item $\frac {1-\cos z}z$,
\item $e^{-\frac 1{(z-1)^2}}$,
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
