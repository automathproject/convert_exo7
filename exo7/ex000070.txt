\exercice{70, bodin, 1998/09/01}

\enonce
Soient $z,z' \in \Cc$ tels que $\text{Arg}(z)-\text{Arg}(z')=\frac{\pi}{2}$.
\begin{enumerate}
    \item Montrer que $z\overline{z'}+\overline{z}z' = 0$.
    \item Montrer que $|z+z'|^2=|z-z'|^2=|z|^2+|z'|^2$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
