\exercice{6522, drutu, 2011/10/16}
\enonce
 Résoudre les congruences simultanées suivantes :
\begin{enumerate}
\item $x\equiv 2$ (mod 7), $x\equiv 3$ (mod 6) ;

\item $3x\equiv 2$ (mod 5), $x\equiv 6$ (mod 7), $x\equiv 1$ (mod 6). 
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
