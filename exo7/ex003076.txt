\exercice{3076, quercia, 2010/03/08}

\enonce[${\cal A}_n$ est engendr{\'e} par les 3-cycles]
\begin{enumerate}
  \item Calculer $(a\,b\,c) \circ (b\,c\,d)$.
  \item Montrer que le sous-groupe altern{\'e} ${\cal A}_n$ est engendr{\'e}
    par les 3-cycles ($n \ge 3$).
\end{enumerate}
\finenonce

\noindication

\correction
\begin{enumerate}
 \item $(a\ b)\circ(c\ d)$.
 \item  
\end{enumerate}

\fincorrection

\finexercice
