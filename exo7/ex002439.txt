\exercice{2439, matexo1, 2002/02/01}

\enonce
Soit $G$ l'ensemble des matrices de la forme
$$
\left(
\begin{array}{cc}
1 & x \\ 0 & 1
\end{array}
\right), \qquad x \in \R.$$
Montrer que c'est un groupe multiplicatif isomorphe au groupe
additif r\'eel.
\finenonce

\noindication
\nocorrection

\finexercice
