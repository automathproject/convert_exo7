\exercice{3288, quercia, 2010/03/08}

\enonce[$F(X+1) - F(X) = \dots$]
Trouver les fractions $F \in {\R(X)}$ telles que :
$F(X+1) - F(X) = \frac {X+3}{X(X-1)(X+1)}$.
\finenonce

\noindication

\correction
$F(X+1) - F(X) = \frac 2{X-1} - \frac 3X + \frac 1{X+1}
          \Rightarrow 
         F(X) = \frac 1X -\frac 2{X-1} + \text{cste}$.
\fincorrection
\finexercice
