\exercice{144, ridde, 1999/11/01}
\enonce
Soient $A, B \subset E$. R\'esoudre les \'equations \`a l'inconnue $X \subset E$
\begin{enumerate}
\item $A \cup X = B$.
\item $A \cap X = B$.
\end{enumerate}

\finenonce 

\noindication

\correction
\begin{enumerate}
  \item $B\setminus A \subset X \subset B$.
  \item $B \subset X \subset B \cup \complement{A}$.
\end{enumerate}
\fincorrection
\finexercice
