\exercice{6580, hueb, 2011/10/16}
\enonce
\begin{enumerate}
\item Montrer que $u(x,y) = e^{-x}(x \sin y - y \cos y)$ est
  harmonique.
\item Déterminer $v$ telle que $f = u +iv$ soit holomorphe.
\item Ecrire la fonction $f$ trouvée ci-dessus comme fonction d'une
  variable complexe.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
