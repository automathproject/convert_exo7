\exercice{40, cousquer, 2003/10/01}

\enonce
Pour tout nombre complexe $Z$, on pose $P(Z)=Z^4-1$.
\begin{enumerate}
\item  Factoriser $P(Z)$ et en déduire les solutions dans $\mathbb{C}$ de 
l'équation $P(Z)=0$.
\item  Déduire de 1. les solutions de l'équation d'inconnue $z$ :
$$((2z+1)/(z-1))^4=1$$
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
