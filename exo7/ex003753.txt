\exercice{3753, quercia, 2010/03/11}
\enonce[$A{}^t\!A + A + {}^t\!A = 0$]

\begin{enumerate}
  \item Trouver les matrices $A \in \mathcal{M}_n(\R)$ telles que :
    $A{}^t\!A + A + {}^t\!A = 0$.
    
  \item Montrer que pour une telle matrice, $|\det A| \le 2^n$.
    

\end{enumerate}
\finenonce

\noindication

\correction
\begin{enumerate}
  \item $A = P-I$, $P \in {\cal O}(n)$.
  \item Hadamard.
\end{enumerate}
 \fincorrection

\finexercice
