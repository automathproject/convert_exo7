\exercice{905, cousquer, 2003/10/01}

\enonce
 Soit $E$ un espace vectoriel sur $\mathbb{R}$ et $x,y,z,t$ une famille libre
d'éléments de $E$, les familles suivantes sont-elles libres~?
\begin{enumerate}
\item  $x,2y,z$.
\item  $x,z$.
\item  $x,2x+t,t$.
\item  $3x+z, z, y+z$.
\item  $2x+y,x-3y,t, y-x$. 
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
