\exercice{3073, quercia, 2010/03/08}

\enonce[$(\sqrt3+1)^{2n+1} - (\sqrt3-1)^{2n+1}$]
On note $x_n = (\sqrt3 + 1)^{2n+1}$, $y_n = (\sqrt3 - 1)^{2n+1}$, et
$z_n = [x_n]$.
\begin{enumerate}
  \item   Montrer que $z_n = x_n - y_n$.
  \item   En d{\'e}duire que $2^{n+1}$ divise $z_n$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
