\exercice{3514, quercia, 2010/03/10}
\enonce[$P(2-X)$]

Déterminer les éléments propres de l'endomorphisme
$\theta : { K[X]} \to { K[X]}, P \mapsto {P(2-X).}$
\finenonce

\noindication

\correction
$\lambda = 1\phantom-$ : $P = Q\bigl((X-1)^2\bigr)$. \par
         $\lambda =-1$          : $P = (X-1)Q\bigl((X-1)^2\bigr)$.
\fincorrection

\finexercice
