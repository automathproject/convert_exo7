\exercice{3287, quercia, 2010/03/08}

\enonce[$F'(X)/F(X) = \dots$]
Soient $a_1, \dots, a_n \in  K$ distincts et
$\alpha_1, \dots, \alpha_n \in  K$.
Existe-t-il $F \in { K(X)}$ telle que :
$\frac {F'(X)}{F(X)} = \sum_{k=1}^n \frac {\alpha_k}{X-a_k}$ ?
\finenonce

\noindication

\nocorrection

\finexercice
