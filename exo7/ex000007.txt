\enonce

\begin{enumerate}
\item $(3+2i)(1-3i)$.

nombre complexe de module $3$ et d'argument $-5\pi /6$.

\item Quotient du nombre complexe de module $2$ et d'argument $\pi /3$ par le
nombre complexe de module $3$ et d'argument $-5\pi /6$.
\end{enumerate}

\finenonce
