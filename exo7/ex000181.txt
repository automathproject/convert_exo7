\exercice{181, gourio, 2001/09/01}

\enonce
Montrer que :
$$\forall n\geq 2,u_{n}=\sum\limits_{k=1}^{n}\frac{1}{k}\notin \Nn.$$
\emph{Indication} : montrer que
$$\forall n\geq 2,\exists (p_{n},q_{n})\in (\Nn^{*})^{2},u_{n}=\frac{2p_{n}+1}{2q_{n}}.$$
\finenonce

\noindication

\nocorrection

\finexercice
