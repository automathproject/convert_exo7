\exercice{906, cousquer, 2003/10/01}

\enonce
 Dans $\mathbb{R}^4$, comparer les sous-espaces $F$ et $G$ suivants~:
\begin{eqnarray*}
    F & = & \mbox{Vect}\{(1,0,1,1), (-1,-2,3,-1), (-5,-3,1,-5)\}\\
    G & = & \mbox{Vect}\{(-1,-1,1,-1), (4,1,2,4)\}
\end{eqnarray*}
\finenonce

\noindication

\nocorrection

\finexercice
