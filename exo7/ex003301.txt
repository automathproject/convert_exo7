\exercice{3301, quercia, 2010/03/09}
\enonce[\'Etude de liberté]

\'Etudier la liberté des familles suivantes :
\begin{enumerate}
  \item $E=\{  {\text{fcts}} : \R    \to    \R \}$, ${\cal F}=(\sin, \cos)$.
  \item $E=\{ {\text{fcts}} : {\R^{+*}} \to \R \}$, ${\cal F}=(f_a : x \longmapsto x^a),\ a\in\R$.
  \item $E=\{ {\text{fcts}} : \R      \to  \R \}$, ${\cal F}=(f_a : x \longmapsto |x-a|),\ a\in\R$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
