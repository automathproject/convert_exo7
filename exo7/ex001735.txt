\exercice{1735, gourio, 2001/09/01}

\enonce
Soit $(a_{n})_{n\in \Nn}\in \left( {\Rr}^{+*}\right) ^{\Nn},
u_{n}=\sum\limits_{k=1}^{n}a_{k}^{2},v_{n}=\sum\limits_{k=1}^{n}\frac{a_{k}}{k}.$
Montrer que si $(u_{n})_{n}$ converge alors $(v_{n})_{n}$
aussi.
\finenonce

\noindication

\nocorrection

\finexercice
