\exercice{6613, hueb, 2011/10/16}
\enonce
\begin{enumerate}
\item Trouver la transformée de Fourier en cosinus de $f(x) =
  e^{-m|x|},\ m>0$.
\item Utiliser le résultat de (1) pour montrer que, pour $p>0,\ 
  \beta >0$,
  $$\int_{0}^{\infty} \frac{\cos pv} {v^2 + \beta^2} dv = \frac \pi {2
    \beta} e^{-p \beta}.$$
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
