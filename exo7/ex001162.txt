\exercice{1162, gourio, 2001/09/01}

\enonce
Soit $ A=\left( a_{i,j}\right) _{(i,j)\in \{1,...,n\}^{2}}\in M_{n}({\Rr})$
telle que :
\begin{eqnarray*}
\forall i &\in &\{1,...,n\},\sum_{j=1}^{n}a_{i,j}\leq 1, \\
\forall (i,j) &\in &\{1,...,n\}^{2},a_{i,j}\in [0,1[.
\end{eqnarray*}
Montrer que $ \left| \det(A)\right| <1.$
\finenonce

\noindication

\nocorrection

\finexercice
