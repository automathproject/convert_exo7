\exercice{3219, quercia, 2010/03/08}

\enonce[$X^8 + X^4 + 1$ divise $X^{8n} + pX^{4n} + q$] Donner une CNS
sur $p,q \in \C$ pour que $X^8 + X^4 + 1$ divise $X^{8n} + pX^{4n} +
q$ ($n \in \N^*$ fix{\'e}).
 \finenonce

\noindication

\correction
$\iff X^2 + X + 1\ |\ X^{2n} + pX^n + q \iff j^{2n} + pj^n + q = 0$.
\fincorrection

\finexercice
