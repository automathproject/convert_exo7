\exercice{3272, quercia, 2010/03/08}

\enonce[Ensemble image d'une fonction rationelle]
\label{Imf}
Soit $F \in {\C(X)}$. {\'E}tudier $F( \C\setminus \{\text{p{\^o}les}\} )$.
\finenonce

\noindication

\correction
$F = \frac PQ$.
Si $P = \lambda Q$ : $\Im F = \{\lambda\}$.\par
Si $P = \lambda Q + \mu$ : $\Im F = \C \setminus \{\lambda\}$.\par
Sinon, $\Im F = \C$.
\fincorrection
\finexercice
