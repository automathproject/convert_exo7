\enonce[$(1+1/t)^t$]

\begin{enumerate}
  \item Montrer que : $\forall\ t > 1,\ \left(1+\frac 1t\right)^t < e < \left(1+\frac 1{t-1}\right)^t$.
    
  \item Montrer que : $\forall\ x,y>0,\ \left(1+\frac xy\right)^y < e^x < \left(1+\frac xy\right)^{x+y}$.
    

\end{enumerate}
