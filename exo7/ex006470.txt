\exercice{6470, drutu, 2011/10/16}
\enonce
Démontrer les propriétés suivantes 

\begin{enumerate}
\item $(\C / \R ,+) \simeq (\R,+)$ ;
\item $\C^*/\R_+^*\simeq \mathbb{S}^1$ ;
\item $\C^*/\mathbb{S}^1\simeq \R_+^*\simeq \R^*/\{ \pm 1 \}$ ;
\item $\R^*/ \R_+^*\simeq \{ \pm 1 \}$ ;
\item $(\C / \Z ,+)\simeq \C^*$ ;
\item $\C^* /\R^* \simeq \mathbb{S}^1/\{ \pm 1 \}$.
\end{enumerate} 
\finenonce

\noindication

\nocorrection

\finexercice
