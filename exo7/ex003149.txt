\exercice{3149, quercia, 2010/03/08}

\enonce[{\'E}quations alg{\'e}briques]
D{\'e}terminer $x \in \Q$ sachant que :
\begin{enumerate}
  \item $2x^3-x^2+x+1 = 0$.        
  \item $6x^5+11x^4-x^3+5x-6 = 0$. 
  \item $2x^3-x-4 = 0$.   
\end{enumerate}
\finenonce

\noindication

\correction
\begin{enumerate}
  \item $x=-\frac 12$.
  \item $x=\frac 23$.
  \item Pas de solution.
\end{enumerate}
\fincorrection
\finexercice
