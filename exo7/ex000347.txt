\exercice{347, ridde, 1999/11/01}

\enonce
Montrer que les nombres suivants ne sont pas premiers :
\begin{enumerate}
\item $n^4-20n^2 + 4$ pour $n \in \Nn$.
\item $\frac14 (n^3 + (n + 2)^3)$ pour $n \geq 2$.
\item $a^4 + 4b^4$ pour $a, b \geq 2$.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
