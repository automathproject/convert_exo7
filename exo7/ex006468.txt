\exercice{6468, drutu, 2011/10/16}
\enonce
\begin{enumerate}
\item Montrer que tout sous-groupe d'un groupe cyclique est cyclique.

\item Soit $d\in \N^*$ un diviseur de $n\in \N^*$. Montrer qu'il existe un unique sous-groupe d'ordre $d$ dans $\Z /n\Z $. 
\end{enumerate} 
\finenonce

\noindication

\nocorrection

\finexercice
