\exercice{391, cousquer, 2003/10/01}

\enonce
 Soit $A\in \mathbb{C}[X]$ et $B\in\mathbb{C}[X]$.
\begin{enumerate}
\item A-t-on $\mbox{pgcd}(A,B)=1\iff\mbox{pgcd}(A+B,AB)=1$~?
\item A-t-on $\mbox{pgcd}(A,B)=\mbox{pgcd}(A+B,AB)$~?
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
