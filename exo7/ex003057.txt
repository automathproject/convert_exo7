\enonce[R{\'e}currence d'ordre 2]
On note $a_n = 25^n + 2^{3n+4}$.
\begin{enumerate}
  \item  Trouver $a,b\in\Z$ tels que :
     $\forall\ n\in\N,\ a_{n+2} = a.a_{n+1} + b.a_n$.

  \item  En d{\'e}duire que : $\forall\ n \in \N,\ a_n$ est divisible par $17$.
\end{enumerate}
\finenonce

\noindication

\correction

 \item  $a=33$, $b=-200$.
 \item
\end{enumerate}
\fincorrection
