\exercice{704, cousquer, 2003/10/01}

\enonce

applications suivantes~:
$$ f : x \mapsto x\vert x\vert, \qquad g : x \mapsto  \frac{x}{1+\vert x\vert}, \qquad  h :  \mapsto  \frac{1}{1+\vert x\vert}.$$
  
\finenonce

\noindication

\nocorrection

\finexercice
