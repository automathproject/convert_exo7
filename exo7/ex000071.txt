\exercice{71, bodin, 1998/09/01}

\enonce
\begin{enumerate}
    \item D\'eterminer l'ensemble des points $M$ du plan complexe,
d'affixe $z$ tels que : $\overline{z}(z-1)=z^2(\overline{z}-1)$.
    \item D\'eterminer l'ensemble des points $M$ du plan complexe,
d'affixe $z$ tels que les images de $1$, $z$, $1+z^2$ soient align\'ees.
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
