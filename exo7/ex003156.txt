\exercice{3156, quercia, 2010/03/08}

\enonce[$(\Z/2^n\Z)^*$]
\begin{enumerate}
  \item Montrer que pour tout entier $a$ impair et tout $n\ge 3$ : $a^{2^{n-2}} \equiv 1
    (\mathrm{mod}\, {2^n})$.
  \item Le groupe $\bigl(\Z/2^n\Z\bigr)^*$ est-il cyclique ?
\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
