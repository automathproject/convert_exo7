\exercice{2959, quercia, 2010/03/08}

\enonce[$z = (1+ia)/(1-ia)$]
Soit $z \in \mathbb{U}$. Peut-on trouver $a \in \R$ tel que $z = \frac {1+ia}{1-ia}$ ?
\finenonce

\noindication

\correction
$z = e^{i\theta}  \Rightarrow  a = \tan\frac\theta2$
         pour $\theta \not \equiv \pi (\mathrm{mod}\,{2\pi})$.
\fincorrection
\finexercice
