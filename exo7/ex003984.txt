\exercice{3984, quercia, 2010/03/11}
\enonce[Fonction convexe bornée]

\begin{enumerate}
  \item Soit $f : {\R^+} \to {\R}$ convexe et bornée. Montrer que $f$ est décroissante.
  \item Soit $f : {\R} \to {\R}$ convexe et bornée. Montrer que $f$ est constante.

\end{enumerate}
\finenonce

\noindication

\nocorrection

\finexercice
